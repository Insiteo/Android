	/**
	 * Inner class used to handle touch events
	 */
	private class ZoneTouch {
		private Point 	mReferencePoint = new Point();
		private boolean mTouchDown		= false;
		private boolean	mTouchOn		= true;


		private void setTouchable(boolean aOn) {
			mTouchOn = aOn;
		}

		public ETouchObjectResult onTouchDown(Touch aTouch) {	

			if(mTouchOn) {
				// if i am under the touch, notify observers
				if (isTouchInsideZone(aTouch)) {
					mReferencePoint.set(aTouch.getPosition(0).x, aTouch.getPosition(0).y);
					mTouchDown = true;
					GfxZone.this.show();
					return ETouchObjectResult.RESULT_CONSUME;
				}
			}
			return ETouchObjectResult.RESULT_NOTHING;
		}

		public ETouchObjectResult onTouchMove(Touch aTouch) {
			// if no zone selected, no need to check this event
			if (mTouchOn && mTouchDown) {
				if (aTouch.isMultiTouch()) {
					mTouchDown = false;
					GfxZone.this.hide();
				}
				else {
					Point currentPoint = aTouch.getPosition();
					//Log.d("zone", "MOVE " + Zone.this.getZoneId());
					//if the touch position change too much, deselect the zone
					int dx = Math.abs(currentPoint.x - mReferencePoint.x);
					int dy = Math.abs(currentPoint.y - mReferencePoint.y);
					if ((dx > CommonConstants.CLICKABLE_ZONE_SIZE) || (dy > CommonConstants.CLICKABLE_ZONE_SIZE)) {
						onTouchUp(aTouch);
						//notify observers or "release" won't work
						return ETouchObjectResult.RESULT_NOTIFY;
					}
				}
			}
			return ETouchObjectResult.RESULT_NOTHING;
		}

		public ETouchObjectResult onTouchUp(Touch aTouch) {
			if (mTouchOn && mTouchDown) {
				mTouchDown = false;
				GfxZone.this.hide();
				//Log.d("zone", "HIDE " + Zone.this.getZoneId());
				return ETouchObjectResult.RESULT_NOTIFY_CONSUME;
			}
			return ETouchObjectResult.RESULT_NOTHING;
		}

		private boolean isTouchInsideZone(Touch aTouch){
			boolean result = false;

			if (mRenderMode == ERenderMode.RENDER_MODE_3D) {
				if (mGraphics != null && mGraphics.get3DZone() != null) {
					SimpleVector pos = aTouch.getTouchedRays().get(0).getPos();
					SimpleVector dir = aTouch.getTouchedRays().get(0).getDir();
					if (mGraphics.get3DZone().rayIntersectsAABB(pos, dir, true) != Object3D.RAY_MISSES_BOX) {
						result = (mGraphics.get3DZone().wasTargetOfLastCollision());//true;
					}
				}
			} else {
				PointF p = new PointF((float)((aTouch.getPosition(0).x - mLastOffset.x) / mLastRatio), 
						(float)((aTouch.getPosition(0).y - mLastOffset.y) / mLastRatio));

				result = Topology.isPointInsidePoly(getPolygon(), p);
			}

			return result;
		}
	}

		private void draw(Canvas aCanvas, Point aOffset, double aRatio, float aAngle) {
			if(mIsVisible) {
				// draw zone
				List<PointF> poly = GfxZone.this.getPolygon();
				if (poly.size() > 2) {

				}

		private void drawPolygon(Canvas canvas, List<PointF> polygon, Point aOffset, double ratio) {
			PointF point = polygon.get(0);
			Point start = new Point((int)(point.x * ratio) + aOffset.x, (int)(point.y * ratio) + aOffset.y);
			Point a 	= new Point(start);
			Point b 	= new Point();

			for (int i = 1; i < polygon.size(); i++) {
				point = polygon.get(i);
				b.x = (int)(point.x * ratio) + aOffset.x;
				b.y = (int)(point.y * ratio) + aOffset.y;

				canvas.drawCircle(a.x, a.y, mCircleRadius, mPaint);
				canvas.drawLine(a.x, a.y, b.x, b.y, mPaint);

				a.x = b.x; 
				a.y = b.y;
			}
			canvas.drawCircle(start.x, start.y, mCircleRadius, mPaint);
			canvas.drawLine(a.x, a.y, start.x, start.y, mPaint);
		}

		private Object3D createZone(List<PointF> points, float height){
			List<PolygonPoint> polyPoints = new ArrayList<PolygonPoint>();

			for (int i = 0; i < points.size(); i++) {
				polyPoints.add(new PolygonPoint(points.get(i).x, points.get(i).y));
			}

			Polygon poly = new Polygon(polyPoints);
			Poly2Tri.triangulate(poly);

			List<DelaunayTriangle> triangles = poly.getTriangles();

			Object3D roof = new Object3D(triangles.size());


			for (int i = triangles.size()-1; i >= 0 ; i--) {
				DelaunayTriangle triangle = triangles.get(i);

				SimpleVector v1, v2, v3;

				TriangulationPoint vertex1 = triangle.points[2];
				TriangulationPoint vertex2 = triangle.points[1];
				TriangulationPoint vertex3 = triangle.points[0];

				v1 = new SimpleVector(vertex1.getX(), vertex1.getY(), -height);
				v2 = new SimpleVector(vertex2.getX(), vertex2.getY(), -height);
				v3 = new SimpleVector(vertex3.getX(), vertex3.getY(), -height);

				roof.addTriangle(v1, v2, v3);
			}

			Object3D sides = new Object3D(points.size()*2);

			for (int i = 0; i < points.size(); i++) {

				Object3D t = new Object3D(2);

				SimpleVector v1, v2, v1h, v2h;

				if (i == points.size()-1) {
					v1 = new SimpleVector(points.get(i).x, points.get(i).y, 0);
					v2 = new SimpleVector(points.get(0).x, points.get(0).y, 0);
					v1h = new SimpleVector(points.get(i).x, points.get(i).y, -height);
					v2h = new SimpleVector(points.get(0).x, points.get(0).y, -height);
				} else {
					v1 = new SimpleVector(points.get(i).x, points.get(i).y, 0);
					v2 = new SimpleVector(points.get(i+1).x, points.get(i+1).y, 0);
					v1h = new SimpleVector(points.get(i).x, points.get(i).y, -height);
					v2h = new SimpleVector(points.get(i+1).x, points.get(i+1).y, -height);
				}

				t.addTriangle(v1h, v1, v2h);
				t.addTriangle(v1, v2, v2h);

				sides = Object3D.mergeObjects(sides, t);
			}

			roof.calcNormals();
			VertexController roofVC = new VertexController();
			roofVC.setPolyManager(roof.getPolygonManager());
			roof.getMesh().setVertexController(roofVC, false);			
			Mesh mesh = roof.getMesh();
			mesh.applyVertexController();

			sides.calcNormals();
			VertexController sideVC = new VertexController();
			sideVC.setSide();
			sideVC.setPolyManager(sides.getPolygonManager());
			sides.getMesh().setVertexController(sideVC, false);			
			mesh = sides.getMesh();
			mesh.applyVertexController();

			Object3D zone = Object3D.mergeObjects(roof, sides);
			zone.setCollisionMode(Object3D.COLLISION_CHECK_OTHERS);
			zone.strip();
			zone.compile();

			return zone;


		}

		private void createZone3D(float mapScale){
			if (m3DZone == null) {

				List<PointF> poly = (List<PointF>) mPolygon.clone();

				for (int j = 0; j < poly.size(); j++) {
					poly.get(j).x = poly.get(j).x / mapScale;
					poly.get(j).y = poly.get(j).y / mapScale;
				}

				cleanZone(poly);

				mHeight = 3f / mapScale;
				if(mHeight < 2) mHeight = 2;
				m3DZone = createZone(poly, mHeight);
				m3DZone.setTexture(MapConstants.DEFAULT_MAP_ZONE_COLOR + "");

				m3DZone.setCollisionMode(Object3D.COLLISION_CHECK_OTHERS);

				m3DZone.forceGeometryIndices(true);
				m3DZone.setName(CommonConstants.ZONE_OBJECT_3D_NAME);
				m3DZone.strip();
				m3DZone.setCollisionOptimization(true);
				m3DZone.build();
			} 
		}

		private void cleanZone(List<PointF> points){
			points.remove(points.size()-1);

			int signedArea = 0;

			for (int i = 0; i < points.size(); i++) {
				PointF ptA = points.get(i);
				PointF ptB = points.get((i == points.size() -1) ? 0 : i+1);
				signedArea += (ptA.x * ptB.y - ptB.x * ptA.y);
			}

			if (signedArea > 0) {
				Collections.reverse(points);
			}

		}

		class VertexController extends GenericVertexController {

			private final static String TAG = "VertexController.class";

			private static final long serialVersionUID = 1L;

			private PolygonManager mPolyManager;
			private boolean isSide = false;

			public VertexController() {
			}

			public void setPolyManager(PolygonManager poly){
				mPolyManager = poly;
			}

			public void setSide(){
				isSide = true;
			}

			public void apply() {
				SimpleVector[] source = getSourceMesh();
				SimpleVector[] destinationNormals = getDestinationNormals();

				Log.d(TAG, "sources: " + source.length);


				for (int i = 0; i < source.length; i++) {
					if (isSide) {
						int[] polyIds = getPolygonIDs(i, 3);

						SimpleVector normal = new SimpleVector();

						List<SimpleVector> usedNormal = new ArrayList<SimpleVector>();

						for (int j = 0; j < polyIds.length; j++) {
							SimpleVector t = mPolyManager.getTransformedNormal(polyIds[j]);
							boolean in = false;
							for (int k = 0; k < usedNormal.size(); k++) {
								if (usedNormal.get(k).x == t.x && 
										usedNormal.get(k).y == t.y && 
										usedNormal.get(k).z == t.z) {
									in = true;
									break;
								}
							}

							if (!in) {
								usedNormal.add(t);
								normal.x += t.x;
								normal.y += t.y;
								normal.z = 0;

							}

							destinationNormals[i] = normal;
						}
					} else {
						destinationNormals[i].set(new SimpleVector(0, 0, -1));
					}
				}
			}
		}
