package com.insiteo.map.entities;

import android.graphics.Rect;
import android.graphics.RectF;

/**
 * Class containing map data needed to display a map, such as map's scale, zones color, map's bounds etc.
 */
public class MapData implements IMapData {

	private int mId;
	private String mName;
	private boolean mIsDebugMap;
	private int mSortIndex;
	
	private boolean mIsOriented;
	private float mAzimuth;
	private boolean mIsGeoreferenced;
	private GeoMatrix mGeoMatrix;
	
	private Integer mTypeId;
	private Integer mFloor;
	private String mFloorName;
		
	private String mExtra1;
	private String mExtra2;
	private String mExtra3;
	private Float mFExtra1;
	
	private float mScale; // in meters/pixel
	private SparseArray<ZoomLevel> mZoomLevels;
	private int mZoomMin;
	private int mZoomMax;
	
	private boolean mDisplayBackgroundTile;
	private boolean mIsSlideActive;
	
	private Integer mDefaultZoomLevel;
	private PointD mStartCenterPosition;
	private EStartViewMode mStartViewMode;
	private Integer mBackgroundColor;
	private Integer mZoneColor;
	
	private String mDispExtra1;
	private String mDispExtra2;
	private String mDispExtra3;
	private Float mDispFExtra1;
	
	
	/**
	 * Constructor
	 * @param id the ID of the map
	 * @param name the name of the map
	 * @param isDebug if true, the map exists only for debug purpose
	 * @param sortIndex used to sort maps
	 * @param scale
	 * @param zoomLevels
	 * @param displayBackgroundTile
	 * @param isSlideActive
	 */
	public MapData(int id, String name, boolean isDebug, int sortIndex, float scale, SparseArray<ZoomLevel> zoomLevels, boolean displayBackgroundTile, boolean isSlideActive) {
		mId = id;
		mName = name;
		mIsDebugMap = isDebug;
		mSortIndex = sortIndex;
		
		mIsOriented = false;
		mAzimuth = 0f;
		mIsGeoreferenced = false;
		mGeoMatrix = null;
		
		mTypeId = null;
		mFloor = null;
		mFloorName = null;
		mExtra1 = null;
		mExtra2 = null;
		mExtra3 = null;
		mFExtra1 = null;
		
		mScale = scale;
		mZoomLevels = zoomLevels;
		mIsSlideActive = isSlideActive;		
		mDisplayBackgroundTile = displayBackgroundTile;
		
		//set zoomMin and zoomMax : search for levels in zoomLevel list
		mZoomMin = 3000;
		mZoomMax = -1;
		
		if (zoomLevels != null) {
			for (int i=0 ; i<zoomLevels.size() ; i++) {
				int level = zoomLevels.keyAt(i);
				if (mZoomMin > level) {
					mZoomMin = level;
				}
				if (mZoomMax < level) {
					mZoomMax = level;
				}
			}
		}
		
		mDefaultZoomLevel = null;
		mStartCenterPosition = null;
		mStartViewMode = EStartViewMode.FIT_TO_SCREEN; 
		mBackgroundColor = null;
		mZoneColor = null;
		mDispExtra1 = null;
		mDispExtra2 = null;
		mDispExtra3 = null;
		mDispFExtra1 = null;
	}

	/**
	 * Set the map's azimuth. If given azimuth is valid, map will be oriented.
	 * @param azimuth the map's azimuth. If null, map will not be oriented.
	 */
	public void setAzimuth(Float azimuth) {
		if (azimuth == null) {
			mIsOriented = false;
		}
		else {
			mIsOriented = true;
		}
		this.mAzimuth = azimuth;

	public void setZoomLevels(SparseArray<ZoomLevel> zooms) {
		mZoomLevels = zooms;
		if (zooms != null) {
			for (int i=0 ; i<zooms.size() ; i++) {
