package com.insiteo.init;

import java.io.File;

import com.insiteo.common.CommonConstants;
import com.insiteo.common.utils.FileUtils;
import com.insiteo.common.utils.Log;

/**
 * Enum describing the package types used by Insiteo modules
 */
public enum EPackageType {

	/**
	 * @hide
	 */
	FULL(""),
	/**
	 * Package containing all the data required to process location.
	 */
	LOCATION("location"),
	/**
	 * Package containing all the data required to use the itinerary service.
	 */
	ITINERARY("itinerary"),
	/**
	 * Package containing all the data required to use the geofencing service.
	 */
	GEOFENCING("geofencing"),
	/**
	 * Package containing all the data that are related to the map.
	 */
	MAPDATA("mapdata"),
	/**
	 * Package containing all the map tiles.
	 */
	TILES("tiles"),
	/**
	 * @hide
	 */
	CONTENTLRR("contentLRR"),
	/**
	 * @hide
	 */
	CONTENTHRR("contentHRR"),
	/**
	 * Package containing all the data required to process a site fingerprint.
	 */
	FINGERPRINT("fingerprint"),
	/**
	 * Package containing all the data required to used the analytics service.
	 */
	ANALYTICS("analytics"),
	/**
	 * @hide
	 */
	UNKNOWN("");

	private String mDirName;	//the directory where to deploy the package

	private EPackageType(String aDirName) {
		mDirName = aDirName;
	}

	/**
	 * Return the name of the directory where a package of this type will be deployed.
	 * @return the name of the directory where a package of this type will be deployed. 
	 */
	public String getDirName() {
		return mDirName;
	}

	//strings used to convert type from server 
	/*package*/ static final String FULL_STR = "Full";
	/*package*/ static final String LOCATION_STR = "Loc";
	/*package*/ static final String ITINERARY_STR = "Iti";
	/*package*/ static final String GEOFENCING_STR = "Geofence";
	/*package*/ static final String MAPDATA_STR = "MapData";
	/*package*/ static final String TILES_STR = "Tiles";
	/*package*/ static final String CONTENTLRR_STR = "LRRContent";
	/*package*/ static final String CONTENTHRR_STR = "HRRContent";
	/*package*/ static final String FINGERPRINT_STR = "Fingerprint";
	/*package*/ static final String ANALYTICS_STR = "Datamining";

	/*package*/ static EPackageType getFromValue(String value) {
		EPackageType type = UNKNOWN;

		if (FULL_STR.equalsIgnoreCase(value)) {
			type = FULL;
		}
		else if (LOCATION_STR.equalsIgnoreCase(value)) {
			type = LOCATION;
		}
		else if (ITINERARY_STR.equalsIgnoreCase(value)) {
			type = ITINERARY;
		}
		else if (GEOFENCING_STR.equalsIgnoreCase(value)) {
			type = GEOFENCING;
		}
		else if (MAPDATA_STR.equalsIgnoreCase(value)) {
			type = MAPDATA;
		}
		else if (TILES_STR.equalsIgnoreCase(value)) {
			type = TILES;
		}
		else if (CONTENTLRR_STR.equalsIgnoreCase(value)) {
			type = CONTENTLRR;
		}
		else if (CONTENTHRR_STR.equalsIgnoreCase(value)) {
			type = CONTENTHRR;
		}
		else if (FINGERPRINT_STR.equalsIgnoreCase(value)) {
			type = FINGERPRINT;
		} 
		else if (ANALYTICS_STR.equalsIgnoreCase(value)){
			type = ANALYTICS;
		}

		return type;
	}

	/**
	 * Return the local version of this package for the current site.
	 * @return The version of this package or -1 if it doesn't exist.
	 */
	public int getVersion(){
		int version = -1;
		File f = new File(InitProvider.getInstance().getDataDirPath() + "/" + getDirName() + "/" + CommonConstants.VERSION_FILENAME);
		String strVers = FileUtils.readTextFile(f);

		try{
			version = Integer.parseInt(strVers);
		} catch (NumberFormatException e){
			Log.d(getDirName(), "Version file for package could not be parsed properly");
		}
		return version;
	}

}
