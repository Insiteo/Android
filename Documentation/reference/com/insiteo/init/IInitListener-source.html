public interface

IInitListener

package com.insiteo.init;

import com.insiteo.common.InsiteoError;

/**
 * Interface that defines the callback that get triggered upon Insiteo API initialization.
 * @author Insiteo
 *
 */
public interface IInitListener {
	
	/**
	 * This will be called once API initialization is done.
	 * @param aResult the status of initialization 
	 * <ul>
	 * <li><code>SUCCESS</code> if init was successful
	 * <li><code>SUCCESS_WITH_NEW_DATA</code> if init was successful and new data packages are available on server (call InitProvider.updatePackages to download new data)
	 * <li><code>FAIL</code> otherwise (call {@link InitProvider#hasPackage(EPackageType) InitProvider.hasPackage} to check data availability for each module you want to use)
	 * </ul>
	 * @param aError an InsiteoError if init failed with known reason, null otherwise. 
	 */
	public void onInitDone(EInitResult aResult, InsiteoError aError);
	
	/**
	 * This will be called during package download, and can be used to fill a progress bar.
	 * This will be called on UI thread.
	 * @param aDownloadedBytes the number of bytes already downloaded for the current package
	 * @param aTotalBytes the size of the package, in byte
	 */
	public void onDownloadProgress(long aDownloadedBytes, long aTotalBytes);
	
	/**
	 * This will be called during package install, and can be used to fill a progress bar.
	 * This will be called on UI thread.
	 * @param aCurrentFile the number of files already deployed
	 * @param aTotalFiles the number of file to deploy
	 */
	public void onInstallProgress(long aCurrentFile, long aTotalFiles);
	
	/**
	 * This will be called when package update is finished (download + install).
	 * @param aSuccess the status of update : <code>true</code> if update was successful, <code>false</code> otherwise
	 * @param aError an error associated to the status. This error contains a code, and a either a message (as a string) or 
	 * a messageId (to get message from resources strings).
	 */
	public void onDataUpdateDone(boolean aSuccess, InsiteoError aError);
	
	/**
	 * This will be called when a new package starts downloading.
	 * @param aPkgType
	 */
	public void onPackageDowloadBegin(EPackageType aPkgType);
}
