package com.insiteo.meetme;

import java.util.HashSet;
import java.util.Set;

import android.graphics.Bitmap;

import com.insiteo.common.utils.geom.Position;

/**
 * Class storing a person's data
 */
public class Person {
	
	// person data
	private Position 		mLocation		= null;
    private String 			mUserID			= null;		
    private String 			mExtUserID		= null;
    private String			mUserName 		= null;
    private Bitmap 			mPortrait 		= null;		//TODO : not used now 
    private Set<Integer>	mGroupIDs		= new HashSet<Integer>(4);
    private int mColor;
	
	/**
	 * Default constructor. This should not be used : Persons are created by MeetMeProvider
	 */
	public Person(Position aLocation, String aUserID, String aExtUserId, String aUserName, int aColor) {
    	mLocation = aLocation;
    	mUserID = aUserID;
    	mExtUserID = aExtUserId;
    	mUserName = aUserName;
    	mColor = aColor;
	}	
	
	/**
	 * Update this person's data, with the person passed as parameter 
	 * @param person the person to get values from
	 */
	public void update(Person person) {
		mExtUserID = person.mExtUserID;
		mUserName = person.mUserName;
		setLocation(person.mLocation);
		setGroupIDs(person.mGroupIDs);
	}
	
	
	/**
	 * @return the location of this person, or null if no location set
	 */
	public Position getLocation() {
		if (mLocation == null) {
			return null;
		}
		else {
			return new Position(mLocation);
		}
	}
	
	/**
	 * @param aLocation set Person's location
	 */
	public void setLocation(Position aLocation) {
		if (aLocation == null) {
			mLocation = null;
		}
		else {
			if (mLocation == null) {
				mLocation = new Position(aLocation);
			}
			else {
				mLocation.setPosition(aLocation);
			}
		}
	}
	
	/**
	 * @return true if this person has a location, false otherwise
	 */
	public boolean hasLocation() {
		return mLocation != null;
	}
	
	/**
	 * @return the GUID of the Person on Insiteo servers
	 */
	public String getUserID() {
		return mUserID;
	}
	
	/**
	 * @return the external ID of the Person, or null if the Person is from an Insiteo account
	 */
	public String getExternalUserID() {
		return mExtUserID;
	}
	
	/**
	 * @return the name of the Person
	 */
	public String getName() {
		return mUserName;
	}
	
	/**
	 * @return the Bitmap representing this person's portrait. Be careful : it is not a copy that is returned.
	 */
	public Bitmap getPortrait() {
		return mPortrait;
	}
	
	/**
	 * @param aPortrait set Person's image to display
	 */
	public void setPortrait(Bitmap aPortrait) {
		mPortrait = aPortrait;
	}

	/**
	 * @return Group IDs which this Person is connected to
	 */
	public Set<Integer> getGroupIDs() {
		return mGroupIDs;
	}
	
	public void setGroupIDs(Set<Integer> aGroupIDs) {
		if (aGroupIDs != null) {
			mGroupIDs = aGroupIDs;
		}
	}
	
	public int addGroupID(int aGroupID) {
		mGroupIDs.add(aGroupID);
		return mGroupIDs.size();
	}
	
	public int removeGroupID(int aGroupID) {
		mGroupIDs.remove(aGroupID);
		return mGroupIDs.size();
	}

	/**
	 * @return the color assigned to this person on the server
	 */
	public int getColor() {
		return mColor;
	}
	
	/**
	 * Return the name of this person, followed by (*) if its location is known. 
	 */
	@Override
	public String toString() {
		if (mLocation != null) {
			return mUserName + "(*)";
		}
		else {
			return mUserName;
		}
	}
	

}
