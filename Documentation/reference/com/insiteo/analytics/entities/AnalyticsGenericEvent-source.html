AnalyticsGenericEvent

package com.insiteo.analytics.entities;

import java.util.ArrayList;
import java.util.List;

import com.insiteo.analytics.AnalyticsConstants;
import com.insiteo.analytics.AnalyticsManager;
import com.insiteo.common.utils.geom.Position;

import android.content.Context;

/**
 * Generic event that can be used to catch specific behavior. This event will be sort by its type and can contain 2 String parameters, 2 int parameters, 2 double parameters
 * and 2 Positions. All the required field of the event must be filled before passing it to the AnalyticsManager.
 *
 */
public class AnalyticsGenericEvent extends AnalyticsAbstractEvent{

	private final static String TAG = "AnalyticsGenericEvent";
	
	private String mType;
	
	private Position mPos1;
	private Position mPos2;
	
	private String mString1;
	private String mString2;
	
	private double mDouble1;
	private double mDouble2;
	
	private int mInt1;
	private int mInt2;
		
	/**
	 * Create an AnalyticsGenericEvent with all its parameters
	 * @param aId
	 * @param aTimestamp
	 * @param aGuid
	 * @param aType
	 * @param aPos1
	 * @param aPos2
	 * @param aString1
	 * @param aString2
	 * @param aDouble1
	 * @param aDouble2
	 * @param aInt1
	 * @param aInt2
	 */
	public AnalyticsGenericEvent(long aId, long aTimestamp, String aGuid, String aType, Position aPos1, Position aPos2, String aString1, String aString2, double aDouble1, double aDouble2, int aInt1, int aInt2) {
		super(aId, aTimestamp, aGuid);
		this.mType = aType;
		this.mPos1 = aPos1;
		this.mPos2 = aPos2;
		this.mString1 = aString1;
		this.mString2 = aString2;
		this.mDouble1 = aDouble1;
		this.mDouble2 = aDouble2;
		this.mInt1 = aInt1;
		this.mInt2 = aInt2;
	}

	/**
	 * Create an AnalyticsGenericEvent only with its type.
	 * @param aType
	 */
	public AnalyticsGenericEvent(String aType){
		super(AnalyticsConstants.TYPE_GENERIC_EVENT);
		this.mType = aType;
	}

	public String getType() {
		return mType;
	}


	public Position getPos1() {
		return mPos1;
	}


	public Position getPos2() {
		return mPos2;
	}


	public String getString1() {
		return mString1;
	}


	public String getString2() {
		return mString2;
	}


	public double getDouble1() {
		return mDouble1;
	}


	public double getDouble2() {
		return mDouble2;
	}


	public int getInt1() {
		return mInt1;
	}


	public int getInt2() {
		return mInt2;
	}
	
	public List<Position> getPositions(){
		
		List<Position> positions = new ArrayList<Position>();
		
		if (mPos1 != null) {
			positions.add(mPos1);
		}
		
		if (mPos2 != null) {
			positions.add(mPos2);
		}
		
		return positions;
	}

	public void setType(String aType) {
		this.mType = aType;
	}

	public void setPos1(Position aPos1) {
		this.mPos1 = aPos1;
	}

	public void setPos2(Position aPos2) {
		this.mPos2 = aPos2;
	}

	public void setString1(String aString1) {
		this.mString1 = aString1;
	}

	public void setString2(String aString2) {
		this.mString2 = aString2;
	}

	public void setDouble1(double aDouble1) {
		this.mDouble1 = aDouble1;
	}

	public void setDouble2(double aDouble2) {
		this.mDouble2 = aDouble2;
	}

	public void setInt1(int aInt1) {
		this.mInt1 = aInt1;
	}

	public void setInt2(int aInt2) {
		this.mInt2 = aInt2;
	}
	
	public String toString(){
		String val = "Generic Event with : \n label = " + mType;
		
		if (mPos1 != null) {
			val += " position 1 = " + mPos1.toString(); 
		}
		
		if (mPos2 != null) {
			val += " position 2 = " + mPos2.toString(); 
		}
		
		if (mString1 != null) {
			val += " string 1 = " + mString1; 
		}
		
		if (mString2 != null) {
			val += " string 2 = " + mString2; 
		}
		
		if (mInt1 != 0) {
			val += " int 1 = " + mInt1; 
		}
		
		if (mInt2 != 0) {
			val += " int 2 = " + mInt2; 
		}
		
		if (mDouble1 != 0.0d) {
			val += " double 1 = " + mDouble1; 
		}
		
		if (mDouble2 != 0.0d) {
			val += " double 2 = " + mDouble2; 
		}
		
		return val;
	}

	
}
