package com.insiteo.lbs.meetme.requests;

import java.net.URLEncoder;

import com.insiteo.lbs.R;
import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.InsiteoError;
import com.insiteo.lbs.common.init.InitProvider;
import com.insiteo.lbs.common.parser.ProtoLoader;
import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.location.LocationConstants;
import com.insiteo.lbs.meetme.IMeetMeRequestListener;
import com.insiteo.lbs.meetme.MeetMeProvider;
import com.insiteo.lbs.meetme.parser.InsCreateTmpUserResponseParser;

public class InsCreateTmpUserRequest extends AbstractMeetMeRequest {

	private String mName = null;
		
	public InsCreateTmpUserRequest(MeetMeProvider aProvider, IMeetMeRequestListener aListener, String aName) {
		super(aProvider, aListener);
		mName = aName;
	}

	public String getName() {
		return mName;
	}
	
	@Override
	public void start() {
		new Thread(new Runnable() {
			@Override public void run() {
				InsiteoError error;
				InsCreateTmpUserResponseParser parser = new InsCreateTmpUserResponseParser();
				InitProvider iProv = InitProvider.getInstance();
				
				if (iProv.isInitialized()) {
					//create request url
					final String url = iProv.getLbsUrl() + LocationConstants.MM_INS_CREATETMPUSER_SERVICE
																			 .replace(CommonConstants.URLKEY_SITE_ID, ""+iProv.getSiteId())
			  		 														 .replace(CommonConstants.URLKEY_APP_VERS, ""+iProv.getAppVersion())
			  		 														 .replace(CommonConstants.URLKEY_LANG, iProv.getLang())
			  		 														 .replace(LocationConstants.URLKEY_NAME, URLEncoder.encode(mName));
					//execute http request
					error = ProtoLoader.load(url, parser);
				}
				else {
					Log.d("MeetMe", "Unable to execute request : API is not initialized");
					error = new InsiteoError(InsiteoError.API_NOT_INIT_ERROR_CODE, R.string.insiteo_common_error_init_api);
				}
				
				if (!mCancel) {
					mProvider.onTempConnectionDone(InsCreateTmpUserRequest.this, error, parser);
				}
			}
		}).start();
	}

}
