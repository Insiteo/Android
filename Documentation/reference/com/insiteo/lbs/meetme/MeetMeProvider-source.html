/**
 * This class is the entry point for MeetMe service. You can retrieve an instance of it by calling <code>(MeetMeProvider)LocationProvider.getInstance().getModule(ELocationModule.MEETME)</code>.
 * It handles requests with server, and notify its listener when new data is available.
 * Groups management can be done using external events (in this case, events are managed by application, using APIs such as facebook or linkedin). 
 * It can also be done using Insiteo events (in this case, MeetMeProvider provides methods to create insiteo events).
 * Login can be done using external profile (managed by application, using APIs such as facebook or linkedin).
 * It can also be done using insiteo profile :
 * <ul><li>either with a permanent profile (login + password + nickname)</li><li>or with a temporary profile, using only a nickname</li></ul> 
 */

		if (aName != null) {
			//TODO VDC : add mutex ? 
			for (int i=0 ; i<mGroups.size() ; i++) {
				String name = mGroups.valueAt(i).getName();

	/**
	 * Get the list of connected groups. A connected group is a group in which user is registered.
	 * @return the list of connected groups
	 */
	public List<Integer> getConnectedGroups() {
		List<Integer> groupIDs = new ArrayList<Integer>();
		//TODO VDC : add mutex ?
		for (int i=0 ; i<mGroups.size() ; i++) {
			Group group = mGroups.valueAt(i);
			if (group.getUserStatus() != UserStatus.DISCONNECTED) {
				groupIDs.add(group.getID());
			}
		}
		return groupIDs;
	}
	
	/**
	 * Get the list of persons that are connected to a group
	 * @param aGroupID the ID of the group to get persons from
	 * @return the list of connected persons
	 */
	public List<Person> getPersonsFromGroup(int aGroupID) {
		List<Person> persons;
		Group group = mGroups.get(aGroupID);
		if (group != null) {
			persons = new ArrayList<Person>(group.getPersonIDs().size());
			for (String id : group.getPersonIDs()) {
				Person person = mPersons.get(id);
				if (person != null) {
					persons.add(person);
				}
			}
		}
		else {
			persons = new ArrayList<Person>(0);
		}
		return persons;
	}

			PBReqGetLocMmContent.Builder mmBuilder = PBReqGetLocMmContent.newBuilder();

			isAdded = true;

	@Override
	/**
	 * @hide
	 */
	public void setLbsResponse(LbsModuleResponse aLocResponse) {
		if (mIsConnected) {
			LocMessageParser parser = new LocMessageParser(); 
			InsiteoError error = parser.parse(aLocResponse);
			if (InsiteoError.NO_ERROR_CODE.equals(error.getCode())) {
				//request OK => update groups 
				updatePersons(parser.getGroups(), parser.getPersons());
				setPersonsInRenderer();
			}
			else {
				if (!InsiteoError.LBS_MODULE_ABSENT_ERROR_CODE.equals(error.getCode())) {
					//there is a meetme element in loc response
					if (mListener != null) {
						mListener.onMeetMeError(error);
					}
				}
			}
		}
	}

	}

	private void refreshGroups(List<Group> aGroups, boolean keepStatus) {
		//create a list containing all existing groups
		List<Group> groupsKicked = new ArrayList<Group>(mGroups.size());
		for (int i=0 ; i<mGroups.size() ; i++) {
			groupsKicked.add(mGroups.valueAt(i));
		}
		
		//update groups with new data
		for (Group newGroup : aGroups) {
			Group group = mGroups.get(newGroup.getID());
			if (group == null) {
				//group does not exist yet => add it
				mGroups.put(newGroup.getID(), newGroup);
			}
			else {
				//group already exist : update it
				if (keepStatus) {
					//save current status
					newGroup.setUserStatus(group.getUserStatus());
				}
				if (newGroup.getName().length() == 0) {
					newGroup.setName(group.getName());	//set name if no name defined 
				}
				newGroup.setPersonIDs(group.getPersonIDs());	//save current person list
				group.update(newGroup);
				if (group.getUserStatus() == UserStatus.DISCONNECTED) {
					//erase person list
					for (String id : group.getPersonIDs()) {
						Person person = mPersons.get(id); 
						if (person != null && person.removeGroupID(group.getID()) <= 0) {
							mPersons.remove(id);
						}
					}
					group.getPersonIDs().clear();
				}
				
				//group is updated : remove it from list of kicked
				groupsKicked.remove(group);
			}
		}
		
		//update group from which we have been kicked by server
		for (Group group : groupsKicked) {
			group.setUserStatus(UserStatus.DISCONNECTED);		
		}
	}
	

	private void forceLeaveGroup(Group group) {
		if (group != null) {
			if (group.getUserStatus() != UserStatus.DISCONNECTED) {
				//group needs to be disconnected
				group.setUserStatus(UserStatus.DISCONNECTED);
				int userCount = group.getUsersCount() - 1;
				if (userCount < 0) {
					userCount = 0;
				}
				group.setUsersCount(userCount);
				
				//update group count for all this group's persons
				for (String persID : group.getPersonIDs()) {
					Person person = mPersons.get(persID);
					if (person != null && person.removeGroupID(group.getID()) <= 0) {
						//no more group referencing this person : remove it
						mPersons.remove(persID);
					}
				}
				
				//clear group's persons
				group.setPersonIDs(null);
			}
		}
	}
	
	
	private void leaveAllOtherGroups(int groupIdToKeep) {
		//disconnect all other groups
		for (int i=0 ; i<mGroups.size() ; i++) {
			Group group = mGroups.valueAt(i);
			if (group.getID() != groupIdToKeep) {
				forceLeaveGroup(group);
			}
		}
	}
	
	
	private void updatePersons(SparseArray<Group> aGroups, HashMap<String, Person> aPersons) {
		//set new persons
		mPersons = aPersons;		
		
		//add all new groups in mGroups
		for (int i=0 ; i<aGroups.size() ; i++) {
			Group newGroup = aGroups.valueAt(i);
			Group oldGroup = mGroups.get(newGroup.getID());
			if (oldGroup == null) {
				//new group does not exist yet in mGroups : add it
				mGroups.put(newGroup.getID(), newGroup);
			}
			else {
				newGroup.setName(oldGroup.getName());
				newGroup.setUserStatus(oldGroup.getUserStatus());
				newGroup.setExternalId(oldGroup.getExternalId());
				oldGroup.update(newGroup);
				mGroups.put(oldGroup.getID(), oldGroup);
			}
		}
		
		//notify listener
		if (mListener != null) {
			SparseArray<Group> groups = new SparseArray<Group>(mGroups.size());
			for (int i=0 ; i<mGroups.size() ; i++) {
				groups.put(mGroups.keyAt(i), mGroups.valueAt(i));			
			}
			mListener.onPersonsReceived(groups, new HashMap<String, Person>(mPersons));
		}
	}
	
	
	private void updateGroup(Group group, List<Person> aPersons) {
		//create persons list for the newly joined group
		List<String> personIDs = new ArrayList<String>(aPersons.size());
		int groupID = group.getID();
		
		//update persons hashmap
		for (Person newPerson : aPersons) {
			Person oldPerson = mPersons.get(newPerson.getUserID()); 
			if (oldPerson != null) {
				//person already exists : update it
				oldPerson.update(newPerson);
				oldPerson.addGroupID(groupID);	//the person is in the new group
			}
			else {
				//person does not exist yet : add it
				newPerson.addGroupID(groupID);	//the person is in the new group
				mPersons.put(newPerson.getUserID(), newPerson);
			}
			
			//add person's ID in group's person list
			personIDs.add(newPerson.getUserID());
		}
		
		//add the persons in this group
		group.setPersonIDs(personIDs);
	}
	
	
	private void setPersonsInRenderer() {
		if (mRenderer != null) {
			//TODO VDC : mutex ?
			ArrayList<Person> persons = new ArrayList<Person>(mPersons.values());
			mRenderer.updatePersons(persons);
		}
	}

	/**
	 * Force MeetMeProvider disconnection from Insiteo servers. This is an offline, synchronous request. 
	 */
	public void disconnect() {
		mIsConnected = false;
		mGroups = new SparseArray<Group>();
		mPersons = new HashMap<String, Person>();
		mUserID = "";
		mUserName = "";
		setPersonsInRenderer();
	}
	/**
	 * Request connection to Insiteo server, from an insiteo profile. 
	 * This is an asynchronous call, that will cause an event to be fired to a IMeetMeRequestListener. 
	 * Returned request can be canceled ; if it is canceled, no event will be fired.   
	 * @param aListener the listener that will receive notification when request ends
	 * @param aLogin
	 * @param aPassword
	 * @return the request object
	 */
	public AbstractMeetMeRequest requestConnection(IMeetMeRequestListener aListener, String aLogin, String aPassword) {
		AbstractMeetMeRequest request = new InsConnectRequest(this, aListener, aLogin, aPassword);
		request.start();
		return request;
	}
	/** @hide Internal callback : application must not use it ! */
	public void onConnectionDone(InsConnectRequest aRequest, InsiteoError aError, IResponseParser aParser) {
		boolean success = false;
		if (InsiteoError.NO_ERROR_CODE.equals(aError.getCode())) {
			//update user's data and groups
			InsConnectResponseParser parser = (InsConnectResponseParser)aParser;
			mUserID = parser.getGUID();
			mUserName = parser.getNickname();
			mPersons = new HashMap<String, Person>(0);
			refreshGroups(parser.getGroups(), false);
			mIsConnected = true;
			success = true;
			setPersonsInRenderer();
		}
		//notify listener
		aRequest.notifyListener(success, aError);
	}

	 * @return the request object

	public AbstractMeetMeRequest requestGroupCreation(IMeetMeRequestListener aListener, String aGroupName, String aGroupPassword, long aEndDate, long aCloseDate, boolean aIsExclusive) {

	/** @hide Internal callback : application must not use it ! */
	public void onGroupCreationDone(InsCreateGroupRequest aRequest, InsiteoError aError, IResponseParser aParser) {
		boolean success = false;
		if (InsiteoError.NO_ERROR_CODE.equals(aError.getCode())) {
			InsCreateGroupResponseParser parser = (InsCreateGroupResponseParser)aParser;
			//add new group to groups list
			Group group = new Group(parser.getGroupID(), aRequest.getGroupName(), 1, null, UserStatus.VISIBLE, null);
			int groupID = group.getID();
			mGroups.put(groupID, group);
			
			//check if the join is exclusive (ie : no more than one group connected)
			if (aRequest.isExclusive()) {
				leaveAllOtherGroups(groupID);
			}
			
			success = true;
			setPersonsInRenderer();			
		}
		//notify listener
		aRequest.notifyListener(success, aError);
	}
