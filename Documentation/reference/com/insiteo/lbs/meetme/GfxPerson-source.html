	@Override
	public void setResources(Resources res) {
		int color = mPerson.getColor();
		String userName = mPerson.getName();
		
		//init default portrait if needed
		if (mDefaultPortrait == null) {
			mDefaultPortrait = BitmapFactory.decodeResource(res, R.drawable.insiteo_meetme_person_default_portrait);
		}
		
		//init loc icon
		mLocIcon = BitmapFactory.decodeResource(res, R.drawable.insiteo_meetme_position);
		mLocIconWidth = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_loc_width);
		mLocIconHeight = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_loc_height);
		mLocIconRect = new Rect();
		mLocIconPaint = new Paint();
		int porterDuffColor = Color.argb(255, Color.red(color), Color.green(color), Color.blue(color));
		mLocIconPaint.setColorFilter(new PorterDuffColorFilter(porterDuffColor, Mode.MULTIPLY));
		
		//init animated circle
		mAnimRadius = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_loc_anim_radius);
		mAnimPaint = new Paint();
		mAnimPaint.setColor(color);
		mAnimPaint.setAlpha(ANIM_MAX_ALPHA);
		mAnimPaint.setStyle(Style.FILL);
		mAnimBorderPaint = new Paint();
		mAnimBorderPaint.setColor(color);
		mAnimBorderPaint.setAlpha(ANIM_MAX_ALPHA);
		mAnimBorderPaint.setStyle(Style.STROKE);
		mAnimBorderPaint.setStrokeWidth(ANIM_STROKE_WIDTH);
		mAnimBorderPaint.setAntiAlias(true);
		
		
		//get all sizes from resources
		mPictureSize = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_picture);
		mNameTextSize = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_name_text);
		mTextSpace = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_name_text_space);
		mBackgHOffset = res.getDimensionPixelSize(R.dimen.insiteo_meetme_person_name_background_h_offset);
		
		//create text paint for name
		float shadowRadius = res.getDimension(R.dimen.insiteo_meetme_person_name_text_shadow_radius);
		mNamePaint = new TextPaint();
    	mNamePaint.setColor(color);
    	mNamePaint.setAntiAlias(true);
        mNamePaint.setAlpha(255);
        mNamePaint.setTextSize(mNameTextSize);
        mNamePaint.setShadowLayer(shadowRadius, 0f, 0f, Color.WHITE);
        
        //calculate name width
        int nameWidth = (int)DynamicLayout.getDesiredWidth(userName,  mNamePaint);
        
        //load background 9-patch
		Drawable d = res.getDrawable(R.drawable.insiteo_meetme_person_name_background);
		mNameBackg = (NinePatchDrawable)d;
        Rect rPadding = new Rect();
		mNameBackg.getPadding(rPadding);
        
		//compute content width and height
		int contentHeight = Math.max(mNameTextSize, mPictureSize);
		int imgBackgHeight = rPadding.top + contentHeight + rPadding.bottom;
		int imgMinHeight = mNameBackg.getMinimumHeight();
		if (imgBackgHeight < imgMinHeight) {
			contentHeight += imgMinHeight - imgBackgHeight; 
		}
		
		int contentWidth = contentHeight /*square picture*/ + mTextSpace + nameWidth; 
        int imgBackgWidth = rPadding.left + contentWidth + rPadding.right;
		int imgMinWidth = mNameBackg.getMinimumWidth();
        if (imgBackgWidth < imgMinWidth) {
        	contentWidth += imgMinWidth - imgBackgWidth;
        }
		
        mContentRect = new Rect(rPadding.left, rPadding.top, rPadding.left + contentWidth, rPadding.top + contentHeight);
		
        //compute background sizes
        mBackgRect = new Rect();
        mNameBackgWidth = rPadding.left + mContentRect.width() + rPadding.right;
        mNameBackgHeight = rPadding.top + mContentRect.height() + rPadding.bottom;
        
        //compute text vertical margin, to center it in its rectangle
        Rect txtBounds = new Rect();
		mNamePaint.getTextBounds(userName, 0, userName.length(), txtBounds);
		mNameVMargin = (mContentRect.height() - txtBounds.height()) / 2;
		
		mIsInit = true;
	}
	

	@Override

		mIsInit = false;
		mNameBackg = null;
		mNamePaint = null;
		mContentRect = null;
		mBackgRect = null;
		mLocIconRect = null;
		mLocIcon = null;
		mLocIconPaint = null;
		mAnimPaint = null;
		mAnimBorderPaint = null;
	}


	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if (mPerson.hasLocation()) {
			drawIcon(aCanvas, aOffset, aRatio, aAngle);
			drawName(aCanvas, aOffset, aRatio, aAngle);
		}
	}
		
	
	//Draws the standard person icon, and draw an animated circle if name is displayed.
	public void drawIcon(Canvas aCanvas, Point aOffset, double aRatio, float aAngle) {
		Position location = mPerson.getLocation();
    	if(location != null && mIsInit) {
    		//get person location on screen
			Point pos = new Point((int)(location.getCoord().x * aRatio + aOffset.x), (int)(location.getCoord().y * aRatio + aOffset.y));
	
			//draw animated circle if person name is visible
			if(mIsNameDisplayed) {
	    		float radiusRatio = (SystemClock.elapsedRealtime() % ANIM_DURATION) / (float)ANIM_DURATION;
	    		float alphaRatio = 1f - radiusRatio;
		    	mAnimPaint.setAlpha((int)(ANIM_MAX_ALPHA * alphaRatio));
	    		aCanvas.drawCircle(pos.x, pos.y, mAnimRadius * radiusRatio, mAnimPaint);
	    		aCanvas.drawCircle(pos.x, pos.y, mAnimRadius * radiusRatio, mAnimBorderPaint);
	    	}
			
	    	// draw position point
	    	mLocIconRect.left = pos.x - mLocIconWidth/2;
	    	mLocIconRect.top = pos.y - mLocIconHeight/2; 
	    	mLocIconRect.right = mLocIconRect.left + mLocIconWidth;
	    	mLocIconRect.bottom = mLocIconRect.top + mLocIconHeight;
	    	aCanvas.drawBitmap(mLocIcon, null, mLocIconRect, mLocIconPaint);
    	}
    }
	
	public void drawName(Canvas aCanvas, Point aOffset, double aRatio, float aAngle) {
		Position location = mPerson.getLocation();
		Bitmap portrait = mPerson.getPortrait();
		if (mIsNameDisplayed && mIsInit && location != null) {
			//get person location on screen
			Point pos = new Point((int)(location.getCoord().x * aRatio + aOffset.x), (int)(location.getCoord().y * aRatio + aOffset.y));
	
			//rotate the canvas back if needed
			if (aAngle != 0f) {
				aCanvas.save();
				aCanvas.rotate(-aAngle, pos.x, pos.y);
			}
			
			//draw name background
			mBackgRect.left = pos.x-mBackgHOffset;
			mBackgRect.top = pos.y-mNameBackgHeight;
			mBackgRect.right = mBackgRect.left + mNameBackgWidth;
			mBackgRect.bottom = mBackgRect.top + mNameBackgHeight;
			mNameBackg.setBounds(mBackgRect);
			mNameBackg.draw(aCanvas);
			
			//draw name text
			Rect contentRect = new Rect(mBackgRect.left+mContentRect.left, mBackgRect.top+mContentRect.top, mBackgRect.left+mContentRect.left+mContentRect.width(), mBackgRect.top+mContentRect.top+mContentRect.height());
			aCanvas.drawText(mPerson.getName(), contentRect.left+mContentRect.height()+mTextSpace, contentRect.bottom-mNameVMargin, mNamePaint);
	
			//draw user picture, if any
			Rect imgRect = new Rect(contentRect.left, contentRect.top, contentRect.left+contentRect.height(), contentRect.bottom);
			if (portrait != null) {
				aCanvas.drawBitmap(portrait, null, imgRect, null);
			}
			else if (mDefaultPortrait != null) {
				aCanvas.drawBitmap(mDefaultPortrait, null, imgRect, null);
			}
			
			//restore canvas if needed
			if (aAngle != 0f) {
				aCanvas.restore();
			}
		}
	}

	private boolean mIsLocIconTouched = false;
	private boolean mIsNameTouched = false;
	private Point mTouchRefPoint = new Point();
	
	@Override
	public ETouchObjectResult onTouchDown(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;
		if (mPerson.hasLocation()) {
			//check loc icon
			if (mLocIconRect != null && mLocIconRect.contains(aTouch.getPosition().x, aTouch.getPosition().y)) {
				mIsLocIconTouched = true;
				mTouchRefPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);
				result = ETouchObjectResult.RESULT_CONSUME;
			}
//			//check name rect (not used yet : change if click on person is needed)
//			else if (mIsNameDisplayed && mBackgRect != null && mBackgRect.contains(aTouch.getPosition().x, aTouch.getPosition().y)) {
//				mIsNameTouched = true;
//				mTouchRefPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);
//				result = ETouchObjectResult.ACTION_CONSUME;
//			}
		}
		return result;
	}

	@Override
	public ETouchObjectResult onTouchPointerDown(Touch aTouch) {
		return ETouchObjectResult.RESULT_NOTHING;
	}
	
	@Override
	public ETouchObjectResult onTouchMove(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;
		
		if (mIsLocIconTouched || mIsNameTouched) {
			if (aTouch.isMultiTouch()) {
				mIsLocIconTouched = false;
				mIsNameTouched = false;
    		}
    		else {
				Point currentPoint = aTouch.getPosition();
		    	
		    	//if the touch position change too much, deselect person
		    	int dx = Math.abs(currentPoint.x - mTouchRefPoint.x);
		        int dy = Math.abs(currentPoint.y - mTouchRefPoint.y);
		    	if ((dx > CommonConstants.CLICKABLE_ZONE_SIZE) || (dy > CommonConstants.CLICKABLE_ZONE_SIZE)) {
		    		mIsLocIconTouched = false;
		    		mIsNameTouched = false;
		    	}	   
		    	
		    	result = ETouchObjectResult.RESULT_CONSUME;
    		}
		}
    	
		return result;
	}
	
	@Override
	public ETouchObjectResult onTouchPointerUp(Touch aTouch) {
		return ETouchObjectResult.RESULT_NOTHING;
	}
	
	@Override
	public ETouchObjectResult onTouchUp(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;
		//check loc icon
		if (mIsLocIconTouched && mLocIconRect != null && mLocIconRect.contains(aTouch.getPosition().x, aTouch.getPosition().y)) {
			//loc icon was clicked => toggle name infowindow
			mIsNameDisplayed = !mIsNameDisplayed;
			result = ETouchObjectResult.RESULT_CONSUME;
		}
//		//check name rect (not used yet : change if click on person is needed)
//		else if (mIsNameTouched && mIsNameDisplayed && mBackgRect != null && mBackgRect.contains(aTouch.getPosition().x, aTouch.getPosition().y)) {
//			result = ETouchObjectResult.ACTION_NOTIFY_CONSUME;
//		}
		mIsLocIconTouched = false;
		mIsNameTouched = false;
		return result;
	}
