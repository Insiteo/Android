package com.insiteo.lbs.location;

import com.insiteo.lbs.common.utils.geometry.PointD;
import com.insiteo.lbs.common.utils.geometry.Position;

/**
 * An extension of a PointD object. 
 * Gives a location in application coordinates system, as defined with insiteo
 */
public class InsLocation {
		
	public static final int UNKNOWN_MAP = -1; 
	
	private Position mPosition;
	private float mAccuracy;	// the position's accuracy
	private EInsLocationSource mSource;

	/**
	 * Test if given object is equal to InsLocation. 
	 * Two InsLocation are equal if their map ID and coordinates are equal
	 * @param o object on which to test equality
	 * return true if this instance of InsLocation is equal to given object
	 */
	@Override
    public boolean equals(Object o) {
        boolean isEqual = false;
        if (this == o) {
        	isEqual = true;
        }
        else {
			if ((o instanceof InsLocation)) {
	            InsLocation loc = (InsLocation) o;
	            isEqual = (loc.getMapID() == mPosition.getMapID() && loc.getCoord().equals(mPosition.getCoord()));
			}
        }
		return isEqual;
    }	
	
	//========================================================================
	// Constructors
	//========================================================================
	
	/**
	 * Constructor for the Location object.
	 * @param aMapID		The ID of the map where this Location is valid.
	 * @param aX			The X coordinate, in the defined coordinates system 
	 * @param aY			The Y coordinate, in the defined coordinates system 
	 * @param aAccuracy		The accuracy, expressed as a radius
	 * @param aSource		The source of location
	 */
	 public InsLocation(int aMapID, double aX, double aY, float aAccuracy, EInsLocationSource aSource) {
		mPosition = new Position(aMapID, aX, aY);
		mAccuracy = aAccuracy;
		mSource = aSource;
	 }
	 
	 /**
	 * Constructor for the Location object.
	 * @param aMapID	The ID of the map where this Location is valid.
	 * @param aCoord	The X/Y coordinate pair, in the defined coordinates system 
	 * @param aAccuracy	The accuracy, expressed as a radius
	 * @param aSource	The source of location
	 */
	 public InsLocation(int aMapID, PointD aCoord, float aAccuracy, EInsLocationSource aSource) {
		this(aMapID, aCoord.x, aCoord.y, aAccuracy, aSource);
	 }
	 
	/**
	 * Constructor for the Location object.
	 * @param aMapID	The ID of the map where this Location is valid.
	 * @param aX		The X coordinate, in the defined coordinates system 
	 * @param aY		The Y coordinate, in the defined coordinates system 
	 * @param aAccuracy	The accuracy, expressed as a radius
	 */
	 public InsLocation(int aMapID, double aX, double aY, float aAccuracy) {
		 this(aMapID, aX, aY, aAccuracy, EInsLocationSource.UNKNOWN);
	 }
	
	/**
	 * Constructor for the Location object, with a zero accuracy.
	 * @param aMapID	The ID of the map where this Location is valid.
	 * @param aX		The X coordinate, in the defined coordinates system 
	 * @param aY		The Y coordinate, in the defined coordinates system 
	 */
	public InsLocation(int aMapID, double aX, double aY) {
		this(aMapID, aX, aY, 0, EInsLocationSource.UNKNOWN);
	}	
	
	/**
	 * Constructor for the Location object, with a zero accuracy.
	 * @param aMapID	The ID of the map where this Location is valid.
	 * @param aCoord	The X/Y coordinate pair, in the defined coordinates system 
	 */
	public InsLocation(int aMapID, PointD aCoord) {
		this(aMapID, aCoord.x, aCoord.y);
	}	
	
	/**
	 * Constructor for the Location object.
	 * @param aMapID	The ID of the map where this Location is valid.
	 * @param aCoord	The X/Y coordinate pair, in the defined coordinates system 
	 * @param aAccuracy	The accuracy, expressed as a radius
	 */
	public InsLocation(int aMapID, PointD aCoord, float aAccuracy) {
		this(aMapID, aCoord.x, aCoord.y, aAccuracy, EInsLocationSource.UNKNOWN);
	}
		 	 
	/**
	 * Copy-constructor for the Location object.
	 * @param aLocation	The Location object to duplicate.
	 */
	public InsLocation(InsLocation aLocation) {
		this(aLocation.getMapID(), aLocation.getCoord().x, aLocation.getCoord().y, aLocation.getAccuracy(), aLocation.getSource());
	}
	
	
	//========================================================================
	// GETTERS
	//========================================================================
	
	/**
	 * Get a Position object filled with this location data
	 * @return a position filled with this InsLocation data
	 */
	public Position getPosition() {
		return new Position(mPosition);
	}	
	
	/**
	 * Gets the Map unique identifier from this Location object.
	 * @return	A unique MapID.
	 */
	public int getMapID() {
		return mPosition.getMapID();
	}
		
	/**
	 * Gets the X/Y pair from this Location object.
	 * @return the x/y pair of this position
	 */
	public PointD getCoord() {
		return mPosition.getCoord();
	}
	
	/**
	 * Get the X value from this Location object
	 * @return the x of this location
	 */
	public double getX() {
		return mPosition.getX();
	}
	
	/**
	 * Get the Y value from this Location object
	 * @return the y of this location
	 */
	public double getY() {
		return mPosition.getY();
	}
		
	/**
	 * Gets the accuracy value from this Location object.
	 * @return accuracy radius
	 */
	public float getAccuracy() {
		return mAccuracy;
	}
		
	/**
	 * Get the source of location
	 * @return an enum representing location source
	 */
	public EInsLocationSource getSource() {
		return mSource;
	}
	
	
	//========================================================================
	// SETTERS
	//========================================================================
	
	/**
	 * Copy values from aLocation. No allocation is done. 
	 * @param aLocation the Position to copy values from.
	 */
	public void setLocation(InsLocation aLocation) {
		mPosition.setMap(aLocation.getMapID());
		mPosition.setCoord(aLocation.getCoord());
		mAccuracy = aLocation.getAccuracy();
	}
		
	/**
	 * Set map id
	 * @param map map's id to set
	 */
	public void setMap(int map) {
		mPosition.setMap(map);
	}
	
	/**
	 * Set x/y pair
	 * @param coord x/y pair to set
	 */
	public void setCoord(PointD coord) {
		mPosition.setCoord(coord);
	}
	
	/**
	 * Set x value
	 * @param x the value to set
	 */
	public void setX(double x) {
		mPosition.setX(x);
	}
	
	/**
	 * Set y value
	 * @param y the value to set
	 */
	public void setY(double y) {
		mPosition.setX(y);
	}

	/**
	 * Set the accuracy radius
	 * @param accuracy accuracy radius
	 */
	public void setAccuracy(float accuracy) {
		mAccuracy = accuracy;
	}


	
}
