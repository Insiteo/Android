package com.insiteo.lbs.location;

import java.util.LinkedList;
import java.util.List;
import java.util.concurrent.Semaphore;

import android.content.res.Resources;
import android.graphics.Canvas;
import android.graphics.Point;

import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.map.entities.Map;
import com.insiteo.lbs.map.render.ERenderMode;
import com.insiteo.lbs.map.render.IRTO;
import com.insiteo.lbs.map.render.IRTOListener;
import com.insiteo.lbs.map.render.IRenderer;
import com.insiteo.lbs.map.render.Touch;
import com.threed.jpct.World;


/**
 * Class that can draw Insiteo Location on an Insiteo MapView
 */
public class LocationRenderer implements IRenderer {

	private int				mPriority		= 0;
	private int 			mDisplayAngle	= 0;
	protected Map			mCurrentMap		= null;
	protected boolean 		mIsDisplayed	= true;
	private boolean 		mIsTouchable	= false;
	protected Semaphore 	mRTOSema		= new Semaphore(1);
	private Resources		mResources		= null;

	// Custom colors
	private int mLocationColor 					= -1;
	private int mLocationLostColor 				= -1;
	private int mLocationAccuracyColor			= -1;
	private int mLocationAccuracyStrokeColor	= -1;

	protected List<GfxLocation> mLocations = new LinkedList<GfxLocation>();


	public LocationRenderer(Resources aResources) {
		mResources = aResources;
	}

	@Override
	public void setPriority(int aPriority) {
		mPriority = aPriority;
	}

	@Override
	public int getPriority() {
		return mPriority;
	}

	@Override
	public void setDisplayOrientation(int aDisplayOrientation) {
		mDisplayAngle = aDisplayOrientation;
	}

	@Override
	public void setCurrentMap(final Map aMap) {
		mCurrentMap = aMap;
	}

	@Override
	public Class<? extends IRTO> getRTOClass() {
		return GfxLocation.class;
	}

	@Override
	public void clear() {
		try {
			mRTOSema.acquire();
			mLocations.clear();
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mRTOSema.release();
		}
	}

	//******************************************************************************************************************
	// 	UI Custom
	// *****************************************************************************************************************

	/**
	 * Set the color used for the location.
	 * @param color
	 */
	public void setLocationColor(int color){
		mLocationColor = color;

		for (GfxLocation loc : mLocations) {
			loc.setLocationColor(mLocationColor);
		}
	}

	/**
	 * Set the color used for the location when lost.
	 * @param color
	 */
	public void setLocationLostColor(int color){
		mLocationLostColor = color;

		for (GfxLocation loc : mLocations) {
			loc.setLocationLostColor(mLocationLostColor);
		}
	}

	/**
	 * Set the color used for the location's accuracy circle.
	 * @param color
	 */
	public void setLocationAccuracyColor(int color){
		mLocationAccuracyColor = color;

		for (GfxLocation loc : mLocations) {
			loc.setLocationAccuracyColor(mLocationAccuracyColor);
		}
	}

	/**
	 * Set the color used for the location's accuracy circle.
	 * @param color
	 */
	public void setLocationAccuracyStrokeColor(int color){
		mLocationAccuracyStrokeColor = color;

		for (GfxLocation loc : mLocations) {
			loc.setLocationAccuracyStrokeColor(mLocationAccuracyColor);
		}
	}

	//******************************************************************************************************************
	//  IRTO & ZONE
	// *****************************************************************************************************************

	@Override
	public void clearZone(int aZoneId) {
		Log.d("LocationRenderer", "CLEAR ZONE : THIS SHOULD NEVER BE CALLED");
	}

	@Override
	public boolean addRTO(IRTO aObject) {
		boolean isAdded = false;
		if(aObject.getClass() == getRTOClass()) {
			try {
				mRTOSema.acquire();
				
				aObject.setRenderMode(mRenderMode);
				aObject.setResources(mResources);
				
				
				isAdded = mLocations.add((GfxLocation)aObject);

				if (mLocationColor != -1) {
					((GfxLocation)aObject).setLocationColor(mLocationColor);
				}

				if (mLocationLostColor != -1) {
					((GfxLocation)aObject).setLocationLostColor(mLocationLostColor);
				}

				if (mLocationAccuracyColor != -1) {
					((GfxLocation)aObject).setLocationAccuracyColor(mLocationAccuracyColor);
				}

				if (mLocationAccuracyStrokeColor != -1) {
					((GfxLocation)aObject).setLocationAccuracyStrokeColor(mLocationAccuracyStrokeColor);
				}
			}
			catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				mRTOSema.release();
			}
		}
		return isAdded;
	}

	@Override
	public void removeRTO(IRTO aObject) {
		if(aObject.getClass() == getRTOClass()) {
			try {
				mRTOSema.acquire();
				mLocations.remove(aObject);
				aObject.freeResources();
			}
			catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				mRTOSema.release();
			}
		}
	}

	@Override
	public boolean addRTOInZone(int aZoneId, IRTO aObject) {
		Log.d("LocationRenderer", "ADD OBJECT IN ZONE THIS SHOULD NEVER BE CALLED");
		return false;
	}

	@Override
	public void removeRTOFromZone(int aZoneId, IRTO aObject) {
		Log.d("LocationRenderer", "DEL OBJECT FROM ZONE THIS SHOULD NEVER BE CALLED");
	}

	//******************************************************************************************************************
	// 	DRAWING
	// *****************************************************************************************************************

	private ERenderMode mRenderMode;
	
	@Override
	public void setDisplayEnabled(boolean aOn) {
		mIsDisplayed = aOn;
	}

	@Override
	public boolean isDisplayEnabled() {
		return mIsDisplayed;
	}
	
	@Override
	public void setRenderMode(ERenderMode renderingMode) {
		mRenderMode = renderingMode;
		
		for (GfxLocation loc : mLocations) {
			loc.setRenderMode(renderingMode);
			loc.setResources(mResources);
		}
	}

	@Override
	public ERenderMode getRenderMode() {
		return mRenderMode;
	}

	//******************************************************************************************************************
	// 	DRAWING 2D
	// *****************************************************************************************************************

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if(mIsDisplayed) {
			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					if (mCurrentMap != null) {
						boolean isMapOriented = mCurrentMap.isOriented();
						float mapAzimuth = mCurrentMap.getAzimuth();
						int mapID = mCurrentMap.getId();

						for (GfxLocation loc : mLocations) {
							if (loc.getMapID() == mapID) { 
								loc.setMapAzimuth(isMapOriented ? mapAzimuth : null);	//set azimuth to null if map is not oriented
								loc.setDisplayOrientation(mDisplayAngle);
								loc.draw(aCanvas, aRatio, aOffset, aAngle);
							}
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

	//******************************************************************************************************************
	// 	DRAWING 3D
	// *****************************************************************************************************************

	public void draw3D(World world, double ratio, float angle){
		if(mIsDisplayed) {
			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					if (mCurrentMap != null) {
						boolean isMapOriented = mCurrentMap.isOriented();
						float mapAzimuth = mCurrentMap.getAzimuth();
						int mapID = mCurrentMap.getId();

						for (GfxLocation loc : mLocations) {
							if (loc.getMapID() == mapID) { 
								loc.setMapAzimuth(isMapOriented ? mapAzimuth : null);	//set azimuth to null if map is not oriented
								loc.setDisplayOrientation(mDisplayAngle);
								loc.draw3D(world, mCurrentMap, ratio, angle);
							}
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

	//===========================================================
	// TOUCH Management
	//===========================================================

	@Override
	public void setRTOListener(IRTOListener listener) {
		//loc is not touchable
	}

	@Override
	public void setTouchEnabled(boolean aOn) {
		//loc is not touchable
	}

	@Override
	public boolean isTouchEnabled() {
		return mIsTouchable;
	}

	@Override
	public boolean onTouchDown(Touch aTouch) {
		return false;
	}
	
	@Override
	public boolean onTouchPointerDown(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchMove(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchPointerUp(Touch aTouch) {
		return false;
	}
	
	@Override
	public boolean onTouchUp(Touch aTouch) {
		return false;
	}


}
