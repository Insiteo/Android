	public void setLocation(InsLocation aLoc) {
		if (mLocation != null && aLoc != null && mLocation.getMapID() == aLoc.getMapID()) {

			// If previously the location was lost we reset the proper colors
			if (isLocationLost) {
				isLocationLost = false;
				setDrawingColors();
			}

			//animate location from old point to new one
			//compute distance between the two positions
			final double xDistance = (aLoc.getCoord().x - mLocation.getCoord().x) / CommonConstants.COS_SQUARE_THETA_SUM;
			final double yDistance = (aLoc.getCoord().y - mLocation.getCoord().y) / CommonConstants.COS_SQUARE_THETA_SUM;
			//set new location with old position
			aLoc.setCoord(mLocation.getCoord());
			mLocation = aLoc;
			//get a reference on the position object, and run a thread to update it
			final PointD coord = mLocation.getCoord();
			new Thread(new Runnable() {
				@Override public void run() {
					for (double d : CommonConstants.COS_SQUARE_THETA) {
						coord.x += xDistance * d;
						coord.y += yDistance * d;
						try {
							Thread.sleep(10);
						} 
						catch (InterruptedException e) {
							e.printStackTrace();
						}
					}
				}
			}).start();
		} else {
			mLocation = aLoc;
		}

		if(aLoc == null) mAzimuth = null;

		mIsLocUpdated = true;
	}

	public void setLocationLost(InsLocation aLastKnownLoc) {
		setLocation(aLastKnownLoc);

		if (!isLocationLost) {
			isLocationLost = true;	
			setDrawingColors();
		}
	}

	public void setAzimuth(Float aAzimuth) {
		if(mRenderMode == ERenderMode.RENDER_MODE_3D) {
			if(mAzimuth == null) {
				mDrawingAngle = mMapAzimuth + mDisplayAngle;
				mDrawingAngle += aAzimuth;
			} else {
				mDrawingAngle += aAzimuth - mAzimuth;
			}

		}

		mAzimuth = aAzimuth;
	}

	private void set2DResources(Resources aResources) {

		mFovImage = BitmapFactory.decodeResource(aResources, R.drawable.compass);

		mLocationRadius = aResources.getDimensionPixelSize(R.dimen.insiteo_location_position_size);		

		mStrokeSize = aResources.getDimensionPixelSize(R.dimen.insiteo_location_position_stroke_size);
		mPaint.setStrokeWidth(mStrokeSize);

		mAccuracyPaint.setStrokeWidth(aResources.getDimensionPixelSize(R.dimen.insiteo_location_acc_stroke_width));

		mStrokeColor = aResources.getColor(R.color.insiteo_location_stroke);
		mLocationColor = aResources.getColor(R.color.insiteo_location_userlocation);

		mLocationAccuracyStrokeColor = aResources.getColor(R.color.insiteo_location_accuracy_stroke_color);

	private void setDrawingColors() {
		mColor = (isLocationLost) ? mLocationLostColor : mLocationColor;
		mFovPaint.setColorFilter(new PorterDuffColorFilter(mColor, Mode.MULTIPLY));

		if(mRenderMode == ERenderMode.RENDER_MODE_3D) setLocationTexture();
	}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if (mLocation != null && mFovImage != null) {
			//calculate position on screen
			int posX =  (int) (mLocation.getCoord().x * aRatio) + aOffset.x;
			int posY =  (int) (mLocation.getCoord().y * aRatio) + aOffset.y;
			float accRadius = (float)(mLocation.getAccuracy() * aRatio);

			drawAccuracyCircle(aCanvas, posX, posY, accRadius);
			drawCompass(aCanvas, posX, posY);
			drawPosition(aCanvas, posX, posY);
		}
	}

	private void drawAccuracyCircle(Canvas canvas, float x, float y, float radius) {
		mAccuracyPaint.setStyle(Style.FILL);
		mAccuracyPaint.setColor(mLocationAccuracyColor);
		canvas.drawCircle(x, y, radius, mAccuracyPaint);
		mAccuracyPaint.setStyle(Style.STROKE);
		mAccuracyPaint.setColor(mLocationAccuracyStrokeColor);
		canvas.drawCircle(x, y, radius, mAccuracyPaint);
	}

	private void drawCompass(Canvas canvas, float x, float y) {
		canvas.save();

		if (mAzimuth != null && mMapAzimuth != null) {

			float angle = computeDrawingAzimuth();

			canvas.rotate(angle - mMapAzimuth + mDisplayAngle, x, y);
			canvas.drawBitmap(mFovImage, x - mFovImage.getWidth() / 2, y - mLocationRadius - mFovImage.getHeight(), mFovPaint);
		}

		canvas.restore();
	}

	private void drawPosition(Canvas canvas, float x, float y) {
		mPaint.setColor(mColor);
		mPaint.setStyle(Style.FILL);
		canvas.drawCircle(x, y, (float) mLocationRadius, mPaint);
		mPaint.setColor(mStrokeColor);
		mPaint.setStyle(Style.STROKE);
		canvas.drawCircle(x, y, (float) mLocationRadius, mPaint);
	}

	private void create3DObject() {

		//		Texture color = new Texture(1, 1, ColorUtils.getRGBColor(mLocationColor));

		setLocationTexture();


		int width = MapUtils.closestTwoPower((int) (2 * mLocationRadius));

		m3DObject = Primitives.getPlane(width, 1);
		m3DObject.setTexture(TEXTURE);
		m3DObject.setTransparency(20);
//		m3DObject.setBillboarding(Object3D.BILLBOARDING_ENABLED);
//		m3DObject.setSortOffset(0);
//				m3DObject.calcTextureWrapSpherical();
		m3DObject.setOrigin(new SimpleVector(0, 0, -0.5));


	}

	private void setLocationTexture() {

		Bitmap.Config config = Bitmap.Config.ARGB_8888;

		int size = MapUtils.closestTwoPower((int) (2 * (mLocationRadius + mStrokeSize + mFovImage.getHeight())));

		Bitmap bitmap = Bitmap.createBitmap(size, size, config);		
		Canvas canvas = new Canvas(bitmap);

		canvas.drawBitmap(mFovImage, size / 2 - mFovImage.getWidth() / 2, size / 2 - mLocationRadius - mFovImage.getHeight(), mFovPaint);
		drawPosition(canvas, size / 2, size / 2);

		Texture locTexture = new Texture(bitmap, true);

		if (TextureManager.getInstance().containsTexture(TEXTURE)) {
			TextureManager.getInstance().replaceTexture(TEXTURE, locTexture);
		} else {
			TextureManager.getInstance().addTexture(TEXTURE, locTexture);
		}
	}
