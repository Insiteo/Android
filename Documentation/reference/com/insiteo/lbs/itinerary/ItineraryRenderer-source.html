package com.insiteo.lbs.itinerary;

import java.util.LinkedList;
import java.util.List;
import java.util.concurrent.Semaphore;

import android.content.res.Resources;
import android.graphics.Canvas;
import android.graphics.Point;

import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.map.entities.Map;
import com.insiteo.lbs.map.render.ERenderMode;
import com.insiteo.lbs.map.render.ETouchObjectResult;
import com.insiteo.lbs.map.render.IRTO;
import com.insiteo.lbs.map.render.IRTOListener;
import com.insiteo.lbs.map.render.IRenderer;
import com.insiteo.lbs.map.render.Touch;
import com.threed.jpct.World;


/**
 * Renderer for itinerary. This class is meant to be used with com.insiteo.map.MapView
 * If you want the itinerary to react to touch events, you need to set an IItineraryRendererListener with setListener().
 */
public class ItineraryRenderer implements IRenderer {

	private int				mPriority		= 0;
	private Map		mCurrentMap		= null;
	private boolean 		mIsDisplayed	= true;
	private boolean 		mIsTouchable	= true;
	private Semaphore 		mRTOSema		= new Semaphore(1);
	private Resources 		mResources 		= null;

	private List<GfxItinerary> mItineraries = new LinkedList<GfxItinerary>();
	private IItineraryRendererListener mListener = null;
	private boolean mLinkToDeparture = false;
	private boolean mLinkToArrival = false;
	private boolean mAnimateDrawing = false;


	/**
	 * Constructor
	 * @param aResources resources used to get images and strings
	 */
	public ItineraryRenderer(Resources aResources) {
		mResources = aResources;
	}

	/**
	 * Set listener to be notified on itinerary graphical events. 
	 * @param aListener the listener that will receive notifications. If null, no notification will be sent.
	 */
	public void setListener(IItineraryRendererListener aListener) {
		mListener = aListener;
		try {
			mRTOSema.acquire();
			for (GfxItinerary iti : mItineraries) {
				iti.setListener(aListener);
			}
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	/**
	 * Use this method to display a line between itinerary points and departure and arrival positions.
	 * @param aLinkToDeparture if true, first itinerary point drawn will be departure point.
	 * @param aLinkToArrival if true, last itinerary point drawn will be destination point.
	 */
	public void setLinkToEnds(boolean aLinkToDeparture, boolean aLinkToArrival) {
		mLinkToDeparture = aLinkToDeparture;
		mLinkToArrival = aLinkToArrival;
		try {
			mRTOSema.acquire();
			for (GfxItinerary iti : mItineraries) {
				iti.setLinkToEnds(mLinkToDeparture, mLinkToArrival);
			}
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	/**
	 * If set to true the itinerary will be drawn with an animation
	 * @param aEnable True if it should use animation.
	 */
	public void setAnimated(boolean aEnable) {
		mAnimateDrawing = aEnable;
		try {
			mRTOSema.acquire();
			for (GfxItinerary iti : mItineraries) {
				iti.setAnimated(mAnimateDrawing);
			}
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	/**
	 * Set the index of the selected instruction (thus updating rendering)
	 * @param aInstructionIndex the index of the instruction that will be active 
	 */
	public void setSelectedInstructionIndex(int aInstructionIndex) {
		//TODO : handle multi itinerary (add a param containing Itinerary ID)
		//TODO : mutex ?
		GfxItinerary iti = mItineraries.get(0);
		iti.setSelectedInstructionIndex(aInstructionIndex);
	}

	@Override
	public void setPriority(int aPriority) {
		mPriority = aPriority;
	}

	@Override
	public int getPriority() {
		return mPriority;
	}



	@Override
	public void setDisplayOrientation(int aDisplayOrientation) {
	}

	@Override
	public void setCurrentMap(final Map aMap) {
		mCurrentMap = aMap;
	}

	@Override
	public Class<? extends IRTO> getRTOClass() {
		return GfxItinerary.class;
	}

	@Override
	public void clear() {
		try {
			mRTOSema.acquire();
			mItineraries.clear();
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}



	//******************************************************************************************************************
	//  IRTO & ZONE
	// *****************************************************************************************************************

	@Override
	public void clearZone(int aZoneId) {
		Log.d("ItineraryRenderer", "CLEAR ZONE : THIS SHOULD NEVER BE CALLED");
	}

	@Override
	public boolean addRTO(IRTO aObject) {
		boolean isAdded = false;
		if(aObject.getClass() == GfxItinerary.class) {
			try {
				mRTOSema.acquire();
				GfxItinerary iti = (GfxItinerary)aObject;
				iti.setResources(mResources);
				iti.setListener(mListener);
				iti.setLinkToEnds(mLinkToDeparture, mLinkToArrival);
				iti.setAnimated(mAnimateDrawing);
				isAdded = mItineraries.add(iti);
			}
			catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				mRTOSema.release();
			}
		}
		return isAdded;
	}

	@Override
	public void removeRTO(IRTO aObject) {
		if(aObject.getClass() == GfxItinerary.class) {
			try {
				mRTOSema.acquire();
				mItineraries.remove(aObject);
				aObject.freeResources();
			}
			catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				mRTOSema.release();
			}
		}
	}

	@Override
	public boolean addRTOInZone(int aZoneId, IRTO aObject) {
		Log.d("ItineraryRenderer", "ADD OBJECT IN ZONE THIS SHOULD NEVER BE CALLED");
		return false;
	}

	@Override
	public void removeRTOFromZone(int aZoneId, IRTO aObject) {
		Log.d("ItineraryRenderer", "DEL OBJECT FROM ZONE THIS SHOULD NEVER BE CALLED");
	}


	//******************************************************************************************************************
	//  DRAWING
	// *****************************************************************************************************************

	private ERenderMode mRenderMode;
	
	@Override
	public void setDisplayEnabled(boolean aOn) {
		mIsDisplayed = aOn;
	}

	@Override
	public boolean isDisplayEnabled() {
		return mIsDisplayed;
	}

	@Override
	public void setRenderMode(ERenderMode renderingMode) {
		mRenderMode = renderingMode;		
	}

	@Override
	public ERenderMode getRenderMode() {
		return mRenderMode;
	}
	
	//******************************************************************************************************************
	//  DRAWING 2D
	// *****************************************************************************************************************

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if(mIsDisplayed) {
			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					if (mCurrentMap != null) {
						int mapID = mCurrentMap.getId();
						for (GfxItinerary iti : mItineraries) {
							iti.setCurrentMap(mapID);
							iti.draw(aCanvas, aRatio, aOffset, aAngle);
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

	//******************************************************************************************************************
	//  DRAWING 3D
	// *****************************************************************************************************************

	public void draw3D(World world, double ratio, float angle){

	}


	//******************************************************************************************************************
	//  TOUCH
	// *****************************************************************************************************************

	@Override
	public void setRTOListener(IRTOListener listener) {
		//do nothing
	}

	@Override
	public void setTouchEnabled(boolean aOn) {
		mIsTouchable = aOn;
	}

	@Override
	public boolean isTouchEnabled() {
		return mIsTouchable;
	}

	@Override
	public boolean onTouchDown(Touch aTouch) {
		//TODO : mutex ?
		boolean res = false;
		if (mIsTouchable && mListener != null) {
			for (GfxItinerary iti : mItineraries) {
				ETouchObjectResult tor = iti.onTouchDown(aTouch);
				if (tor == ETouchObjectResult.RESULT_CONSUME || tor == ETouchObjectResult.RESULT_NOTIFY_CONSUME) {
					res = true;
					break;
				}
			}
		}
		return res;
	}

	@Override
	public boolean onTouchPointerDown(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchMove(Touch aTouch) {
		//TODO : mutex ?
		boolean res = false;
		if (mIsTouchable && mListener != null) {
			for (GfxItinerary iti : mItineraries) {
				ETouchObjectResult tor = iti.onTouchMove(aTouch);
				if (tor == ETouchObjectResult.RESULT_CONSUME || tor == ETouchObjectResult.RESULT_NOTIFY_CONSUME) {
					res = true;
					break;
				}
			}
		}
		return res;
	}

	@Override
	public boolean onTouchPointerUp(Touch aTouch) {
		return false;
	}
	
	@Override
	public boolean onTouchUp(Touch aTouch) {
		//TODO : mutex ?
		boolean res = false;
		if (mIsTouchable && mListener != null) {
			for (GfxItinerary iti : mItineraries) {
				ETouchObjectResult tor = iti.onTouchUp(aTouch);
				if (tor == ETouchObjectResult.RESULT_CONSUME || tor == ETouchObjectResult.RESULT_NOTIFY_CONSUME) {
					res = true;
					break;
				}
			}
		}
		return res;
	}

	public boolean onTouch3D(Object obj){
		return false;
	}

}
