package com.insiteo.lbs.itinerary.entities;

import java.util.ArrayList;

import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.utils.geometry.Position;


/**
 * Class containing itinerary data. It is composed of departure and arrival points, plus a list of instructions (that can be empty).
 */
public class Itinerary {

	private int mID;
	
	/** The list of instructions for this itinerary */
	private ArrayList<Instruction> mInstructions;
	
	/** Total time (in milliseconds) from departure to arrival */
	private int mTotalTime;
	
	/** Total distance (in millimeters) from departure to arrival */
	private int mTotalDistance;
	
	/** Remaining time (in milliseconds) from current location to arrival */
	private int mRemainingTime;
	
	/** Remaining distance (in millimeters) from current location to arrival */
	private int mRemainingDistance;
	
	private Position mDestination;
	private Position mDeparture;
	
	/** Total number of steps in this Itinerary */
	private int mStepsCount = 0;
	
	/** Index of the current instruction, based on current location */
	private int mCurrentInstructionIndex = 0;
	
	/** Index of the current section in the current instruction, based on current location */
	private int mCurrentSectionIndex = 0;
	
	/** Dynamic information for current instruction */
	private String mCurrentInstructionInfo = "";
	

	
	/** Constructor */
	public Itinerary(ArrayList<Instruction> aInstructions, int aTotalTime, int aTotalDistance, Position aDeparture, Position aDestination) {
		mID = CommonConstants.NULL_ID;
		mInstructions = aInstructions;
		mTotalTime = aTotalTime;
		mTotalDistance = aTotalDistance;
		mRemainingTime = aTotalTime;
		mRemainingDistance = aTotalDistance;
		mDeparture = aDeparture;
		mDestination = aDestination;
		for (Instruction i : aInstructions) {
			mStepsCount += i.steps.size();
		}
		mCurrentInstructionIndex = 0;
		mCurrentSectionIndex = 0;
		if (aInstructions.size() > 0) {
			mCurrentInstructionInfo = aInstructions.get(0).information;
		}
	}
	
	
	/**
	 * @hide
	 * Internal use only ! Application must not use this method.
	 * This will only set instructions. Other members must be initialized afterward.
	 */
	public Itinerary(Instruction[] aInstructions) {
		mID = CommonConstants.NULL_ID;
		mInstructions = new ArrayList<Instruction>(aInstructions.length);
		for (Instruction i : aInstructions) {
			mInstructions.add(i);
			mStepsCount += i.steps.size();
		}
		mCurrentInstructionIndex = 0;
		mCurrentSectionIndex = 0;
		mTotalTime = 0;
		mTotalDistance = 0;
		mRemainingTime = 0;
		mRemainingDistance = 0;
		mDestination = null;
		mDeparture = null;
		mCurrentInstructionInfo = null;
	}

	/**
	 * Set the Itinerary ID
	 * @param aID the ID to set
	 */
	public void setID(int aID) {
		mID = aID;
	}
	
	/**
	 * @return the ID of the itinerary
	 */
	public int getID() {
		return mID;
	}
	
	/**
	 * @return true if this itinerary has instructions (departure and arrival are not considered)
	 */
	public boolean hasInstruction() {
		return mInstructions != null && mInstructions.size() > 0;
	}
	
	/**
	 * Get the list of instructions for this itinerary (without departure and arrival)
	 * @return the list of instructions
	 */
	public ArrayList<Instruction> getInstructions() {
		return mInstructions;
	}
	
	/**
	 * Get the time from departure to arrival
	 * @return the total time to arrival in milliseconds
	 */
	public int getTotalTime() {
		return mTotalTime;
	}

	/**
	 * Get the total distance from departure to arrival
	 * @return the total distance to arrival in millimeters
	 */
	public int getTotalDistance() {
		return mTotalDistance;
	}

	/**
	 * Set the time from current location to arrival
	 * @param aRemainingTime the remaining time to arrival in milliseconds 
	 */
	public void setRemainingTime(int aRemainingTime) {
		mRemainingTime = aRemainingTime;
	}
		
	/**
	 * Get the time from current location (or departure if no location is set) to arrival
	 * @return the remaining time to arrival in milliseconds
	 */
	public int getRemainingTime() {
		return mRemainingTime;
	}

	/**
	 * Set the distance from current location to arrival
	 * @param aRemainingDistance the remaining distance to arrival in millimeters 
	 */
	public void setRemainingDistance(int aRemainingDistance) {
		mRemainingDistance = aRemainingDistance;
	}
		
	/**
	 * Get the remaining distance from current location (or departure if no location is set) to arrival
	 * @return the remaining distance to arrival in millimeters
	 */
	public int getRemainingDistance() {
		return mRemainingDistance;
	}
	
	/**
	 * Get the departure point for this itinerary
	 * @return the departure point given through itinerary request
	 */
	public Position getDeparture() {
		return mDeparture;
	}
	
	/**
	 * Get the destination point for this itinerary
	 * @return the destination point given through itinerary request
	 */
	public Position getDestination() {
		return mDestination;
	}

	/**
	 * Get the total number of steps in this itinerary (departure and arrival are not considered)
	 * @return the number of steps in this itinerary
	 */
	public int getStepsCount() {
		return mStepsCount;
	}
	
	/**
	 * Set the current instruction index
	 * @param aInstructionIndex the current instruction index to set
	 */
	public void setCurrentInstructionIndex(int aInstructionIndex) {
		mCurrentInstructionIndex = aInstructionIndex;
	}
	
	/**
	 * Get the current instruction index, based on current location.
	 * Instructions at indexes lesser than this one are already traveled along. 
	 * @return the current instruction index.
	 */
	public int getCurentInstructionIndex() {
		return mCurrentInstructionIndex;
	}
	
	/**
	 * Set the current section index
	 * @param aSectionIndex the current section index (in current instruction) to set
	 */
	public void setCurrentSectionIndex(int aSectionIndex) {
		mCurrentSectionIndex = aSectionIndex;
	}
	
	/**
	 * Get the current section index, based on current location.
	 * Sections at indexes lesser than this one, in the current instruction, are already traveled along. 
	 * @return the current section index.
	 */
	public int getCurrentSectionIndex() {
		return mCurrentSectionIndex;
	}
	
	/**
	 * Set the temporary information for current instruction.
	 * This is a temporary information, based on current location. Instruction's information will not be affected by this.
	 * @param aCurrentInstructionInfo the temporary information for current instruction
	 */
	public void setCurrentInstructionInfo(String aCurrentInstructionInfo) {
		mCurrentInstructionInfo = aCurrentInstructionInfo;
	}
	
	/**
	 * Get the current instruction information. 
	 * This is a temporary information, based on current location. 
	 * @return the current instruction temporary information.
	 */
	public String getCurrentInstructionInfo() {
		return mCurrentInstructionInfo;
	}
	
	public String toString(){
		String res = "Itinerary = ";
		if (mInstructions != null) {
			for (int i = 0; i < mInstructions.size(); i++) {
				res += "\n " + mInstructions.get(i).toString();
			}
		}
		
		return res;
	}
}
