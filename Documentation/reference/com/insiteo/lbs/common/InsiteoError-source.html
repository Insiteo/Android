package com.insiteo.lbs.common;

import android.content.res.Resources;
import android.content.res.Resources.NotFoundException;



/**
 * Class containing Insiteo error information. Use {@link #toString(Resources)} or {@link #toString()} to get the message associated to this error.
 * It contains :
 * <ul>
 * <li>a code, determining the error type</li>
 * <li>a message, containing a readable string describing the error</li>
 * <li>a messageId : if different from {@link CommonConstants#NULL_ID}, it gives the id of a string resource containing localized error description</li>
 * </ul>
 * If the InsiteoError contains a message, it will not contain a messageId.
 * If the InsiteoError contains a messageId, it will not contain a message. 
 */
public class InsiteoError {

	public final static String NO_ERROR_CODE 				= "XMLLNO";
	public final static String COMM_ERROR_CODE 				= "XMLLCOMM";
	public final static String PARSING_ERROR_CODE 			= "XMLLPARS";
	public final static String PARSINGERROR_ERROR_CODE 		= "XMLLPARSERR";
	public final static String PULLPARSER_ERROR_CODE 		= "XMLLPULLP";
	public final static String API_NOT_INIT_ERROR_CODE 		= "APINOTINIT";
	public final static String EMBEDDED_INIT_ERROR_CODE 	= "EMBINIERR";
	public final static String SESSION_ERROR_CODE 			= "LBSSessionInvalid";
	public final static String CONTENT_VERSION_ERROR_CODE 	= "ContentVersionTooOld";
	public final static String NO_DEPARTURE_ERROR_CODE 		= "ITINODEPART";
	public final static String NO_DESTINATION_ERROR_CODE 	= "ITINODESTINATION";
	public final static String NO_ITINERARY_PACKAGE		 	= "NOITINERARYPACKAGE";
	public final static String DATABASE_ERROR_CODE 			= "DBERR";
	public final static String LBS_MODULE_ABSENT_ERROR_CODE = "MODABSERR";
	public final static String NO_SDCARD_ERROR_CODE 		= "NOSDCARD";
	public final static String NO_SDCARD_SPACE_ERROR_CODE 	= "NOSDCARDSPACE";
	public final static String CANCELED_ERROR_CODE 			= "CANCELERR";
	public final static String MD5_ERROR_CODE 				= "MD5ERR";
	public final static String PACKAGE_TYPE_ERROR_CODE 		= "UNKWNTYPERR";
	public final static String UNKNOWN_ERROR_CODE 			= "UNKWNERR";
	public final static String NO_PACKAGE_DL_ERROR_CODE 	= "NOPACKDL";
	public final static String EMBEDDED_ITI_ERROR_CODE 		= "EMBITIERR";
	public final static String FILE_NOT_FOUND_ERROR_CODE 	= "FNFERR";
	public final static String HTTP_ERROR_CODE 				= "HTTPERR";
	public final static String HTTP_ERROR_CODE_FORBIDDEN	= "HTTPERR403";
	public final static String HTTP_ERROR_CODE_NOT_FOUND	= "HTTPERR404";
	public final static String HTTP_ERROR_CODE_UNAUTHORIZED	= "HTTPERR401";
	public final static String POST_NO_CONTENT_ERROR_CODE 	= "POSTEMPTYERR";
	public final static String EMBEDDED_OPTIM_ERROR_CODE 	= "EMBOPTIMERR";
	public final static String LOC_FLAGS_ERROR				= "LOCFLAGSERR";
	

	public final static String LOC_SERVICE_NOT_DECLARED		= "LOCSERVICEDECLARATIONERR";
	
	// protobuf error
	public final static String PROTO_URL_ERROR_CODE 		= "ProtoURLErrorCode";
	public final static String BAD_CLIENT_DATA_ERROR        = "DispatchBadClientData";
	
	private String mCode;
	private String mMessage;
	private int mMessageResId;
	
	public InsiteoError(String code, String message) {
		mCode = code;
		mMessage = message;
		mMessageResId = CommonConstants.NULL_ID;
	}
	
	public InsiteoError(String code, int messageId) {
		mCode = code;
		mMessage = "";
		mMessageResId = messageId;
	}
	
	public String getCode() {
		return mCode;
	}
	
	public String getMessage() {
		return mMessage;
	}
	
	public int getMessageResId() {
		return mMessageResId;
	}
	
	/**
	 * Get the message associated to this error. 
	 * You should better use {@link #toString(Resources)} when possible.
	 * @return the message associated to this error, not localized.
	 */
	@Override
	public String toString() {
		if (mMessage != null) {
			return mMessage;
		}
		else {
			return "";
		}
	}
	
	/**
	 * Get the message associated to this error.
	 * @param aResources application resources
	 * @return the message associated to this error, localized.
	 */
	public String toString(Resources aResources) {
		String res;
		if (aResources == null) {
			res = toString();
		}
		else {
			if (mMessageResId == CommonConstants.NULL_ID) {
				res = toString();
			}
			else {
				try  {
					res = aResources.getString(mMessageResId);
				}
				catch (NotFoundException e) {
					//resource does not exists : try getting string message
					res = toString();
				}
			}			
		}
		return res;
	}
}
