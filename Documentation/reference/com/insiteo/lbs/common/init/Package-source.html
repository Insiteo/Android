package com.insiteo.lbs.common.init;

import java.io.File;

import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.utils.FileUtils;

/**
 * Class storing information about an Insiteo package.
 */
public class Package {

	private EPackageType mType;
	private String mUrl;
	private int mVersion;
	private String mMD5;
	private long mSize;
	private String mFileName;
	
	/**
	 * @hide
	 * Default constructor. All values are set to <code>null</code>
	 */
	public Package() {
		mType = EPackageType.UNKNOWN;
		mUrl = "";
		mVersion = CommonConstants.NULL_VERSION;
		mMD5 = "";
		mSize = 0;
		mFileName = null;
	}
	
	/**
	 * @hide
	 * Constructor
	 * @param aType the package type, as a String
	 * @param aUrl the package URL
	 * @param aVersion the package version
	 * @param aMD5 the package MD5 sum (as a String representation) 
	 * @param aSize the package size (in bytes)
	 */
	public Package(String aType, String aUrl, int aVersion, String aMD5, long aSize) {
		this(EPackageType.getFromValue(aType), aUrl, aVersion, aMD5, aSize);
	}
	
	/**
	 * @hide
	 * Constructor
	 * @param aType the package type
	 * @param aUrl the package URL
	 * @param aVersion the package version
	 * @param aMD5 the package MD5 sum (as a String representation) 
	 * @param aSize the package size (in bytes)
	 */
	public Package(EPackageType aType, String aUrl, int aVersion, String aMD5, long aSize) {
		mType = aType;
		mUrl = aUrl;
		mVersion = aVersion;
		mMD5 = aMD5;
		mSize = aSize;
		//get filename from url
		int i = (mUrl == null) ? -1 : mUrl.lastIndexOf("/");
		if (i != -1 && i < mUrl.length()-1) {
			mFileName = mUrl.substring(i+1);
		}
		else {
			mFileName = mUrl;
		}
	}
	
	/**
	 * Return the {@link EPackageType} of this {@link Package}.
	 * @return The package type.
	 */
	public EPackageType getType() {
		return mType;
	}

	/**
	 * Return the URL used to download the file associated to this {@link Package}.
	 * @return The URL the file can be downloaded from
	 */
	public String getUrl() {
		return mUrl;
	}

	/**
	 * Return the version of this {@link Package}.
	 * @return the package version, as defined on Insiteo's server
	 */
	public int getVersion() {
		return mVersion;
	}

	/**
	 * Return the MD5 sum.
	 * @return the package MD5 sum, represented as a String
	 */
	public String getMD5() {
		return mMD5;
	}

	/**
	 * Return the package size.
	 * @return the package file size (in bytes)
	 */
	public long getSize() {
		return mSize;
	}
	
	/**
	 * Return the {@link Package} file name.
	 * @return the package file name
	 */
	public String getFileName() {
		return mFileName;
	}
	
	/**
	 * Check if a package of the given type with the given version is already deployed
	 * @param aDataDirPath the path to insiteo data directory
	 * @param aType the type of package to check
	 * @param aVersion the version of package to check
	 * @return true if the package is already deployed, false otherwise
	 */
	public static boolean isDeployed(String aDataDirPath, EPackageType aType, int aVersion) {
		boolean isDeployed = false;
		File f = new File(aDataDirPath + "/" + aType.getDirName() + "/" + CommonConstants.VERSION_FILENAME);
		String strVers = FileUtils.readTextFile(f);
		if (strVers != null) {
			try {
				int deployedVersion = Integer.parseInt(strVers);
				if (deployedVersion == aVersion) {
					isDeployed = true;
				}
			}
			catch (NumberFormatException e) {
				//nothing to do : package will be considered as not deployed
			}
		}		
		return isDeployed;
	}
	
}
