IAuthListener

package com.insiteo.lbs.common.auth;

import com.insiteo.lbs.common.InsiteoError;
import com.insiteo.lbs.common.auth.entities.Site;
import com.insiteo.lbs.common.auth.entities.User;

/**
 * Interface to be notified of authentication results.
 * @author Insiteo
 *
 */
public interface IAuthListener {
	
	/**
	 * Gets triggered when the login completed.
	 * @param success True if authentication succeeded.
	 * @param user The authenticated {@link User}.
	 * @param error The {@link InsiteoError} if something happened.
	 */
	public void onLoginDone(boolean success, User user, InsiteoError error);
	
	/**
	 * Gets triggered when the logout completed.
	 * @param success True if logout succeeded.
	 * @param error The {@link InsiteoError} if something happened.
	 */
	public void onLogoutDone(boolean success, InsiteoError error);
	
	/**
	 *  Gets triggered when the {@link Site} refresh is done.
	 */
	public void onRefreshDone();

}
