package com.insiteo.lbs.common.auth;

import android.os.AsyncTask;

import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.InsiteoError;
import com.insiteo.lbs.common.auth.entities.User;
import com.insiteo.lbs.common.auth.parsers.LoginRequestEncoder;
import com.insiteo.lbs.common.auth.parsers.LoginResponseParser;
import com.insiteo.lbs.common.init.InitProvider.EServerType;
import com.insiteo.lbs.common.parser.ProtoLoader;
import com.insiteo.lbs.common.utils.EncodeUtils;
import com.insiteo.lbs.common.utils.Log;

/**
 * Manager that should be used to authenticate a user to its Insiteo account.
 * @author Insiteo
 *
 */
public class AuthManager {

	private final static String TAG = AuthManager.class.getSimpleName();

	private final static String ENCRYPTION_SALT = "1ns1te0$r0cKsS*!||#";

	private static final String SERVER_KEY = "__SERVER_KEY__";

	private static final String BASE_URL = "https://services." + SERVER_KEY + ".insiteo.com/v" + 
			CommonConstants.URL_VERSION + "/Insiteo.Dispatch." + SERVER_KEY + "/Insiteo.Dispatch.svc";

	private static final String PROD_URL = "https://services2." + SERVER_KEY + ".insiteo.com/v" + 
			CommonConstants.URL_VERSION + "/Insiteo.Dispatch." + SERVER_KEY + "/Insiteo.Dispatch.svc";


	private static final String DEFAULT_LOGIN_URL = BASE_URL + "/rest/api/auth/login";
	private static final String DEFAULT_LOGOUT_URL = BASE_URL + "/rest/api/auth/logout";

	private static final String PROD_LOGIN_URL = PROD_URL + "/rest/api/auth/login";
	private static final String PROD_LOGOUT_URL = PROD_URL + "/rest/api/auth/logout";

	private static AuthManager sInstance;

	private EServerType mCurrentServer;
	private User mLoggedUser;

	private IAuthListener mListener;

	/**
	 * Returns the {@link AuthManager} singleton instance.
	 * @return the {@link AuthManager} instance.
	 */
	public static AuthManager getInstance(){
		if(sInstance == null){
			sInstance = new AuthManager();
		}
		return sInstance;
	}

	private AuthManager(){
	}

	/**
	 * Returns the current {@link EServerType} on which the {@link User} is authenticated.
	 * @return the current {@link EServerType}.
	 */
	public EServerType getServer(){
		return mCurrentServer;
	}

	/**
	 * Returns the current authenticated {@link User}.
	 * @return the authenticated {@link User}.
	 */
	public User getUser(){
		return mLoggedUser;
	}

	/**
	 * Tries to login with the given credentials. The {@link IAuthListener} will be notified of the result.
	 * @param server the {@link EServerType} to authenticate on.
	 * @param email the email of the {@link User}.
	 * @param password the account password.
	 * @param listener the {@link IAuthListener} that will be notified.
	 */
	public void login(EServerType server, String email, String password, IAuthListener listener) {
		mListener = listener;

		LoginTask loginTask = new LoginTask();
		loginTask.execute(server.toString(), email, password);
	}


	/**
	 * Tries to logout the current {@link User} from the given {@link EServerType}.
	 * @param server the {@link EServerType} to logout from.
	 * @param listener The {@link IAuthListener} that will be notified.
	 */
	public void logout(EServerType server, IAuthListener listener) {
		mListener = listener;
		
		LogoutTask logoutTask = new LogoutTask();
		logoutTask.execute(server.toString());
	}

	public static String encrypt(String email, String password) {
		return EncodeUtils.getMD5SumStr(new String(email
				+ ENCRYPTION_SALT + password).getBytes());
	}


	//******************************************************************************************************************
	// 	LOGIN
	// *****************************************************************************************************************

	/**
	 * Represents an asynchronous login/registration task used to authenticate
	 * the user.
	 */
	private class LoginTask extends AsyncTask<String, Void, Boolean> {

		EServerType server;
		String email, password;

		InsiteoError error = null;
		LoginResponseParser response = null;

		@Override
		protected Boolean doInBackground(String... params) {

			if(params.length < 3){
				Log.d(TAG, "Missing required parameters");
				return false;
			}

			server = EServerType.valueOf(params[0]);
			email = params[1];
			password = params[2];

			String url = (server == EServerType.PROD) ? PROD_LOGIN_URL.replaceAll(SERVER_KEY, server.toString()) : 
				DEFAULT_LOGIN_URL.replaceAll(SERVER_KEY, server.toString()); 

			LoginRequestEncoder req = new LoginRequestEncoder(email, password);
			response = new LoginResponseParser();

			error = ProtoLoader.load(url, req, response);

			return true;
		}

		@Override
		protected void onPostExecute(final Boolean success) {
			if (success) {
				if(error.getCode().equals(InsiteoError.NO_ERROR_CODE)){

					String encryptedPwd = encrypt(email, password);

					mLoggedUser = new User(email, encryptedPwd, response.getRole(), server, response.getUserSites());

					CommonConstants.API_KEY = response.getAPIKey();
					CommonConstants.INSITEO_TOKEN = response.getISToken();

					mCurrentServer = server;

					if (mListener != null) mListener.onLoginDone(true, mLoggedUser, error);
				} else {
					if (mListener != null) mListener.onLoginDone(false, null, error);					
				}
			} else {
				if (mListener != null) mListener.onLoginDone(false, null, error);					
			}			
		}

		@Override
		protected void onCancelled() {

		}
	}

	//******************************************************************************************************************
	// 	LOGOUT
	// *****************************************************************************************************************


	private class LogoutTask extends AsyncTask<String, Void, Boolean> {

		EServerType server;
		InsiteoError error = null;

		@Override
		protected Boolean doInBackground(String... params) {

			server = EServerType.valueOf(params[0]);
			
			String url = (server == EServerType.PROD) ? PROD_LOGOUT_URL.replaceAll(SERVER_KEY, server.toString()) : 
				DEFAULT_LOGOUT_URL.replaceAll(SERVER_KEY, server.toString()); 

			error = ProtoLoader.load(url, null);

			return true;
		}

		@Override
		protected void onPostExecute(final Boolean success) {
			if (mListener != null) {
				mListener.onLogoutDone(success, error);
			}								
		}

		@Override
		protected void onCancelled() {

		}
	}
}
