package com.insiteo.lbs.common.auth.entities;

import java.util.ArrayList;
import java.util.List;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;

import com.insiteo.lbs.protobuf.dispatch.response.auth.ProtoRepDispatchLoginRootContent.PBRepUserSite;

/**
 * Class describing an Insiteo site 
 * @author Insiteo
 *
 */
public class Site {

	private int mId;
	private String mLabel;

	private List<Integer> mVersions;
	private List<String> mLangs;

	private int mVersion;
	private String mLang;


	/**
	 * @hide
	 */
	public Site(int id, String label, List<Integer> versions, List<String> langs) {
		super();
		mId = id;
		mLabel = label;
		mVersions = versions;
		mLangs = langs;
	}

	/**
	 * @hide
	 * @param site
	 */
	public Site(PBRepUserSite site){
		mId = site.getIdSite();
		mLabel = site.getSiteLabel();

		mVersions = site.getVersionsList();
		mLangs = site.getLangList();
	}

	/**
	 * Returns the id of the {@link Site}.
	 * @return the id.
	 */
	public int getId(){
		return mId;
	}

	/**
	 * Returns the label of the {@link Site}.
	 * @return the label.
	 */
	public String getLabel(){
		return mLabel;
	}

	/**
	 * Returns the version of the {@link Site}.
	 * @return the version.
	 */
	public int getVersion() {
		return mVersion;
	}

	/**
	 * Sets the version of the {@link Site}.
	 * @return the id.
	 */
	public void setVersion(int version){
		mVersion = version;
	}

	/**
	 * Returns the last used lang of the {@link Site}.
	 * @return the lang.
	 */
	public String getLang() {
		return mLang;
	}

	/**
	 * Sets the last lang of the {@link Site}.
	 * @return the id.
	 */
	public void setLang(String lang) {
		mLang = lang;
	}
	
	/**
	 * Returns the {@link List} of all the langs available for this site.
	 * @return
	 */
	public List<String> getLangs() {
		return mLangs;
	}

	@Override
	public String toString(){
		return mLabel + " - " + mId;
	}
	
	//******************************************************************************************************************
	// 	JSON
	// *****************************************************************************************************************

	final static String ID  			= "SITE_ID";
	final static String LABEL  			= "SITE_LABEL";
	final static String VERSIONS  		= "SITE_VERSIONS";
	final static String LANGS  			= "SITE_LANGS";

	/**
	 * Returns the corresponding {@link JSONObject}.
	 * @return the encoded {@link JSONObject}.
	 */
	public JSONObject encode(){
		JSONObject obj = new JSONObject();
		try {
			obj.put(ID, mId);
			obj.put(LABEL, mLabel);

			JSONArray versions = new JSONArray();
			for (Integer version : mVersions) {
				versions.put(version);
			}
			obj.put(VERSIONS, versions);

			JSONArray langs = new JSONArray();
			for (String lang : mLangs) {
				langs.put(lang);
			}
			obj.put(LANGS, langs);
		} catch (JSONException e) {
			e.printStackTrace();
		}

		return obj;

	}

	/**
	 * Returns the {@link Site} corresponding to this {@link JSONObject}.
	 * @param json the {@link JSONObject} to decode.
	 * @return the decoded {@link Site}.
	 */
	public static Site decode(JSONObject json){
		Site site = null;

		if (json != null) {
			try {
				site = new Site(json.getInt(ID), json.getString(LABEL), decodeVersions(json), decodeLangs(json));
			} catch (JSONException e) {
				e.printStackTrace();
			}

		}

		return site;
	}

	private static List<Integer> decodeVersions(JSONObject json){
		List<Integer> result = new ArrayList<Integer>();
		try {
			JSONArray versionJson = json.getJSONArray(VERSIONS);

			for (int i = 0; i < versionJson.length(); i++) {
				int version = versionJson.getInt(i);
				result.add(version);
			}

		} catch (JSONException e) {
			e.printStackTrace();
		}
		return result;		
	}

	private static List<String> decodeLangs(JSONObject json){
		List<String> result = new ArrayList<String>();
		try {
			JSONArray langsJson = json.getJSONArray(LANGS);

			for (int i = 0; i < langsJson.length(); i++) {
				String lang = langsJson.getString(i);
				result.add(lang);
			}

		} catch (JSONException e) {
			e.printStackTrace();
		}
		return result;
	}
}
