package com.insiteo.lbs.common.auth.entities;

public class User {

	private String mEmail;
	private String mEncryptedPassword;


	private HashMap<EServerType, EUserRole> mRoles;
	private HashMap<EServerType, List<Site>> mSites;

	private EServerType mServer = EServerType.PROD;

	/**
	 * 
	 * @param login
	 * @param password
	 * @param role
	 * @param server
	 * @param sites
	 */
	public User(String login, String password, EUserRole role, EServerType server, List<PBRepUserSite> sites) {
		super();

		mEmail = login;
		mEncryptedPassword = password;

		mServer = server;

		mRoles = new HashMap<InitProvider.EServerType, EUserRole>();
		mRoles.put(server, role);

		List<Site> serverSites = new ArrayList<Site>();

		for (PBRepUserSite site : sites) {
			serverSites.add(new Site(site));
		}
		mSites = new HashMap<InitProvider.EServerType, List<Site>>();
		mSites.put(server, serverSites);
	}

	final static String LOGIN = "USER_LOGIN";
	final static String PASSWORD = "USER_PASSWORD";
	final static String SERVER = "SERVER";
	final static String ROLES = "USER_ROLES";
	final static String SITES = "USER_SITES";

	/**
	 * Returns the corresponding {@link JSONObject}.
	 * @return the encoded {@link JSONObject}.
	 */
	public JSONObject encode() {
		JSONObject userJson = new JSONObject();
		try {
			userJson.put(LOGIN, mEmail);

			userJson.put(PASSWORD, mEncryptedPassword);


			JSONObject devServer = encodeServerData(EServerType.DEV);
			if(devServer != null)userJson.put(EServerType.DEV.toString(), devServer);

			JSONObject testServer = encodeServerData(EServerType.TEST);
			if(testServer != null)userJson.put(EServerType.TEST.toString(), testServer);

			JSONObject prodServer = encodeServerData(EServerType.PROD);
			if(prodServer != null)userJson.put(EServerType.PROD.toString(), prodServer);

		} catch (JSONException e) {
			e.printStackTrace();
		}

		return userJson;

	}


	private JSONObject encodeServerData(EServerType server){
		JSONObject serverJson = null;

		try {
			EUserRole role = mRoles.get(server);

			if (role != null) {
				serverJson = new JSONObject();

				// 1 - add the role of the user on this server
				serverJson.put(ROLES, mRoles.get(server).getValue());

				// 3 -  add the sites available on this server
				JSONArray sitesJson = new JSONArray();
				List<Site> sites = mSites.get(server);
				if (sites != null) {
					for (Site site : mSites.get(server)) {
						sitesJson.put(site.encode());
					}
					serverJson.put(SITES, sitesJson);
				}
			}


		} catch (JSONException e) {
			e.printStackTrace();
		}

		return serverJson;
	}

	/**
	 * Returns the {@link User} corresponding to this {@link JSONObject}.
	 * @param json the {@link JSONObject} to decode.
	 * @return the decoded {@link User}.
	 */
	public static User decode(String jsonValue){

		JSONObject json;
		User user = null;

		String login;
		String password;

		try {
			json = new JSONObject(jsonValue);

			login = json.getString(LOGIN);
			password = json.getString(PASSWORD);

			user = new User(login, password);

			decodeServer(EServerType.DEV, json, user);
			decodeServer(EServerType.TEST, json, user);
			decodeServer(EServerType.PROD, json, user);

		} catch (JSONException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}

		return user;
	}

	private static void decodeServer(EServerType server, JSONObject userJson, User user){
		try {
			if (!userJson.isNull(server.toString())) {
				JSONObject serverJson = userJson.getJSONObject(server.toString());

				if (serverJson != null) {

					// 1 - Retrieve the role on this server
					EUserRole roleOnServer = EUserRole.getFromValue(serverJson.getInt(ROLES));
					user.mRoles.put(server, roleOnServer);

					// 2 - Retrieve the sites available on this server
					JSONArray sitesOnServer = serverJson.getJSONArray(SITES);				
					if (sitesOnServer != null) {
						List<Site> sites = new ArrayList<Site>();
						for (int i = 0; i < sitesOnServer.length(); i++) {
							Site s = Site.decode(sitesOnServer.getJSONObject(i));
							if(s != null) sites.add(s);
						}
						user.mSites.put(server, sites);
					}
				}
			}



		} catch (JSONException e) {
			e.printStackTrace();
		}
	}
