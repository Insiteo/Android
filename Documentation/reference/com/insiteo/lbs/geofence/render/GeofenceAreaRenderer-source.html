package com.insiteo.lbs.geofence.render;

import java.util.HashMap;
import java.util.concurrent.Semaphore;

import android.content.res.Resources;
import android.graphics.Canvas;
import android.graphics.Point;

import com.insiteo.lbs.map.entities.Map;
import com.insiteo.lbs.map.render.ERenderMode;
import com.insiteo.lbs.map.render.IRTO;
import com.insiteo.lbs.map.render.IRTOListener;
import com.insiteo.lbs.map.render.IRenderer;
import com.insiteo.lbs.map.render.Touch;
import com.threed.jpct.World;

/**
 * {@link IRenderer} that is used to display {@link GfxGeofenceArea}.
 * @author Insiteo
 *
 */
public class GeofenceAreaRenderer implements IRenderer {

	private static final String TAG = GeofenceAreaRenderer.class.getSimpleName();

	private final static int RENDERING_PRIORITY = 10;

	private int mPriority;
	private Map mCurrentMap;
	private Resources mResources;	
	private HashMap<String, IRTO> mRTOS;
	private Semaphore mLock = new Semaphore(1);

	public GeofenceAreaRenderer(Resources aResources) {
		mResources = aResources;
		mPriority = RENDERING_PRIORITY;

		mRTOS = new HashMap<String, IRTO>();
	}

	@Override
	public Class<? extends IRTO> getRTOClass() {
		return GfxGeofenceArea.class;
	}

	@Override
	public void setPriority(int aPriority) {
		mPriority = aPriority;
	}

	@Override
	public int getPriority() {
		return mPriority;
	}

	@Override
	public void setCurrentMap(Map aMap) {
		mCurrentMap = aMap;
	}

	@Override
	public void setDisplayOrientation(int aDisplayOrientation) {
	}

	@Override
	public void clear() {
		mRTOS.clear();
	}

	@Override
	public void clearZone(int aZoneId) {
	}


	//******************************************************************************************************************
	//  IRTO
	// *****************************************************************************************************************


	@Override
	public boolean addRTO(IRTO aObject) {
		try {
			mLock.acquire();
			aObject.setResources(mResources);
			String guid = ((GfxGeofenceArea) aObject).getGuid();		
			mRTOS.put(guid, aObject);	
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}

		return true;
	}

	@Override
	public boolean addRTOInZone(int aZoneId, IRTO aObject) {
		return false;
	}

	/**
	 * Remove an {@link IRTO} based on his guid.
	 * @param guid of the {@link IRTO} to remove.
	 */
	public void removeRTO(String guid) {
		try {
			mLock.acquire();
			mRTOS.remove(guid);
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}
	}

	@Override
	public void removeRTO(IRTO aObject) {
		try {
			mLock.acquire();
			String guid = ((GfxGeofenceArea) aObject).getGuid();
			mRTOS.remove(guid);
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}
	}

	@Override
	public void removeRTOFromZone(int aZoneId, IRTO aObject) {
	}

	//******************************************************************************************************************
	//  RENDERING
	// *****************************************************************************************************************

	private ERenderMode mRenderMode;
	private boolean mIsDisplayed;

	@Override
	public void setDisplayEnabled(boolean aOn) {
		mIsDisplayed = aOn;
	}

	@Override
	public boolean isDisplayEnabled() {
		return mIsDisplayed;
	}

	@Override
	public void setRenderMode(ERenderMode renderingMode) {
		mRenderMode = renderingMode;
	}

	@Override
	public ERenderMode getRenderMode() {
		return mRenderMode;
	}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		try {
			mLock.acquire();
			for (String key : mRTOS.keySet()) {
				IRTO rto = mRTOS.get(key);

				if(rto.getMapID() == mCurrentMap.getId() || rto.getMapID() == IRTO.ALL_MAP_ID) {

					switch (mRenderMode) {
					case RENDER_MODE_2D:
						rto.draw(aCanvas, aRatio, aOffset, aAngle);
						break;

					case RENDER_MODE_3D:

						break;
					}

				}
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}
	}

	@Override
	public void draw3D(World world, double ratio, float angle) {
	}

	//******************************************************************************************************************
	//  Touch handling - NOTHING IMPLEMENTED FOR THIS IRENDERER
	// *****************************************************************************************************************

	@Override
	public void setRTOListener(IRTOListener listener) {
	}

	@Override
	public void setTouchEnabled(boolean aOn) {
	}

	@Override
	public boolean isTouchEnabled() {
		return false;
	}

	@Override
	public boolean onTouchDown(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchPointerDown(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchMove(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchPointerUp(Touch aTouch) {
		return false;
	}

	@Override
	public boolean onTouchUp(Touch aTouch) {
		return false;
	}

}
