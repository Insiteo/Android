/**

public class GfxGeofenceArea implements IRTO {
	
	private static int NEXT_ID = 1;

	private Paint mPaint = new Paint();
	
	private WeakReference<GeofenceArea> mGeoArea;
	private int mId;
	
	private List<PointF> mPolygon;
	
	private int mGeofenceAreaColor;
	
	private int mMapId = NO_MAP_ID;
	
	private ERenderMode mRenderMode;

	//******************************************************************************************************************
	//  CONSTRUCTORS
	// *****************************************************************************************************************
	
	public GfxGeofenceArea(GeofenceArea geoArea){
		mId = NEXT_ID;
		NEXT_ID ++;
		mMapId = geoArea.getCenter().getMapID();
		mPolygon = geoArea.getPolygon();
		
		mGeoArea = new WeakReference<GeofenceArea>(geoArea);
	}

	public String getGuid() {
		String guid = null;
		if(mGeoArea != null) {
			guid = mGeoArea.get().getGUID();
		}
		return guid;
	}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		mPaint.setColor(mGeofenceAreaColor);
		drawPolygon(aCanvas, mPolygon, aOffset, aRatio);
	}
	
	private void drawPolygon(Canvas canvas, List<PointF> polygon, Point aOffset, double ratio) {
		if(polygon != null && !polygon.isEmpty()) {
			Path wallpath = new Path();
			
			wallpath.reset(); // only needed when reusing this path for a new build
			
			Point point = new Point((int)(polygon.get(0).x * ratio) + aOffset.x, (int)(polygon.get(0).y * ratio) + aOffset.y);
			wallpath.moveTo(point.x, point.y); // used for first point
			
			for (int i = 1; i < polygon.size(); i++) {
				point = new Point((int)(polygon.get(i).x * ratio) + aOffset.x, (int)(polygon.get(i).y * ratio) + aOffset.y);

				wallpath.lineTo(point.x, point.y);
			}

			canvas.drawPath(wallpath, mPaint);
		}	
	}
