/**
 * This class handles the geofencing process. You can retrieve an instance of it by calling <code>(GeofenceProvider)LocationProvider.getInstance().getModule(ELocationModule.GEOFENCING)</code>. 
 * It provides a way to be notified when location is detected in a certain area (see GeofenceZone), defined on Insiteo servers.
 * To use this class, you just have to call {@link #setListener(IGeofenceListener)}, and start the location. The listener will then receive notification each time
 * location enter/stay/leave an area.  
 */
public class GeofenceProvider implements ILbsModule {

	private static int GEOFENCE_DEFAULT_WIDTH = 2; // in meters

	private boolean init() {
		String dataDir = InitProvider.getInstance().getDataDirPath() + "/" + EPackageType.GEOFENCING.getDirName();

		GeofenceInitParser initParser = new GeofenceInitParser();
		boolean configLoaded = initParser.parseConfig(dataDir);
		boolean mapsLoaded = initParser.parseMaps(dataDir);
		boolean protoLoaded = initParser.parseProtoFile(dataDir);
		boolean geoPushLoaded = initParser.parseGeoPush(dataDir);


		if (configLoaded && mapsLoaded && protoLoaded /*&& geoPushLoaded*/) {
			mCoordSystem = initParser.getCoordSystem();
			mTimeToReset = initParser.getTimeToReset();
			mMapsSettings = initParser.getMapsSettings();

			mGeoPushMapSettings = initParser.getGeoPushMapSettings();

			// Add the protobug geofence zone to the list of the zone handled by the provider
			ArrayList<GeofenceArea> areas = initParser.getAreas();
			for (int i = 0; i < areas.size(); i++) {
				mGeofenceAreas.put(areas.get(i).getGUID(), areas.get(i));
			}

			if(mRenderer != null) {
				for (String key : mGeofenceAreas.keySet()) {				
					GfxGeofenceArea gfxGeofenceArea = new GfxGeofenceArea(mGeofenceAreas.get(key));
					mRenderer.addRTO(gfxGeofenceArea);	
				}
			}


			return true;
		}

		return false;
	}

	private void reset() {
		Log.d(TAG, "Reseting active zones");
		mActiveZones = new ArrayList<GeofenceZone>();
		if (mListener != null) {
			new Thread(new Runnable() {
				@Override public void run() {
					if (mListener != null) {
						mListener.onGeofenceDataCleared();
					}
				}
			}).start();
		}
	}

	private void runGeofencing(InsLocation aLocation) {
		int mapID = aLocation.getMapID();
		MapSettings mapSettings = mMapsSettings.get(mapID);
		if (mapSettings != null) {
			//convert loc in internal coordinates (meters)
			PointD coords = aLocation.getCoord();
			PointD internCoords = mapSettings.toInternalCoordinates(mCoordSystem, coords.x, coords.y);
			Position internPos = new Position(mapID, (float)internCoords.x, (float)internCoords.y);

			//get areas that contains this pos
			Stack<GeofenceArea> detectedAreas = new Stack<GeofenceArea>();
			for (String key : mGeofenceAreas.keySet()) {
				GeofenceArea area = mGeofenceAreas.get(key);
				if (area.containsPosition(internPos)) {
					detectedAreas.push(area);
				}
			}

			//check for each active zone if it has a status update
			for (GeofenceZone zone : mActiveZones) {
				String guid = zone.getGUID();

				//search the given zone in the detected areas stack
				boolean found = false;
				GeofenceArea area = null;
				Iterator<GeofenceArea> it = detectedAreas.iterator();
				while (it.hasNext() && !found) {
					GeofenceArea ga = it.next();
					if (guid.equals(ga.getGUID())) {
						//this area correspond to an active zone
						found = true;
						area = ga;
						//this areas is done : remove it from the stack
						it.remove();
					}
				}

				//update the zone, depending its detection
				updateActiveZone(zone, area, found);
			}

			//check remaining areas in the stack (areas not done yet)
			for (GeofenceArea area : detectedAreas) {
				Log.d(TAG, "ADDING new AREA");
				//this area correspond to no active zone
				//convert area center in external coordinates (meters)
				PointD extCoords = mapSettings.toExternalCoordinates(mCoordSystem, area.getCenter().getCoord().x, area.getCenter().getCoord().y);
				Position extPos = new Position(mapID, (float)extCoords.x, (float)extCoords.y);
				//create a new active zone and add it to the list
				GeofenceZone zone = new GeofenceZone(area.getGUID(), extPos, area.getTimeToLeave(), area.isLeaveEnabled(), area.getExtra1(), area.getExtra2(), area.getExtra3());
				updateActiveZone(zone, area, true);
				mActiveZones.add(zone);
			}

			//create lists for notification and remove active zones, if needed
			final List<GeofenceZone> enteredZones = new ArrayList<GeofenceZone>(3);
			final List<GeofenceZone> stayedZones = new ArrayList<GeofenceZone>(3);
			final List<GeofenceZone> leftZones = new ArrayList<GeofenceZone>(3);
			boolean notify = prepareNotification(enteredZones, stayedZones, leftZones);



			// Analytics
			for (int i = 0; i < enteredZones.size(); i++) { //2
				AnalyticsGenericEvent zoneEvent = new AnalyticsGenericEvent(AnalyticsConstants.IS_GEO_FENCING_LABEL);
				zoneEvent.setString1(enteredZones.get(i).getGUID());
				zoneEvent.setInt1(2);
				AnalyticsManager.getInstance().addGenericEvent(zoneEvent);
			}

			for (int i = 0; i < stayedZones.size(); i++) {//3
				AnalyticsGenericEvent zoneEvent = new AnalyticsGenericEvent(AnalyticsConstants.IS_GEO_FENCING_LABEL);
				zoneEvent.setString1(stayedZones.get(i).getGUID());
				zoneEvent.setInt1(3);
				AnalyticsManager.getInstance().addGenericEvent(zoneEvent);
			}

			for (int i = 0; i < leftZones.size(); i++) {//5
				AnalyticsGenericEvent zoneEvent = new AnalyticsGenericEvent(AnalyticsConstants.IS_GEO_FENCING_LABEL);
				zoneEvent.setString1(leftZones.get(i).getGUID());
				zoneEvent.setInt1(5);
				AnalyticsManager.getInstance().addGenericEvent(zoneEvent);
			}

			//notify listener if any
			if (mListener != null && notify) {
				new Thread(new Runnable() {
					@Override public void run() {
						if (mListener != null) {
							mListener.onGeofenceUpdate(enteredZones, stayedZones, leftZones);
						}
					}
				}).start();
			}
		}
	}	

	private void updateActiveZone(GeofenceZone aZone, GeofenceArea aArea, boolean aIsDetected) {
		long currentTime = SystemClock.elapsedRealtime();
		if (aIsDetected) {
			switch (aZone.getEventType()) {
			case UNDEFINED:
				aZone.setEventType(EventType.ENTERING);
				aZone.setEventTime(currentTime);
				break;
			case ENTERING:
				if (currentTime - aZone.getEventTime() > aArea.getTimeToEnter()) {
					//time inside this zone is long enough => "enter" it
					aZone.setEventType(EventType.ENTERED);
					aZone.setEventTime(currentTime);
					aZone.setNotify(aArea.isEnterEnabled());
				}
				break;
			case ENTERED:
				aZone.setEventType(EventType.STAY);
			case STAY:
				if (aArea.isStayEnabled()) {
					if (currentTime - aZone.getEventTime() > aArea.getTimeToStay()) {
						//time inside this zone is long enough => "stay" in it
						aZone.setEventTime(currentTime);
						aZone.setNotify(aArea.isStayEnabled());
					}
				}
				break;
			case LEAVING:
				//position re-entered the zone
				aZone.setEventType(EventType.STAY);
				aZone.setEventTime(currentTime);
				break;
			case LEFT:
				aZone.setEventType(EventType.ENTERING);
				aZone.setEventTime(currentTime);
				break;
			}
		}
		else {
			switch (aZone.getEventType()) {
			case UNDEFINED:
				break;
			case ENTERING:
				aZone.setEventType(EventType.UNDEFINED);
				break;
			case ENTERED:
			case STAY:
				aZone.setEventType(EventType.LEAVING);
				aZone.setEventTime(currentTime);
				break;
			case LEAVING:
				if (currentTime - aZone.getEventTime() > aZone.getTimeToLeave()) {
					//time outside this zone is long enough => "leave" it
					aZone.setEventType(EventType.LEFT);
					aZone.setEventTime(currentTime);
					aZone.setNotify(aZone.isLeaveEnabled());
				}
				break;
			case LEFT:
				aZone.setEventType(EventType.UNDEFINED);
				break;
			}
		}		
	}

	//returns true if a notification is needed (ie : lists are not empty), false otherwise
	private boolean prepareNotification(List<GeofenceZone> enteredZones, List<GeofenceZone> stayedZones, List<GeofenceZone> leftZones) {
		boolean notify = false;
		Iterator<GeofenceZone> it = mActiveZones.iterator();
		while(it.hasNext()) {
			GeofenceZone zone = it.next();

			switch (zone.getEventType()) {
			case UNDEFINED:
				it.remove();
				break;
			case ENTERING:
				break;
			case ENTERED:
				if (zone.getNotify()) {
					enteredZones.add(zone);
					zone.setNotify(false);
					notify = true;
				}
				break;
			case STAY:
				if (zone.getNotify()) {
					stayedZones.add(zone);
					zone.setNotify(false);
					notify = true;
				}
				break;
			case LEAVING:
				break;
			case LEFT:
				if (zone.getNotify()) {
					leftZones.add(zone);
					zone.setNotify(false);
					notify = true;
				}
				it.remove();						
				break;
			}	
		}
		return notify;
	}


	private GeoPushMapSettings getGeoPushMapSettingsWithZoneID(int zoneId) {
		//		ISgeopushcustomgeowidth;
		Zone zone = InitProvider.getInstance().getZone(zoneId);

		if(zone != null && mGeoPushMapSettings != null) {
			int mapId = zone.getCenter().getMapID();

			GeoPushMapSettings settings = mGeoPushMapSettings.get(mapId);
			if(settings != null) {
				return settings;
			} else {
				return new GeoPushMapSettings(true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, GEOFENCE_DEFAULT_WIDTH);
			}
		} else {
			return new GeoPushMapSettings(true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, GEOFENCE_DEFAULT_WIDTH);
		}
	}


	private GeoPushMapSettings getGeoPushMapSettingsWithMapID(int mapId) {

		if(mGeoPushMapSettings != null) {
			GeoPushMapSettings settings = mGeoPushMapSettings.get(mapId);

				return settings;
			} else {
				return new GeoPushMapSettings(true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, GEOFENCE_DEFAULT_WIDTH);
			}
		} else {
			return new GeoPushMapSettings(true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, true, GEOFENCE_DEFAULT_TIME, GEOFENCE_DEFAULT_WIDTH);
		}
	}

	@Override
	public void setLocation(InsLocation aLocation) {
		//do genfencing only if there is a listener
		if (mListener != null) {

			if (aLocation != null) {
				if (!mIsInitialized) {
//					mIsInitializing = true;
					new Thread(new Runnable() {
						@Override public void run() {
							if (init()) {
								//init is successful
								mIsInitialized = true;
							}
//							mIsInitializing = false;
						}
					}).start();
				} else {
					//module is initialized
					if (mResetTask != null) {
						//cancel reset task if any
						mResetTask.cancel();
						mResetTask = null;
					}
					runGeofencing(aLocation);
				}
			}

			//schedule a reset if no one is already schedule
			if (mIsInitialized && mResetTask == null) {
				mResetTask = new ResetTask();
				mResetTimer.schedule(mResetTask, mTimeToReset);
			}
		}
	}

	@Override
	public void clear() {

		mListener = null;

		mIsInitialized = false;
		mActiveZones = new ArrayList<GeofenceZone>();
		mGeofenceAreas = new HashMap<String, GeofenceArea>();
//		mIsInitializing = false;
		if (mResetTask != null) {
			//cancel reset task if any
			mResetTask.cancel();
			mResetTask = null;
		}

		mRenderer = null;

		init();
	}

			boolean enteredEnabled, long enteredTime,

			float width) {

		GeofenceArea geoArea = new GeofenceArea(zoneId, null, content, enteredEnabled, enteredTime, stayedEnabled, stayedTime, leaveEnabled, leaveTime, width);

		if(geoArea.getPolygon() == null) {

		// Add the created GeofenceArea to the renderer
		GfxGeofenceArea gfxGeofenceArea = new GfxGeofenceArea(geoArea);
		mRenderer.addRTO(gfxGeofenceArea);	

		return geoArea;
	}

	/**
	 * Creates a new {@link GeofenceArea} around the given {@link Position}. The area will be a square constructed with the given size.
	 * @param guid The guid that should be used for the new {@link GeofenceArea}. Guid can not be null or empty!
	 * @param center The center of the {@link GeofenceArea}.
	 * @param size The size for the square. 
	 * @param content The content for the created {@link GeofenceArea}.
	 * @param enteredEnabled True if entered event should be enabled.
	 * @param enteredTime The threshold time to trigger the entered event (in ms).
	 * @param stayedEnabled True if stayed event should be enabled.
	 * @param stayedTime The threshold time to trigger the stayed event (in ms).
	 * @param leaveEnabled True if leave event should be enabled.
	 * @param leaveTime The threshold time to trigger the leave event (in ms).
	 * @return The created {@link GeofenceArea} or null if it was not properly created.
	 */
	public GeofenceArea addGeofenceArea(String guid, Position center, float size, String content,
			boolean enteredEnabled, long enteredTime, 
			boolean stayedEnabled, long stayedTime,
			boolean leaveEnabled, long leaveTime) {

		if(guid == null || guid.isEmpty()) {
			Log.d(TAG, "GUID must be set explicitly");
			return null;
		}

		GeofenceArea geoArea = new GeofenceArea(guid, center, size, content, enteredEnabled, enteredTime, stayedEnabled, stayedTime, leaveEnabled, leaveTime);

		if(geoArea.getPolygon() == null) {
			geoArea = null;
			return null;
		}

		mGeofenceAreas.put(geoArea.getGUID(), geoArea);

		// Add the created GeofenceArea to the renderer
		GfxGeofenceArea gfxGeofenceArea = new GfxGeofenceArea(geoArea);
		mRenderer.addRTO(gfxGeofenceArea);	

		return geoArea;
	}
