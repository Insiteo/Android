package com.insiteo.lbs.geofence;

import com.insiteo.lbs.common.utils.geometry.Position;

/**
 * Class holding data for a geofencing zone. It contains the GUID of the zone, and it's center. It also contains three extra Strings. 
 */
public class GeofenceZone {

	private String mGUID;
	private Position mCenter;	//point at the center of the zone
	private String mExtra1, mExtra2, mExtra3;
	private long mTimeToLeave; //ms : used internally to know when this zone will be left, without needing corresponding area
	private boolean mIsLeaveEnabled; //used internally to know if this zone will notify when leaving, without needing corresponding area
	
	private EventType mEventType;
	private long mEventTime;
	private boolean mNotify;
	
	/**
	 * Constructor. 
	 */
	/*package*/ GeofenceZone(String GUID, Position zoneCenter, long timeToLeave, boolean notifyLeaving, String extra1, String extra2, String extra3) {
		mGUID = GUID;
		mCenter = new Position(zoneCenter);
		mTimeToLeave = timeToLeave;
		mIsLeaveEnabled = notifyLeaving;
		mEventType = EventType.UNDEFINED;
		mEventTime = 0;
		mNotify = false;
		mExtra1 = extra1;
		mExtra2 = extra2;
		mExtra3 = extra3;
	}
	
	/**
	 * Get the unique identifier of this zone
	 * @return the GUID of the zone.
	 */
	public String getGUID() {
		return mGUID;
	}
	
	/**
	 * Get the position representing the center of this zone
	 * @return the center of the zone, in application's coordinate system
	 */
	public Position getCenter() {
		return mCenter;
	}
	
	/**
	 * Get the first extra field of this zone
	 * @return the first extra string
	 */
	public String getExtra1() {
		return mExtra1;
	}
	
	/**
	 * Get the second extra field of this zone
	 * @return the second extra string
	 */
	public String getExtra2() {
		return mExtra2;
	}
	
	/**
	 * Get the third extra field of this zone
	 * @return the third extra string
	 */
	public String getExtra3() {
		return mExtra3;
	}
	
	
	/*package*/ EventType getEventType() {
		return mEventType;
	}
	/*package*/ void setEventType(EventType eventType) {
		mEventType = eventType;
	}
	
	/*package*/ long getEventTime() {
		return mEventTime;
	}
	/*package*/ void setEventTime(long eventTime) {
		mEventTime = eventTime;
	}
	
	/*package*/ boolean getNotify() {
		return mNotify;
	}
	/*package*/ void setNotify(boolean notify) {
		mNotify = notify;
	}
	
	/*package*/ long getTimeToLeave() {
		return mTimeToLeave;
	}
	
	/*package*/ boolean isLeaveEnabled() {
		return mIsLeaveEnabled;
	}
}
