package com.insiteo.lbs.catalog.entities;



public class CatalogEntry implements Comparable<CatalogEntry> {

	protected int mId;
	protected CatalogEntryType mType;
	protected String mName;
	protected String mSubtitle;	//short description
	protected String mImageUrl;
	protected int mSortIndex;
    
	private boolean mIsHrr;
	
	private String mExtra1;
	private String mExtra2;
	private String mExtra3;
	private Float mFExtra1;
	private String mDispExtra1;
	private String mDispExtra2;
	private String mDispExtra3;
	private Float mDispFExtra1;
	
	private String mExtId;
	
    
    public CatalogEntry(int id, String extId, CatalogEntryType type, String name, String subtitle, String imageUrl, int sortIndex) {
    	mId = id;
    	mExtId = extId;
    	mType = type;
    	mName = name;
    	mSubtitle = subtitle;
        mImageUrl = imageUrl;
        mSortIndex = sortIndex;
		
		mIsHrr = false;
        
        mExtra1 = null;
		mExtra2 = null;
		mExtra3 = null;
		mFExtra1 = null;
		mDispExtra1 = null;
		mDispExtra2 = null;
		mDispExtra3 = null;
		mDispFExtra1 = null;
    }

    public int getId() {
        return mId;
    }
    
    public CatalogEntryType getType() {
        return mType;
    }
    
    public String getName() {
    	if (mName == null) {
    		return "";
    	}
        return mName;
    }
    
    public String getSubtitle() {
    	if (mSubtitle == null) {
    		return "";
    	}
        return mSubtitle;
    }

    public boolean hasImageUrl() {
    	return (mImageUrl != null && mImageUrl != "");
    }
    
    public String getImageUrl() {
        return mImageUrl;
    }
    
    public void setImageUrl(String imageUrl) {
    	mImageUrl = imageUrl;
    }
    
    public int getSortIndex() {
    	return mSortIndex;
    }
    
    public void setSortIndex(int sortIndex) {
    	mSortIndex = sortIndex;
    }
    
    public void setHRR(boolean isHRR) {
    	mIsHrr = isHRR;
    }
    
    public boolean isHRR() {
    	return mIsHrr;
    }

    public String getExtId(){
    	return mExtId;
    }
    
    public void setExtId(String extId){
    	mExtId = extId;
    }

		return mExtra1;

		mDispExtra1 = dispExtra1;
		mDispExtra2 = dispExtra2;
		mDispExtra3 = dispExtra3;
		mDispFExtra1 = dispFExtra1;
	}

	@Override
	public int compareTo(CatalogEntry another) {
		if (mId == another.mId) {
			return 0;
		}
		else {
			if (mType == another.mType) {
				if (mSortIndex == another.mSortIndex) {
					return (mName.compareToIgnoreCase(another.mName)); 
				}
				else {
					return (mSortIndex > another.mSortIndex ? 1 : -1);
				}
			}
			else {
				if (mType == CatalogEntryType.CATEGORY) {
					return -1;
				}
				else if (another.mType == CatalogEntryType.CATEGORY) {
					return 1;
				}
				else if (mType == CatalogEntryType.SPONSOR) {
					return -1;
				}
				else {
					return 1;
				}
			}
		}
	}
	
	public String toString(){
		return getName();
	}
}
