package com.insiteo.lbs.catalog;

import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;

import android.content.Context;
import android.media.MediaPlayer;
import android.os.Handler;
import android.os.PowerManager;
import android.view.View;
import android.view.View.OnClickListener;
import android.widget.ImageButton;
import android.widget.SeekBar;
import android.widget.SeekBar.OnSeekBarChangeListener;
import android.widget.Toast;

import com.insiteo.lbs.R;
import com.insiteo.lbs.common.utils.Log;

public class AudioPlayer implements MediaPlayer.OnCompletionListener, OnClickListener, OnSeekBarChangeListener {

	private static final int SEEKBAR_REFRESH_DELAY = 500;	//milliseconds
	
	private MediaPlayer mp;
	private final Handler mSeekbarUpdateHandler = new Handler();
	
	private View mPlayerView;
	private ImageButton mPlayBtn;
	private ImageButton mStopButton;
	private ImageButton mExitButton;
	private SeekBar mSeekbar;
	private boolean mAllowPlay = true;
	private String mAllowReason;
	
	private PowerManager.WakeLock mWakeLock = null; // OHT 2011-09-20 do not want phone to sleep while audio is playing

	public AudioPlayer(View aPlayerView) {
		mPlayerView = aPlayerView;
		mPlayBtn = (ImageButton)aPlayerView.findViewById(R.id.insiteo_catal_btn_player_play);
		mStopButton = (ImageButton)aPlayerView.findViewById(R.id.insiteo_catal_btn_player_stop);
		mExitButton = (ImageButton)aPlayerView.findViewById(R.id.insiteo_catal_btn_player_exit);
		mSeekbar = (SeekBar)aPlayerView.findViewById(R.id.insiteo_catal_seekbar);
		mPlayBtn.setOnClickListener(this);
		mStopButton.setOnClickListener(this);
		mExitButton.setOnClickListener(this);
		mSeekbar.setOnSeekBarChangeListener(this);
		
		// do not want phone to sleep while audio is playing
		PowerManager pm = (PowerManager) aPlayerView.getContext().getSystemService(Context.POWER_SERVICE);
		mWakeLock = pm.newWakeLock(PowerManager.SCREEN_DIM_WAKE_LOCK, "PM_AUDIO_PLAYER");
	}

	@Override
	public void onClick(View v) {
		if (v == mPlayBtn) {
			if (mp.isPlaying()) {
				pause();
			}
			else {
				resume();
			}
		}
		else if (v == mStopButton) {
			stop();
		}
		else if (v == mExitButton) {
			exit();
		}
	}
	
	@Override
	public void onCompletion(MediaPlayer mp) {
		stop();
	}

	public void start(File audioFile, boolean aReadNow) {
		
		if(mAllowPlay == false && mAllowReason != null) {
			Toast t = Toast.makeText(mPlayerView.getContext(), mAllowReason, Toast.LENGTH_LONG);
			t.show();
			return;
		}
		
		try {
			exit();
			mPlayerView.setVisibility(View.VISIBLE);
			mp = new MediaPlayer();
			mp.setOnCompletionListener(this);
			mp.reset();
			FileInputStream fis = new FileInputStream(audioFile);
			mp.setDataSource(fis.getFD());
			mp.prepare();
			mSeekbar.setMax(mp.getDuration());
			if(aReadNow) {
				resume();
			}
		} 
		catch (IOException e) {
			displayError(e);
		}
	}	
	
	public void stop() {
		pause();
		mp.stop();
		
		try {
			mp.prepare();
			mp.seekTo(0);			
		} 
		catch (Throwable t) {
			displayError(t);
		}
	}
	
	public void exit() {
		mPlayerView.setVisibility(View.GONE);
		if (mp != null) {
			mp.release();
			mp = null;
		}
	}

	public void pause() {
		//do not want phone to sleep while audio is playing
		if(mWakeLock != null && mWakeLock.isHeld()) {
			mWakeLock.release();
		}
		if (mp.isPlaying()) {
			mp.pause();
		}
		mPlayBtn.setImageResource(R.drawable.insiteo_catal_play);
	}
	
	public void resume() {
		if(mAllowPlay == false && mAllowReason != null) {
			Toast t = Toast.makeText(mPlayerView.getContext(), mAllowReason, Toast.LENGTH_LONG);
			t.show();
			return;
		}
		
		//do not want phone to sleep while audio is playing
		mWakeLock.acquire();
		
		mp.start();
		mPlayBtn.setImageResource(R.drawable.insiteo_catal_pause);
		seekbarProgressUpdate();
	}


	private void displayError(Throwable t) {
		//TODO
		Log.d("audio", "Exception in audio player", t);
	}

	//this will loop until player is stopped
	public void seekbarProgressUpdate() {
		if (mSeekbar != null && mp != null) {
	    	mSeekbar.setProgress(mp.getCurrentPosition());
			if (mp.isPlaying()) {
				Runnable notification = new Runnable() {
			        public void run() {
			        	seekbarProgressUpdate();
					}
			    };
			    mSeekbarUpdateHandler.postDelayed(notification, SEEKBAR_REFRESH_DELAY);
	    	}
		}
    } 
	
	@Override
	public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
		if (fromUser) {
			mp.seekTo(progress);
		}
	}

	@Override
	public void onStartTrackingTouch(SeekBar seekBar) {
		if (mp.isPlaying()) {
			mp.pause();
		}
	}

	@Override
	public void onStopTrackingTouch(SeekBar seekBar) {
		resume();
	}

	
	public void allowPlay(boolean aAllow, String aReason) {
		mAllowPlay = aAllow;
		mAllowReason = aReason;
	}


}
