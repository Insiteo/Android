CatalogWebView

extends WebView
package com.insiteo.lbs.catalog;

import android.content.Context;
import android.graphics.Canvas;
import android.util.AttributeSet;
import android.webkit.WebView;

/**
 * Class extending a webview, to allow vertical scrolling when pressing back key in catalog
 * This is a dirty bypass because of a bug in Android :
 * scrollTo() on a webview must be called after the page is displayed, and onPageFinished() is sometimes called before the page is really displayed,
 * thus you can't be sure the call to scrollTo() will be taken in account.  
 */
public class CatalogWebView extends WebView {

	private int mVScroll = 0;	//value for scroll
	private int mVScrollBack = 0;	//backup of scroll value 
	private volatile boolean mManualVScrollDone = false;
	
	public CatalogWebView(Context context) {
		super(context);
	}

	public CatalogWebView(Context context, AttributeSet attrs) {
		super(context, attrs);
	}

	public CatalogWebView(Context context, AttributeSet attrs, int defStyle) {
		super(context, attrs, defStyle);
	}
	
	public void setVScroll(int aVScroll) {
		mVScroll = aVScroll;
		mVScrollBack = 0;
		mManualVScrollDone = false;
	}
	
	@Override
	protected void onDraw(Canvas canvas) {
		super.onDraw(canvas);
		
		//scroll if the offset is set
		if (mVScroll != 0) {
			int tmp = mVScroll;
			mVScroll = 0;
			//backup the scroll value, because scroll offset of the webview will be 
			//reset after calling scrollTo (bug : onscrollchanged will be called 2 times when calling scrollTo !)
			mVScrollBack = tmp;	 
			scrollTo(0, tmp);
			//set this boolean so that next call to onScrollChanged will know it has to rescroll
			mManualVScrollDone = true;
		}
	}
	
	@Override
	protected void onScrollChanged(int l, int t, int oldl, int oldt) {
		if (mManualVScrollDone) {
			//rescroll if needed
			mManualVScrollDone = false;
			int tmp = mVScrollBack;
			mVScrollBack = 0;
			scrollTo(0, tmp);
		}
		else {
			//behave normally
			super.onScrollChanged(l, t, oldl, oldt);
		}
	}
	

}
