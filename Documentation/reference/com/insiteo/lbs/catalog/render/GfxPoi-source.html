	private void set2DResources(Resources res){
		//init poi icon
		mPoiPinIcon = BitmapFactory.decodeResource(res, R.drawable.insiteo_catal_poi_pin);
		mPoiPinIconWidth = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_pin_width);
		mPoiPinIconHeight = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_pin_height);
		mPoiPinIconHOffset = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_pin_hoffset);
		mPoiPinIconVOffset = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_pin_voffset);
		mPoiPinRect = new Rect();
		mPoiPinIconPaint = new Paint();

		//get all sizes from resources
		mNameTextMaxLength = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_name_text_length);
		mNameTextSize = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_name_text);
		mTextSpace = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_name_text_space);
		mTextSpeed = res.getDimensionPixelSize(R.dimen.insiteo_catal_poi_name_text_speed);
		mTextRepeatSpace = 2*mNameTextSize;

		//create text paint for name
		float shadowRadius = res.getDimension(R.dimen.insiteo_catal_poi_name_text_shadow_radius);
		mNamePaint = new TextPaint();
		mNamePaint.setColor(Color.WHITE);
		mNamePaint.setAntiAlias(true);
		mNamePaint.setAlpha(255);
		mNamePaint.setTextSize(mNameTextSize);
		mNamePaint.setShadowLayer(shadowRadius, 2f, 2f, Color.BLACK);

		//calculate name width, and make it scroll if it is too long
		String name = mPoi.getName();
		mNameTextLength = (int)DynamicLayout.getDesiredWidth(name,  mNamePaint);
		int nameWidth = mNameTextLength;
		if (nameWidth > mNameTextMaxLength) {
			nameWidth = mNameTextMaxLength;
			mAnimateText = true;
		}
		else {
			mAnimateText = false;
		}

		//load background 9-patch
		Drawable ds = res.getDrawable(R.drawable.insiteo_catal_poi_info_selected);
		mNameBackgSelected = (NinePatchDrawable)ds;
		Drawable dn = res.getDrawable(R.drawable.insiteo_catal_poi_info);
		mNameBackgNormal = (NinePatchDrawable)dn;
		mNameBackg = mNameBackgNormal;
		Rect rPadding = new Rect();
		mNameBackg.getPadding(rPadding);

		//compute content height
		int contentHeight = mNameTextSize;
		int imgBackgHeight = rPadding.top + contentHeight + rPadding.bottom;
		int imgMinHeight = mNameBackg.getMinimumHeight();
		if (imgBackgHeight < imgMinHeight) {
			contentHeight += imgMinHeight - imgBackgHeight; 
		}

		//init itinerary icon
		mItiIcon = BitmapFactory.decodeResource(res, R.drawable.insiteo_catal_btn_iti);
		mItiIconBackgNormal = BitmapFactory.decodeResource(res, R.drawable.insiteo_catal_btn_iti_default);
		mItiIconBackgSelected = BitmapFactory.decodeResource(res, R.drawable.insiteo_catal_btn_iti_pressed);
		mItiIconBackg = mItiIconBackgNormal;
		mItiIconRect = new Rect();
		mItiIconSize = contentHeight;

		//compute content width
		int contentWidth = nameWidth + mItiIconSize + mTextSpace*2; 
		int imgBackgWidth = rPadding.left + contentWidth + rPadding.right;
		int imgMinWidth = mNameBackg.getMinimumWidth();
		if (imgBackgWidth < imgMinWidth) {
			contentWidth += imgMinWidth - imgBackgWidth;
		}

		mContentRect = new Rect(rPadding.left, rPadding.top, rPadding.left + contentWidth, rPadding.top + contentHeight);

		//compute background sizes
		mBackgRect = new Rect();
		mNameBackgWidth = rPadding.left + mContentRect.width() + rPadding.right;
		mNameBackgHeight = rPadding.top + mContentRect.height() + rPadding.bottom;

		//compute text vertical margin, to center it in its rectangle
		Rect txtBounds = new Rect();
		mNamePaint.getTextBounds(name, 0, name.length(), txtBounds);
		mNameVMargin = (mContentRect.height() - txtBounds.height()) / 2;

		mIsInit = true;
	}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		//offset the poi 
		int newOffsetX = aOffset.x + (int)Math.round(mZoneOffset.x * aRatio);
		int newOffsetY = aOffset.y + (int)Math.round(mZoneOffset.y * aRatio);
		Point newOffset = new Point(newOffsetX, newOffsetY);

		//rotate canvas back if needed
		if (aAngle != 0f) {
			aCanvas.save();
			aCanvas.rotate(-aAngle, newOffsetX, newOffsetY);
		}

		if (!isDrawingName) {
			drawIcon(aCanvas, newOffset, aRatio, aAngle);
		} else {
			drawName(aCanvas, newOffset, aRatio, aAngle);
		}

		//restore canvas if needed
		if (aAngle != 0f) {
			aCanvas.restore();
		}

		//save drawing values, for touch
		mLastDrawingAngle = aAngle;
		mLastDrawingPos.set(newOffsetX, newOffsetY);
	}

	//Draws the standard person icon, and draw an animated circle if name is displayed.
	public void drawIcon(Canvas aCanvas, Point aOffset, double aRatio, float aAngle) {
		if(mIsInit) {
			// draw position point
			mPoiPinRect.left = aOffset.x - mPoiPinIconWidth + mPoiPinIconHOffset;
			mPoiPinRect.top = aOffset.y - mPoiPinIconHeight + mPoiPinIconVOffset; 
			mPoiPinRect.right = mPoiPinRect.left + mPoiPinIconWidth;
			mPoiPinRect.bottom = mPoiPinRect.top + mPoiPinIconHeight;
			aCanvas.drawBitmap(mPoiPinIcon, null, mPoiPinRect, mPoiPinIconPaint);
		}
	}

	public void drawName(Canvas aCanvas, Point aOffset, double aRatio, float aAngle) {
		if (mIsNameDisplayed && mIsInit) {
			//draw name background
			mBackgRect.left = aOffset.x - mNameBackgWidth/2;
			mBackgRect.top = aOffset.y - mPoiPinIconHeight + mPoiPinIconVOffset - mNameBackgHeight;
			mBackgRect.right = mBackgRect.left + mNameBackgWidth;
			mBackgRect.bottom = mBackgRect.top + mNameBackgHeight;
			mNameBackg.setBounds(mBackgRect);
			mNameBackg.draw(aCanvas);

			Rect contentRect = new Rect(mBackgRect.left+mContentRect.left, mBackgRect.top+mContentRect.top, mBackgRect.left+mContentRect.left+mContentRect.width(), mBackgRect.top+mContentRect.top+mContentRect.height());

			//draw itinerary button
			mItiIconRect.left = contentRect.left;
			mItiIconRect.top = contentRect.top;
			mItiIconRect.right = mItiIconRect.left + mItiIconSize;
			mItiIconRect.bottom = mItiIconRect.top + mItiIconSize;
			if (mIsItiIconDisplayed) {
				aCanvas.drawBitmap(mItiIconBackg, null, mItiIconRect, null);
				aCanvas.drawBitmap(mItiIcon, null, mItiIconRect, null);
			}

			//draw name text
			int itiIconOffset = mIsItiIconDisplayed ? mItiIconSize : (mAnimateText ? 0 : mItiIconSize/2);
			int left = contentRect.left + itiIconOffset + mTextSpace;
			int top = contentRect.bottom-mNameVMargin;
			String name = mPoi.getName();
			if (mAnimateText) {
				//name must be animated
				if (mLastDrawingTime == 0) {
					//init timer if first drawing
					mLastDrawingTime = SystemClock.elapsedRealtime();
				}
				//save canvas
				aCanvas.save();
				// clip the canvas (to draw only the part of the text inside the background)
				aCanvas.clipRect(new Rect(left, contentRect.top, contentRect.right, contentRect.bottom));
				//move the text using an offset
				left -= mTextAnimOffset;
				//update the offset, depending elpased time (to have a regular animation)
				mTextAnimOffset += Math.round((float)mTextSpeed * ((SystemClock.elapsedRealtime() - mLastDrawingTime) / 1000f));
				if (mTextAnimOffset >= mNameTextLength) {
					//reset offset so that the text will take the place of the second text drawn
					mTextAnimOffset = (mTextAnimOffset % mNameTextLength) - mTextRepeatSpace;
				}
				//draw to text, one after the other (for looping effect)
				aCanvas.drawText(name, left, top, mNamePaint);
				aCanvas.drawText(name, left + mNameTextLength + mTextRepeatSpace, top, mNamePaint);
				//restore canvas
				aCanvas.restore();
				//update drawing timer
				mLastDrawingTime = SystemClock.elapsedRealtime();
			}
			else {
				//draw static text
				aCanvas.drawText(name, left, top, mNamePaint);
			}		
		}
	}

	/**
	 * Changes the "itinerary" bitmap, located on the left of the name box.
	 * You WON'T have control on the size of this bitmap. 
	 * @param aBitmap The image to set.
	 * @param aCropSquare If true, then the bitmap will be cropped to a square w/o changing its aspect ratio.
	 */
	public void setItiIcon(Bitmap aBitmap, boolean aCropSquare) {
		if(aBitmap != null) {
			mItiIcon = aBitmap;

			int w = mItiIcon.getWidth();
			int h = mItiIcon.getHeight();

			if(aCropSquare && (w != h)) {
				if(w < h) {
					mItiIcon = Bitmap.createBitmap(mItiIcon, 0, 0, w, w);
				}
				else {
					mItiIcon = Bitmap.createBitmap(mItiIcon, 0, 0, h, h);
				}

				// resize bitmap
				mItiIcon = Bitmap.createScaledBitmap(mItiIcon, w, h, true);
			}
		}
	}

	private Object3D m3DPinObject;
	private Object3D m3DNameObject;

	private float mLastAngle = 0f;
	private static final String TEXTURE = "GFX_POI_TEXTURE";

	private void create3DObject(){
		Texture color = new Texture(1, 1, RGBColor.BLUE);
		if (!TextureManager.getInstance().containsTexture(TEXTURE)) {
			TextureManager.getInstance().addTexture(TEXTURE, color);
		}

		m3DPinObject = Primitives.getSphere(2);
		m3DPinObject.setTexture(TEXTURE);

		create3DNameObject();
	}

	private void create3DNameObject(){
		Paint paint = new Paint();
		paint.setAntiAlias(true);
		paint.setTypeface(Typeface.create((String)null, Typeface.BOLD));
		paint.setTextSize(38);

		if (!TextureManager.getInstance().containsTexture("poi_" + mPoi.getId() + "_name")) {
			Texture texture = new Texture(getTextBitmap(paint, mPoi.getName()), true);		
			TextureManager.getInstance().addTexture("poi_" + mPoi.getId() + "_name", texture);
		}		

		m3DNameObject = Primitives.getPlane(26, 1);
		SimpleVector ori = m3DPinObject.getOrigin();
		m3DNameObject.rotateX((float) Math.toRadians(-45));
		m3DNameObject.setOrigin(ori);
		m3DNameObject.setTransparency(20);
		m3DNameObject.setTexture("poi_" + mPoi.getId() + "_name");
	}

	@Override
	public void draw3D(World world, Map map, double ratio, float angle) {
		if(!mIsInWorld && m3DPinObject != null){			
			mIsInWorld = true;		

			world.addObject(m3DPinObject);
			world.addObject(m3DNameObject);
		}

		SimpleVector origin = m3DPinObject.getOrigin();
		
		if(mZoneOffset != null){
			m3DPinObject.setOrigin(new SimpleVector(origin.x + mZoneOffset.x, origin.y + mZoneOffset.y, origin.z));
			m3DNameObject.setOrigin(new SimpleVector(origin.x + mZoneOffset.x, origin.y + mZoneOffset.y, origin.z - 0));
		}
		
		m3DNameObject.rotateZ((float) Math.toRadians(-mLastAngle));
		m3DNameObject.rotateZ((float) Math.toRadians(+angle));
		mLastAngle = angle;

		if (ratio < 1) {
			m3DNameObject.setVisibility(Object3D.OBJ_INVISIBLE);
		} else {
			m3DNameObject.setVisibility(Object3D.OBJ_VISIBLE);
			m3DNameObject.setScale((float) ((4/ratio < 1) ? 1 : 4/ratio));
		}

	}


	private Bitmap getTextBitmap(Paint paint, String text){
		FontMetricsInt fontMetrics = paint.getFontMetricsInt();

		//		int baseline = -fontMetrics.top;

		int width = (int)paint.measureText(text);
		int height = (int) paint.getTextSize();


		Bitmap.Config config = Bitmap.Config.ARGB_8888;
		Bitmap charImage = Bitmap.createBitmap(closestTwoPower(width), closestTwoPower(width), config);		
		Canvas canvas = new Canvas(charImage);

		paint.setColor(Color.parseColor("#474747"));
		canvas.drawText(text, (512-width)/2, /*baseline + */((256-height)/2), paint);
		paint.setColor(Color.WHITE);
		paint.setStyle(Style.STROKE);
		paint.setStrokeWidth(1);
		canvas.drawText(text, (512-width)/2, /*baseline + */((256-height)/2), paint);

		return charImage;
	}
	
	

	private int closestTwoPower(int i) {
		int power = 1;
		while (power < i) {
			power <<= 1;
		}
		return 512;//power;
	}

	private boolean mIsItiIconTouched = false;
	protected boolean mIsPoiPinTouched = false;
	private boolean mIsNameTouched = false;
	private Point mTouchRefPoint = new Point();

	@Override
	public ETouchObjectResult onTouchDown(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;

		Touch rotatedTouch = new Touch(aTouch);
		//rotate touch if needed
		if (mLastDrawingAngle != 0f) {
			rotatedTouch.rotate(mLastDrawingAngle, mLastDrawingPos.x, mLastDrawingPos.y);
		}

		//check loc icon
		if (mPoiPinRect != null && mPoiPinRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y)) {
			mIsPoiPinTouched = true;
			mTouchRefPoint.set(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y);
			result = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}
		//check itinerary rect
		else if (mIsNameDisplayed && mItiIconRect != null && mIsItiIconDisplayed && mItiIconRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition().y)) {
			mIsItiIconTouched = true;
			mTouchRefPoint.set(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y);
			mItiIconBackg = mItiIconBackgSelected;
			result = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}
		//check name rect
		else if (mIsNameDisplayed && mBackgRect != null && mBackgRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y)) {
			mIsNameTouched = true;
			mTouchRefPoint.set(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y);
			mNameBackg = mNameBackgSelected;
			result = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}
		return result;
	}

	@Override
	public ETouchObjectResult onTouchPointerDown(Touch aTouch) {

	}

	@Override
	public ETouchObjectResult onTouchMove(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;

		if (mIsPoiPinTouched || mIsItiIconTouched || mIsNameTouched) {
			if (aTouch.isMultiTouch()) {
				mIsPoiPinTouched = false;
				mIsItiIconTouched = false;
				mIsNameTouched = false;
				mItiIconBackg = mItiIconBackgNormal;
				mNameBackg = mNameBackgNormal;
			}
			else {
				Touch rotatedTouch = new Touch(aTouch);
				//rotate touch if needed
				if (mLastDrawingAngle != 0f) {
					rotatedTouch.rotate(mLastDrawingAngle, mLastDrawingPos.x, mLastDrawingPos.y);
				}

				Point currentPoint = rotatedTouch.getPosition(0);

				//if the touch position change too much, deselect person
				int dx = Math.abs(currentPoint.x - mTouchRefPoint.x);
				int dy = Math.abs(currentPoint.y - mTouchRefPoint.y);
				if ((dx > CommonConstants.CLICKABLE_ZONE_SIZE) || (dy > CommonConstants.CLICKABLE_ZONE_SIZE)) {
					mIsPoiPinTouched = false;
					mIsItiIconTouched = false;
					mIsNameTouched = false;
					mItiIconBackg = mItiIconBackgNormal;
					mNameBackg = mNameBackgNormal;
				}	   

				result = ETouchObjectResult.RESULT_CONSUME;
			}
		}

		return result;
	}

		return ETouchObjectResult.RESULT_NOTHING;

	@Override
	public ETouchObjectResult onTouchUp(Touch aTouch) {
		ETouchObjectResult result = ETouchObjectResult.RESULT_NOTHING;

		//rotate touch if needed
		Touch rotatedTouch = new Touch(aTouch);
		if (mLastDrawingAngle != 0f) {
			rotatedTouch.rotate(mLastDrawingAngle, mLastDrawingPos.x, mLastDrawingPos.y);
		}

		//check loc icon
		if (mIsPoiPinTouched && mPoiPinRect != null && mPoiPinRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y)) {
			//loc icon was clicked => toggle name infowindow
			mIsNameDisplayed = !mIsNameDisplayed;
			result = ETouchObjectResult.RESULT_CONSUME;
		}
		//check itinerary rect
		else if (mIsItiIconTouched && mIsItiIconDisplayed && mIsNameDisplayed && mItiIconRect != null && mItiIconRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y)) {
			mItineraryClicked = true;
			result = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}
		//check name rect
		else if (mIsNameTouched && mIsNameDisplayed && mBackgRect != null && mBackgRect.contains(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y)) {
			result = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}
		mIsPoiPinTouched = false;
		mIsItiIconTouched = false;
		mIsNameTouched = false;
		mItiIconBackg = mItiIconBackgNormal;
		mNameBackg = mNameBackgNormal;
		return result;
	}
