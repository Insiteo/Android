	/**
	 * @return an array of poi ID, or null if no favorite
	 */
	public static int[] getFavoriteIDs() {
		int[] favIDs = null;
		Cursor cursor = null;
		RWDBHelper instance = getInstance();

		if (checkFavoriteTable()) {
			//get all favorites IDs
			try {
				instance.lock();
				SQLiteDatabase db = instance.getDB();
				if ((db != null) && db.isOpen()) {
					cursor = db.query("Favorite", null, null, null, null, null, null);
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				instance.unlock();
			}
			if (cursor != null && cursor.moveToFirst()) {
				favIDs = new int[cursor.getCount()];
				int colId = cursor.getColumnIndexOrThrow("poiId");
				int i=0;
				do {
					favIDs[i] = cursor.getInt(colId);
					i++;
				} while (cursor.moveToNext());
			}
			if (cursor != null && !cursor.isClosed()) {
				cursor.close();
			}
		}

		return favIDs;
	}


	/**
	 * @return true if the poi is bookmarked
	 */
	public static boolean isFavorite(int poiId) {
		boolean isFav = false;
		Cursor cursor = null;
		RWDBHelper instance = getInstance();

		if (checkFavoriteTable()) {
			try {
				instance.lock();
				SQLiteDatabase db = instance.getDB();
				if ((db != null) && db.isOpen()) {
					cursor = db.query("Favorite", null, "poiId="+poiId, null, null, null, null);
					if (cursor.getCount() > 0) {
						isFav = true;
					}
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				instance.unlock();
			}

			if (cursor != null && !cursor.isClosed()) {
				cursor.close();
			}
		}

		return isFav; 
	}

	public static void insertFavorite(int poiId) {
		RWDBHelper instance = getInstance();

		if (checkFavoriteTable()) {
			try {
				instance.lock();
				SQLiteDatabase db = instance.getDB();
				if ((db != null) && db.isOpen()) {
					db.beginTransaction();

					ContentValues values = new ContentValues();
					values.put("poiId", poiId);
					long favId = db.insert("Favorite", null, values);

					if (favId != -1) {
						db.setTransactionSuccessful();
					}

					db.endTransaction();
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			} 
			finally {
				instance.unlock();
			}
		}
	}

	public static void deleteFavorite(int poiId) {
		RWDBHelper instance = getInstance();

		if (checkFavoriteTable()) {
			try {
				instance.lock();
				SQLiteDatabase db = instance.getDB();
				if ((db != null) && db.isOpen()) {
					db.delete("Favorite", "poiId="+poiId, null);
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				instance.unlock();
			}
		}
	}

	private static boolean checkFavoriteTable() {
		final String request = "CREATE TABLE IF NOT EXISTS 'Favorite' ('poiId' INTEGER PRIMARY KEY  NOT NULL)";
		boolean result = false;
		RWDBHelper instance = getInstance();

		try {
			instance.lock();
			SQLiteDatabase db = instance.getDB();
			if ((db != null) && db.isOpen()) {
				db.execSQL(request);
				result = true;
			}
		} 
		catch (SQLException ex) {
			Log.d("database", "Unable to create favorites table", ex);
			result = false;
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		} 
		finally {
			instance.unlock();
		}

		return result;
	}

	public static ArrayList<int[]> getStates(){

		ArrayList<int[]> result = new ArrayList<int[]>();

		Cursor cursor = null;

		RWDBHelper instance = getInstance();

		if(checkStateTable()){
			try {
				instance.lock();
				SQLiteDatabase db = instance.getDB();
				if ((db != null) && db.isOpen()) {
					cursor = db.rawQuery(SQLITE_GET_STATES, null);
				}
			} catch (Exception e) {
				e.printStackTrace();
			}
			finally {
				instance.unlock();

				if (cursor != null && !cursor.isClosed()) {
					if (cursor.moveToFirst()) {
						while (!cursor.isAfterLast()) {
							result.add(getValue(cursor));

						}
					}
				}
			} catch (Exception e) {
				// TODO: handle exception
			}
		}

	private static int[] getValue(Cursor cursor) {
