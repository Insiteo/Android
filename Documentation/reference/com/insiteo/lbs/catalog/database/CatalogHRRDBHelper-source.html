package com.insiteo.lbs.catalog.database;

import java.util.ArrayList;

import android.database.Cursor;

import com.insiteo.lbs.catalog.entities.CatalogEntryType;
import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.init.EPackageType;
import com.insiteo.lbs.common.init.InitProvider;

public class CatalogHRRDBHelper extends CatalogLRRDBHelper {
	
	private static CatalogHRRDBHelper sInstance;
	
	public static CatalogHRRDBHelper getInstance() {
		if (sInstance == null) {
			sInstance = new CatalogHRRDBHelper();
			InitProvider.getInstance().registerDBHelper(sInstance, EPackageType.CONTENTHRR.getDirName());
		}
		return sInstance;
	}
	
	private CatalogHRRDBHelper() {
		super();
		mPackageType = EPackageType.CONTENTHRR;
		setDBPath();
		mIsHrr = true;
	}

	@Override
	public void setDBPath() {
		mDBPath = InitProvider.getInstance().getDataDirPath() + "/" + mPackageType.getDirName() + "/" + CommonConstants.RO_DATABASE_FILE;
	}
	
	// ==================================================================================================
	// Pois
	// ==================================================================================================
	
	private static final String SQLITE_GET_POI_IDS = "SELECT poiId, sortIndex FROM CategoryPoi WHERE categId=? AND isDynamicPoi=";
	private static final String SQLITE_GET_SPONSOR_IDS = "SELECT poiId, sortIndex, bitmapName FROM Sponsor WHERE categId=? AND isDynamicPoi=";
	
	/**
	 * Return a table containing poi IDs
	 * @param aCategID id of the categ to get poi from
	 * @param isPoiLRR if true, method will search for pois from LRR referenced in given categ, if false method will search for pois from HRR referenced in given categ.
	 * @return a table containing poi IDs
	 */
	public ArrayList<CategPoiAssoc> getPoiIDs(int aCategID, boolean isPoiLRR) {
		ArrayList<CategPoiAssoc> poiIDs = new ArrayList<CategPoiAssoc>(0);
	    Cursor cursor = null;
	    
		//request database
		try {
			lock();
			if ((mDB != null) && mDB.isOpen()) {
				String request = SQLITE_GET_POI_IDS.replace("?", ""+aCategID) + (isPoiLRR ? "0" : "1");
				cursor = mDB.rawQuery(request, null);
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			unlock();
		}
		
		//parse resulting cursor
		if (cursor != null && cursor.moveToFirst()) {
			poiIDs = new ArrayList<CategPoiAssoc>(cursor.getCount());
			int colPoiId = cursor.getColumnIndexOrThrow("poiId");
			int colSortIndex = cursor.getColumnIndexOrThrow("sortIndex");
			do {
				int poiId = cursor.getInt(colPoiId);
				int sortIndex = cursor.getInt(colSortIndex);
				poiIDs.add(new CategPoiAssoc(poiId, aCategID, sortIndex, null, CatalogEntryType.POI));
			} while (cursor.moveToNext());
		}

		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
	    
	    return poiIDs;
	}
	
	/**
	 * Return a table containing sponsor IDs
	 * @param aCategID id of the categ to get poi from
	 * @param isPoiLRR if true, method will search for pois from LRR referenced in given categ, if false method will search for pois from HRR referenced in given categ.
	 * @return a table containing poi IDs
	 */
	public ArrayList<CategPoiAssoc> getSponsorIDs(int aCategID, boolean isPoiLRR) {
		ArrayList<CategPoiAssoc> poiIDs = new ArrayList<CategPoiAssoc>(0);
	    Cursor cursor = null;
	    
		//request database
		try {
			lock();
			if ((mDB != null) && mDB.isOpen()) {
				String request = SQLITE_GET_SPONSOR_IDS.replace("?", ""+aCategID) + (isPoiLRR ? "0" : "1");
				cursor = mDB.rawQuery(request, null);
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			unlock();
		}
		
		//parse resulting cursor
		if (cursor != null && cursor.moveToFirst()) {
			poiIDs = new ArrayList<CategPoiAssoc>(cursor.getCount());
			int colPoiId = cursor.getColumnIndexOrThrow("poiId");
			int colSortIndex = cursor.getColumnIndexOrThrow("sortIndex");
			int colBitmapUrl = cursor.getColumnIndexOrThrow("bitmapName");
			do {
				int poiId = cursor.getInt(colPoiId);
				int sortIndex = cursor.getInt(colSortIndex);
				String bitmapUrl = getBitmapPath(CatalogEntryType.POI, cursor.getString(colBitmapUrl));
				poiIDs.add(new CategPoiAssoc(poiId, aCategID, sortIndex, bitmapUrl, CatalogEntryType.SPONSOR));
			} while (cursor.moveToNext());
		}

		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
	    
	    return poiIDs;
	}
