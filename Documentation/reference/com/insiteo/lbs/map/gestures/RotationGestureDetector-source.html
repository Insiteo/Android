package com.insiteo.lbs.map.gestures;

import android.view.MotionEvent;

import com.insiteo.lbs.common.utils.geometry.Topology;

public class RotationGestureDetector {

	public final static String TAG = RotationGestureDetector.class.getSimpleName();

	private final static int ANGLE_DETECTION_THRESHOLD = 10;
	private final static int DISTANCE_DETECTION_THRESHOLD = 4;

	private static final int INVALID_POINTER_ID = -1;
	private float fX, fY, sX, sY;
	private int ptrID1, ptrID2;

	private boolean mIsRotationActivated = false;

	private boolean mIsRotationEnabled;

	private float mGlobalAngle = 0f, mLastAngle = 0f;

	private int mDistance = 0;

	private OnRotationGestureListener mListener;

	/**
	 * Return the last rotation value
	 * @return
	 */
	public float getAngle() {
		return - (mGlobalAngle - mLastAngle);
	}

	public boolean isRotating(){
		return mIsRotationActivated;
	}

	public RotationGestureDetector(OnRotationGestureListener listener){
		mListener = listener;
		ptrID1 = INVALID_POINTER_ID;
		ptrID2 = INVALID_POINTER_ID;
	}

	public boolean onTouchEvent(MotionEvent event){

		switch (event.getActionMasked()) {
		case MotionEvent.ACTION_DOWN:
			clearRotation();              
			if (ptrID1 == INVALID_POINTER_ID) ptrID1 = event.getPointerId(event.getActionIndex());
			break;

		case MotionEvent.ACTION_POINTER_DOWN:	
			if (ptrID2 == INVALID_POINTER_ID) ptrID2 = event.getPointerId(event.getActionIndex());			
			if (ptrID1 == INVALID_POINTER_ID) ptrID1 = event.getPointerId(event.getActionIndex());

			if(ptrID1 != INVALID_POINTER_ID && ptrID2 != INVALID_POINTER_ID) {		
				fX = event.getX(event.findPointerIndex(ptrID1));
				fY = event.getY(event.findPointerIndex(ptrID1));
				sX = event.getX(event.findPointerIndex(ptrID2));
				sY = event.getY(event.findPointerIndex(ptrID2));
			}

			mLastAngle = 0f;
			mDistance = 0;
			break;

		case MotionEvent.ACTION_MOVE:
			if (event.getPointerCount() == 2) {
				ptrID1 = event.getPointerId(0);
				ptrID2 = event.getPointerId(1);
				if(ptrID1 != INVALID_POINTER_ID && ptrID2 != INVALID_POINTER_ID) {			

					float nfX, nfY, nsX, nsY;
					nsX = event.getX(event.findPointerIndex(ptrID2));
					nsY = event.getY(event.findPointerIndex(ptrID2));
					nfX = event.getX(event.findPointerIndex(ptrID1));
					nfY = event.getY(event.findPointerIndex(ptrID1));

					mGlobalAngle = angleBetweenLines(fX, fY, sX, sY, nfX, nfY, nsX, nsY);					

					if (mListener != null && mIsRotationActivated && mIsRotationEnabled) {
						mListener.onRotation(this);
					}
					mLastAngle = mGlobalAngle;
				} 
			} 
			break;

		case MotionEvent.ACTION_POINTER_UP:
			if (event.getPointerId(event.getActionIndex()) == ptrID2) ptrID2 = INVALID_POINTER_ID;			
			if (event.getPointerId(event.getActionIndex()) == ptrID1) ptrID1 = INVALID_POINTER_ID;
			break;

		case MotionEvent.ACTION_UP:
			clearRotation();
			break;

		}
		return true;
	}

	private void clearRotation() {
		mLastAngle = 0f;
		mDistance = 0;
		ptrID1 = INVALID_POINTER_ID;
		ptrID2 = INVALID_POINTER_ID;
		mIsRotationActivated = false;
	}

	private float angleBetweenLines (float fX, float fY, float sX, float sY, float nfX, float nfY, float nsX, float nsY) {
		float angle1 = (float) Math.atan2( (fY - sY), (fX - sX) );
		float angle2 = (float) Math.atan2( (nfY - nsY), (nfX - nsX) );

		float angle = ((float)Math.toDegrees(angle1 - angle2)) % 360;
		if (angle < -180.f) angle += 360.0f;
		if (angle > 180.f) angle -= 360.0f;

		float x = nfX - nsX;
		float y = nfY - nsY;
		int newDist = (int)Topology.fastsqrt((x * x) + (y * y));
		int distOffset = Math.abs(newDist - mDistance);
		mDistance = newDist;

		if (!mIsRotationActivated && Math.abs(angle) < ANGLE_DETECTION_THRESHOLD) {
			return 0;
		}

		if (!mIsRotationActivated && distOffset > DISTANCE_DETECTION_THRESHOLD) {
			return 0;
		}

		// We can now start the rotation from the current values
		if (!mIsRotationActivated) {
			this.sX = nsX;
			this.sY = nsY;
			this.fX = nfX;
			this.fY = nfY;
			mIsRotationActivated = true;

			if (mListener != null) {
				mIsRotationEnabled = mListener.onRotationBegin();
			}

			return 0;
		}

		return angle;
	}

	public static interface OnRotationGestureListener {
		public boolean onRotationBegin();
		public boolean onRotation(RotationGestureDetector rotationDetector);
	}
}
