	/**
	 * This methods gets called when the {@link IRTO} is added to its {@link IRenderer}.
	 * It calls {@link GenericRTO#set2DResources(Resources)} or {@link GenericRTO#set3DResources(Resources)}
	 * depending the current {@link ERenderMode}.
	 */
	@Override
	public void setResources(Resources resources) {
		if (resources != null && mRenderingMode != null) {
			if (mRenderingMode == ERenderMode.RENDER_MODE_2D) {
				set2DResources(resources);
			} else {
				set3DResources(resources);
			}
		}
	}

	/**
	 * Sets the resources that should be used when drawing this {@link IRTO} in a 2D context.
	 * @param resources {@link Resources} to access {@link Bitmap}, {@link Color} and all other data from the project resource folder.
	 */
	protected void set2DResources(Resources resources){
		m2DModelBitmap = BitmapFactory.decodeResource(resources, (mPinResId != CommonConstants.NULL_ID) ? mPinResId : R.drawable.ic_insiteo_pin);

		m2DModelBitmapRect = new Rect();
		m2DModelNameRect = new Rect();
		m2DModelPaint = new Paint();

		m2DModelPaint.setTextSize(resources.getDimension(R.dimen.generic_rto_text_size));
		m2DModelPaint.setAntiAlias(true);

		mNameMargin = (int) resources.getDimension(R.dimen.generic_rto_name_margin);
		mNamePadding = (int) resources.getDimension(R.dimen.generic_rto_name_padding);

		mNameBkgColor = resources.getColor(R.color.generic_rto_name_bkg_color);
		mNameTextColor = resources.getColor(R.color.generic_rto_name_text_color);

		m2DModelBitmapWidth = (m2DModelBitmapWidth > 0) ? m2DModelBitmapWidth : m2DModelBitmap.getWidth();
		m2DModelBitmapHeight = (m2DModelBitmapHeight > 0) ? m2DModelBitmapHeight : m2DModelBitmap.getHeight();
		
		setBitmapAnchor(new Point(0, 0));
		
		setNameAnchor(new Point(-m2DModelBitmapHeight, 0));
	}

	/**
	 *  Sets the resources that should be used when drawing this {@link IRTO} in a 3D context.
	 * @param resources {@link Resources} to access {@link Bitmap}, {@link Color} and all other data from the project resource folder.

	protected void set3DResources(Resources resources){
		create3DModel();

	/**
	 * Draws this {@link GenericRTO} on the {@link MapView} {@link Canvas}.
	 */
	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		mLastRatio = aRatio;
		
		int newOffsetX;
		int newOffsetY;
		
		if(mPosition != null) {
			newOffsetX = (int) (aOffset.x + mPosition.getX() * aRatio);
			newOffsetY = (int) (aOffset.y + mPosition.getY() * aRatio);
		} else {
			newOffsetX = aOffset.x;
			newOffsetY = aOffset.y;
		}
		
		//rotate canvas back if needed
		if (aAngle != 0f) {
			aCanvas.save();
			aCanvas.rotate(-aAngle, newOffsetX, newOffsetY);
		}

		Point drawingPosition = getDrawingPosition(aRatio, aOffset);
		draw2DModelBitmap(aCanvas, drawingPosition, aAngle);
		
		if(mIsNameDisplayed) draw2DModelName(aCanvas, drawingPosition);

		//restore canvas if needed
		if (aAngle != 0f) {
			aCanvas.restore();
		}

		mLastAngle = aAngle;
		mLastPos.set(newOffsetX, newOffsetY);
	}

	/**
	 * Returns the position where the {@link GenericRTO} should be drawn.
	 * @param aRatio the ratio used when drawing.
	 * @param aOffset the drawing offset
	 * @return the {@link Point} where the {@link GenericRTO} should be drawn.
	 */
	protected Point getDrawingPosition(double aRatio, Point aOffset){
		Point drawingPosition = null;
		if (mPosition != null) {
			int drawingPosX = aOffset.x + (int) (mPosition.getX() * aRatio);
			int drawingPosY = aOffset.y + (int) (mPosition.getY() * aRatio);;

			drawingPosition = new Point(drawingPosX, drawingPosY);
		} else {
			int drawingPosX = aOffset.x;
			int drawingPosY = aOffset.y;

			drawingPosition = new Point(drawingPosX, drawingPosY);
		}
		return drawingPosition;
	}

	/**
	 * Draws the bitmap associated to this GenericRTO.
	 * @param aCanvas The destination Canvas where this object should be drawn to.
	 * @param aOffset In pixels, the offset from the (0, 0) of the drawing Canvas that should be added to this object coords.
	 * @param aAngle The offset angle in case a rotation should be done. The rotation is relative to the screen center.
	 */
	public void draw2DModelBitmap(Canvas aCanvas, Point drawingPosition, float aAngle) {
		m2DModelBitmapRect.left = drawingPosition.x - m2DModelBitmapWidth / 2 + mBitmapAnchor.x;
		m2DModelBitmapRect.top = drawingPosition.y - m2DModelBitmapHeight + mBitmapAnchor.y; 
		m2DModelBitmapRect.right = m2DModelBitmapRect.left + m2DModelBitmapWidth;
		m2DModelBitmapRect.bottom = m2DModelBitmapRect.top + m2DModelBitmapHeight;

		aCanvas.drawBitmap(m2DModelBitmap, null, m2DModelBitmapRect, m2DModelPaint);
	}

	/**
	 * Draws the name of the {@link GenericRTO} 
	 * @param canvas where the name should be drawn.
	 * @param drawingPosition the {@link GenericRTO} drawing position. The position where the name will be drawn depends on this last position and on the name 
	 * anchor (see {@link GenericRTO#setNameAnchor(Point)}).
	 */
	public void draw2DModelName(Canvas canvas, Point drawingPosition){
		if(mName != null){
			int textWidth = (int) m2DModelPaint.measureText(mName);
			int textHeight = (int) m2DModelPaint.getTextSize();

			m2DModelNameRect.bottom = drawingPosition.y + mNameAnchor.x /*- m2DModelBitmapHeight - mNameMargin */;
			m2DModelNameRect.left = drawingPosition.x - textWidth / 2 - mNamePadding + mNameAnchor.y;
			m2DModelNameRect.top = m2DModelNameRect.bottom - textHeight - 2 * mNamePadding; 
			m2DModelNameRect.right = drawingPosition.x + textWidth / 2 + mNamePadding + mNameAnchor.y;

			m2DModelPaint.setStyle(Style.FILL);
			m2DModelPaint.setColor(mNameBkgColor);
			canvas.drawRoundRect(new RectF(m2DModelNameRect), 10, 10, m2DModelPaint);

			m2DModelPaint.setStyle(Style.FILL);
			m2DModelPaint.setColor(mNameTextColor);
			canvas.drawText(mName, m2DModelNameRect.left + mNamePadding, m2DModelNameRect.bottom - mNamePadding, m2DModelPaint);
		}		
	}

	private final static int MODEL_3D_SIZE = 4;
	private final static int ANIMATION_FRAME = 10;

	// Represent the global 3D object, it has no visual representation but is used to agglomerate several 3D objects
	private ISObject3D m3DObject;

	private Object3D m3DModel;

	private RGBColor m3DModelDefaultColor;
	private RGBColor m3DModelTouchedColor;

	private Object3D m3DNameModel;
	private Texture m3DNameTexture;
	private Paint m3DNamePaint;
	private int m3DNameColor;
	private int m3DNameStrokeColor;
	private int m3DNameOffset = 4;

	private float mAnimCount = 0f;


	@Override
	public void draw3D(World world, Map map, double ratio, float angle) {
		if (m3DModel != null) {

			m3DModel.setOrigin(m3DObject.getOrigin());

			if (m3DModel != null && mPosition != null) {
				SimpleVector origin = new SimpleVector(mPosition.getX()/map.getScale(), mPosition.getY()/map.getScale(), 0);
				m3DModel.setOrigin(origin);
			}

			offset3DModel();

			set3DModelColor((mIsTouched) ? m3DModelTouchedColor : m3DModelDefaultColor);

			if (mIsAnimationActivated) animate();

			draw3DName(world, map, ratio, angle);
		}
	}

	private void offset3DModel() {
		if (mZoneOffset != null) {
			SimpleVector origin = m3DModel.getOrigin();
			m3DModel.setOrigin(new SimpleVector(origin.x + mZoneOffset.x, origin.y + mZoneOffset.y, origin.z));
		}
	}

	private void draw3DName(World world, Map map, double ratio, float angle){
		if(mIsNameDisplayed) {
			if (m3DNameModel == null) {
				create3DNameModel();
			}

			m3DNameModel.setVisibility(Object3D.OBJ_VISIBLE);

			SimpleVector origin = m3DModel.getOrigin();

			float scale = (float) (/*1+*/16/Math.ceil(ratio));

			m3DNameModel.setScale(scale);
			m3DNameModel.setOrigin(new SimpleVector(origin.x, origin.y, origin.z - MODEL_3D_SIZE / 2 - m3DNameOffset));
			m3DNameModel.rotateZ((float) Math.toRadians(-mLastAngle));
			m3DNameModel.rotateZ((float) Math.toRadians(angle));
			mLastAngle = angle;
		} else {
			if (m3DNameModel != null) {
				m3DNameModel.setVisibility(Object3D.OBJ_INVISIBLE);
			}
		}

	}

	@Override
	public ISObject3D get3DObject() {
		return m3DObject;
	}

	private void create3DModel(){
		m3DObject = new ISObject3D();
		m3DModel = ExtendedPrimitives.createCube(MODEL_3D_SIZE);
		m3DModel.setLighting(Object3D.LIGHTING_NO_LIGHTS);

		m3DObject.addChild(m3DModel);
	}

	private void create3DNameModel(){
		m3DNameModel = Primitives.getPlane(12, 1);

		m3DNamePaint = new Paint();
		m3DNamePaint.setAntiAlias(true);
		m3DNamePaint.setTypeface(Typeface.create((String)null, Typeface.BOLD));
		m3DNamePaint.setTextSize(46);

		if (!TextureManager.getInstance().containsTexture(mName)) {
			Texture texture = new Texture(getTextBitmap(m3DNamePaint, mName), true);		
			TextureManager.getInstance().addTexture(mName, texture);
			TextureManager.getInstance().compress();
		}		

		m3DNameModel.setMaxLights(0);

		m3DNameModel.rotateX((float) Math.toRadians(-45));
		m3DNameModel.setTransparency(20);
		m3DNameModel.setTexture(mName);

		m3DObject.addChild(m3DNameModel);
	}

	private void set3DModelColor(RGBColor color){
		if (m3DModel != null) {
			m3DModel.setAdditionalColor(color);
		}
	}

	private Bitmap getTextBitmap(Paint paint, String text){
		int width = (int)paint.measureText(text);
		int height = (int) paint.getTextSize();


		Bitmap.Config config = Bitmap.Config.ARGB_8888;
		Bitmap charImage = Bitmap.createBitmap(512, 512, config);		
		Canvas canvas = new Canvas(charImage);

		paint.setColor(m3DNameColor);
		canvas.drawText(text, (charImage.getWidth() - width) / 2, (charImage.getHeight() - height) / 2, paint);
		paint.setColor(m3DNameStrokeColor);
		paint.setStyle(Style.STROKE);
		paint.setStrokeWidth(1);
		canvas.drawText(text, (charImage.getWidth() - width) / 2, (charImage.getHeight() - height) / 2, paint);

		return charImage;
	}

	private void animate(){
		if (mAnimCount <= 4) {
			mAnimCount += 0.1f;
			m3DModel.translate(0, 0, -0.1f);
		} else {
			m3DModel.translate(0, 0, mAnimCount);
			mAnimCount = 0;
		}
		m3DModel.setTransparency((int) (10 + mAnimCount));
	}

	private Point mTouchPoint = new Point();
	
	private boolean mIsTouchable = true;
	private boolean mIsTouched = false;
	
	private boolean mIsDraggable = false;
	private boolean mIsDragged = false;


	@Override
	public ETouchObjectResult onTouchDown(Touch aTouch) {
		Touch rotatedTouch = new Touch(aTouch);
		// If the map is rotating the touch event needs to be rotated also
		if (mLastAngle != 0f) {
			rotatedTouch.rotate(mLastAngle, mLastPos.x, mLastPos.y);
		}

		debug.set(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y);

		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;

		if (mIsTouchable) {
			if (isTouchInsideObject(rotatedTouch)) {
				mIsTouched = true;
				
				mTouchPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);
				
				touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
			}
		} else {

		}

		return touchResult;
	}

	@Override
	public ETouchObjectResult onTouchPointerDown(Touch aTouch) {
		return ETouchObjectResult.RESULT_NOTHING;
	}

	@Override
	public ETouchObjectResult onTouchMove(Touch aTouch) {
		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;
		if(mIsTouched) {
			if(mIsDraggable && mPosition != null) {
				mIsDragged = true;
				Point currentPoint = aTouch.getPosition();
				
				int dx = currentPoint.x - mTouchPoint.x;
		        int dy = currentPoint.y - mTouchPoint.y;
		    	
	        	//move this RTO 
	        	PointD p = mPosition.getCoord();
	        	p.x += dx/mLastRatio;
	        	p.y += dy/mLastRatio;
	        	
	        	mTouchPoint = currentPoint;
	        	
	        	touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
			}
		}
		return touchResult;
	}

	@Override
	public ETouchObjectResult onTouchPointerUp(Touch aTouch) {
		return ETouchObjectResult.RESULT_NOTHING;
	}

	@Override
	public ETouchObjectResult onTouchUp(Touch aTouch) {
		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;
		Touch rotatedTouch = new Touch(aTouch);
		// If the map is rotating the touch event needs to be rotated also
		if (mLastAngle != 0f) {
			rotatedTouch.rotate(mLastAngle, mLastPos.x, mLastPos.y);
		}

		if (mIsTouched) {
			if (isTouchInsideObject(rotatedTouch)) {
				touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
			}
		}
		
		if(mIsDragged) {
			touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
		}

		mIsTouched = mIsDragged = false;
		return touchResult;
	}

	/**
	 * Checks whether the {@link Touch} is inside our {@link GenericRTO}. It calls
	 * {@link GenericRTO#isTouchInside2DObject(Touch)} or {@link GenericRTO#isTouchInside3DObject(Touch)}
	 * depending on the {@link ERenderMode}.
	 * @param touch to check
	 * @return true if the {@link Touch} is inside the {@link GenericRTO}.
	 */
	public boolean isTouchInsideObject(Touch touch){
		if(mRenderingMode == ERenderMode.RENDER_MODE_3D) {
			return isTouchInside3DObject(touch);
		} else if(mRenderingMode == ERenderMode.RENDER_MODE_2D){
			return isTouchInside2DObject(touch);
		} else {
			return false;
		}
	}

	/**
	 * Checks whether the {@link Touch} is inside our {@link GenericRTO} in a 3D context.
	 * @param touch to check.
	 * @return
	 */
	public boolean isTouchInside3DObject(Touch touch){		
		boolean isInside = false;

		if (m3DObject != null) {
			if (!touch.isMultiTouch()) {
				SimpleVector pos = touch.getTouchedRays().get(0).getPos();
				SimpleVector dir = touch.getTouchedRays().get(0).getDir();

				for (Object3D obj : m3DObject.getObjects()) {
					if (obj.rayIntersectsAABB(pos, dir, true) != Object3D.RAY_MISSES_BOX) {
						isInside = false;
						break;
					}
				}
			}
		}

		return isInside;
	}

	/**
	 * Checks whether the {@link Touch} is inside our {@link GenericRTO} in a 2D context, ie it checks if the {@link Touch} is inside the {@link RectF}
	 * where the marker is being drawn.
	 * @param touch to check.
	 * @return true if the {@link Touch} is inside.
	 */
	public boolean isTouchInside2DObject(Touch touch){
		return m2DModelBitmapRect.contains(touch.getPosition(0).x, touch.getPosition(0).y);
	}
