	/**
	 * Draws this {@link GenericRTO} on the {@link MapView} {@link Canvas}.
	 */
	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		try {
			mLock.acquire();
			mLastRatio = aRatio;

			int newOffsetX;
			int newOffsetY;

			if(mPosition != null) {
				newOffsetX = (int) (aOffset.x + mPosition.getX() * aRatio);
				newOffsetY = (int) (aOffset.y + mPosition.getY() * aRatio);
			} else {
				newOffsetX = aOffset.x;
				newOffsetY = aOffset.y;
			}

			//rotate canvas back if needed
			if (aAngle != 0f) {
				aCanvas.save();
				aCanvas.rotate(-aAngle, newOffsetX, newOffsetY);
			}

			Point drawingPosition = getDrawingPosition(aRatio, aOffset);
			drawMarker(aCanvas, drawingPosition);
			if(mIsLabelDisplayed) drawLabel(aCanvas, drawingPosition);
			if(mIsWindowDisplayed) drawWindow(aCanvas, drawingPosition);

			//restore canvas if needed
			if (aAngle != 0f) {
				aCanvas.restore();
			}

			mLastAngle = aAngle;
			mLastPos.set(newOffsetX, newOffsetY);
		} catch (InterruptedException e) {
			e.printStackTrace();
		}finally {
			mLock.release();
		}
	}

	/**
	 * Returns the position where the {@link GenericRTO} should be drawn.
	 * @param aRatio the ratio used when drawing.
	 * @param aOffset the drawing offset
	 * @return the {@link Point} where the {@link GenericRTO} should be drawn.
	 */
	protected Point getDrawingPosition(double aRatio, Point aOffset){
		Point drawingPosition = null;
		if (mPosition != null) {
			int drawingPosX = aOffset.x + (int) (mPosition.getX() * aRatio);
			int drawingPosY = aOffset.y + (int) (mPosition.getY() * aRatio);;

			drawingPosition = new Point(drawingPosX, drawingPosY);
		} else {
			int drawingPosX = aOffset.x;
			int drawingPosY = aOffset.y;

			drawingPosition = new Point(drawingPosX, drawingPosY);
		}
		return drawingPosition;
	}

	/**
	 * Draws the bitmap associated to this GenericRTO.
	 * @param aCanvas The destination Canvas where this object should be drawn to.
	 * @param aOffset In pixels, the offset from the (0, 0) of the drawing Canvas that should be added to this object coords.
	 * @param aAngle The offset angle in case a rotation should be done. The rotation is relative to the screen center.
	 */
	public void drawMarker(Canvas aCanvas, Point drawingPosition) {
		m2DIconRect.left = drawingPosition.x - mMarkerBitmap.getWidth() / 2;
		m2DIconRect.top = drawingPosition.y - mMarkerBitmap.getHeight() / 2; 
		m2DIconRect.right = m2DIconRect.left + mMarkerBitmap.getWidth();
		m2DIconRect.bottom = m2DIconRect.top + mMarkerBitmap.getHeight();

		aCanvas.drawBitmap(mMarkerBitmap, null, m2DIconRect, null);
	}

	/**
	 * Draws the window and the label of the {@link GenericRTO}.
	 * @param canvas where the name should be drawn.
	 * @param drawingPosition the {@link GenericRTO} drawing position. The position where the annotation will be drawn depends on this last position and on the name 
	 * anchor (see {@link GenericRTO#setWindowMargin(Point)}).
	 */
	public void drawWindow(Canvas canvas, Point drawingPosition){

		Rect textBounds = new Rect();
		mWindowPaint.getTextBounds(mAnnotationLabel, 0, mAnnotationLabel.length(), textBounds);
		int textWidth = textBounds.width();
		int textHeight = textBounds.height();

		int actionWidth = mIsActionDisplayed ? (2 * mActionAnchor.x + mActionBitmap.getWidth()) : 0;
		int indicatorWidth = mIsIndicatorDisplayed ? (2 * mIndicatorMargin.x + mIndicatorBitmap.getWidth()) : 0;

		int rectWidth = textWidth + 2 * mNameMargin.x + actionWidth + indicatorWidth;
		int rectHeight = Math.max(textHeight + 2 * mNameMargin.y, mActionBitmap.getHeight() + 2 * mActionAnchor.y);

		// 1 - Draw the Window background
		mWindowRect.bottom = drawingPosition.y - (mMarkerBitmap.getHeight() / 2 + mWindowAttacherSize) - mWindowMargin.x;
		mWindowRect.left = drawingPosition.x - rectWidth / 2 + mWindowMargin.y;
		mWindowRect.top = mWindowRect.bottom - rectHeight; 
		mWindowRect.right = mWindowRect.left + rectWidth;

		drawAttacher(canvas, drawingPosition, mWindowRect);

		mWindowPaint.setStyle(Style.FILL);
		mWindowPaint.setColor((mIsAnnotationTouched) ? mAnnotationBkgPressedColor : mAnnotationBkgNormalColor);
		canvas.drawRoundRect(mWindowRect, mWindowRadius, mWindowRadius, mWindowPaint);



		float textX = mWindowRect.left + actionWidth + mNameMargin.x;
		float textY = mWindowRect.bottom - ((rectHeight - textHeight) / 2);

		mWindowPaint.setColor(mAnnotationTextColor);
		canvas.drawText(mAnnotationLabel, textX, textY, mWindowPaint);

		// 2 - Draw the action icon
		if(mIsActionDisplayed) drawAction(canvas, mWindowRect);

		// 3 - Draw the indicator icon
		if(mIsIndicatorDisplayed) drawIndicator(canvas, mWindowRect);


	}

	/**
	 * Method that will draw the window attacher.
	 * @param canvas where the attacher will be drawn.
	 * @param drawingPosition is the position where the {@link IRTO} should drawn. By defaults it represents the marker's center.
	 * @param windowRect the {@link RectF} representing the window.
	 */
	protected void drawAttacher(Canvas canvas, Point drawingPosition, RectF windowRect) {
		PointF p1, p2, p3;

		p1 = new PointF(drawingPosition.x - mWindowAttacherSize / 2, windowRect.bottom  - mWindowPaint.getStrokeWidth() - 4);
		p2 = new PointF(drawingPosition.x + mWindowAttacherSize / 2, windowRect.bottom - mWindowPaint.getStrokeWidth() - 4);
		p3 = new PointF(drawingPosition.x, windowRect.bottom + mWindowAttacherSize + mWindowPaint.getStrokeWidth());

		Path path = new Path();
		path.moveTo(p1.x, p1.y);
		path.lineTo(p2.x, p2.y);
		path.lineTo(p3.x, p3.y);
		path.close();

		mWindowPaint.setStyle(Style.FILL);
		mWindowPaint.setColor((mIsAnnotationTouched) ? mAnnotationBkgPressedColor : mAnnotationBkgNormalColor);
		canvas.drawPath(path, mWindowPaint);

	}

	/**
	 * Method that will draw the action in the window.
	 * @param canvas where the action will be drawn.
	 * @param windowRect the {@link RectF} representing the window.
	 */
	protected void drawAction(Canvas canvas, RectF windowRect) {
		mActionBorderRect.bottom = mWindowRect.bottom;
		mActionBorderRect.left = mWindowRect.left;
		mActionBorderRect.top = mWindowRect.top;
		mActionBorderRect.right = mWindowRect.left + mWindowRadius * 2;

		mActionMainRect.bottom =  mWindowRect.bottom;
		mActionMainRect.left = mWindowRect.left + mWindowRadius;
		mActionMainRect.top = mWindowRect.top;
		mActionMainRect.right = mActionMainRect.left + (mActionBitmap.getWidth() + mActionAnchor.x * 2 - mWindowRadius);

		mWindowPaint.setColor((mIsActionTouched) ? mActionColorPressed : mActionColorNormal);
		canvas.drawRoundRect(mActionBorderRect, mWindowRadius, mWindowRadius, mWindowPaint);
		canvas.drawRect(mActionMainRect, mWindowPaint);

		mActionIconRect.bottom = mWindowRect.bottom - ((windowRect.height() - mActionBitmap.getHeight()) / 2);
		mActionIconRect.left =  mWindowRect.left + mActionAnchor.x;
		mActionIconRect.top = mActionIconRect.bottom - mActionBitmap.getHeight();
		mActionIconRect.right = mActionIconRect.left + mActionBitmap.getWidth();
		canvas.drawBitmap(mActionBitmap, null, mActionIconRect, null);
	}

	/**
	 * Method that will draw the indicator in the window.
	 * @param canvas where the indicator will be drawn.
	 * @param windowRect the {@link RectF} representing the window.
	 */
	protected void drawIndicator(Canvas canvas, RectF windowRect) {
		mIndicatorIconRect.bottom = mWindowRect.bottom - ((windowRect.height() - mIndicatorBitmap.getHeight()) / 2);
		mIndicatorIconRect.right =  mWindowRect.right - mIndicatorMargin.x;
		mIndicatorIconRect.top =  mIndicatorIconRect.bottom - mIndicatorBitmap.getHeight();
		mIndicatorIconRect.left =  mWindowRect.right -  mIndicatorBitmap.getWidth();
		canvas.drawBitmap(mIndicatorBitmap, null, mIndicatorIconRect, null);
	}

	/**

	private final static int MODEL_3D_SIZE = 4;

	// Represent the global 3D object, it has no visual representation but is used to agglomerate several 3D objects
	private ISObject3D m3DObject;

	@Override
	public void draw3D(World world, Map map, double ratio, float angle) {
		if (m3DModel != null) {

			m3DModel.setOrigin(m3DObject.getOrigin());

			if (m3DModel != null && mPosition != null) {
				SimpleVector origin = new SimpleVector(mPosition.getX()/map.getScale(), mPosition.getY()/map.getScale(), 0);
				m3DModel.setOrigin(origin);
			}

			offset3DModel();

			set3DModelColor((mIsMarkerTouched) ? m3DModelTouchedColor : m3DModelDefaultColor);

			if (mIsAnimationActivated) animate();

			draw3DName(world, map, ratio, angle);
		}
	}

	private void offset3DModel() {
		if (mZoneOffset != null) {
			SimpleVector origin = m3DModel.getOrigin();
			m3DModel.setOrigin(new SimpleVector(origin.x + mZoneOffset.x, origin.y + mZoneOffset.y, origin.z));
		}
	}

	private void draw3DName(World world, Map map, double ratio, float angle){
		if(mIsLabelDisplayed) {
			if (m3DNameModel == null) {
				create3DNameModel();
			}

			m3DNameModel.setVisibility(Object3D.OBJ_VISIBLE);

			SimpleVector origin = m3DModel.getOrigin();

			float scale = (float) (/*1+*/16/Math.ceil(ratio));

			m3DNameModel.setScale(scale);
			m3DNameModel.setOrigin(new SimpleVector(origin.x, origin.y, origin.z - MODEL_3D_SIZE / 2 - m3DNameOffset));
			m3DNameModel.rotateZ((float) Math.toRadians(-mLastAngle));
			m3DNameModel.rotateZ((float) Math.toRadians(angle));
			mLastAngle = angle;
		} else {
			if (m3DNameModel != null) {
				m3DNameModel.setVisibility(Object3D.OBJ_INVISIBLE);
			}
		}

	}

		return m3DObject;

	private void create3DModel(){
		m3DObject = new ISObject3D();
		m3DModel = ExtendedPrimitives.createCube(MODEL_3D_SIZE);
		m3DModel.setLighting(Object3D.LIGHTING_NO_LIGHTS);

		m3DObject.addChild(m3DModel);
	}

	private void create3DNameModel(){
		m3DNameModel = Primitives.getPlane(12, 1);

		m3DNamePaint = new Paint();
		m3DNamePaint.setAntiAlias(true);
		m3DNamePaint.setTypeface(Typeface.create((String)null, Typeface.BOLD));
		m3DNamePaint.setTextSize(46);

		if (!TextureManager.getInstance().containsTexture(mLabel)) {
			Texture texture = new Texture(getTextBitmap(m3DNamePaint, mLabel), true);		
			TextureManager.getInstance().addTexture(mLabel, texture);
			TextureManager.getInstance().compress();
		}		

		m3DNameModel.setMaxLights(0);

		m3DNameModel.rotateX((float) Math.toRadians(-45));
		m3DNameModel.setTransparency(20);
		m3DNameModel.setTexture(mLabel);

		m3DObject.addChild(m3DNameModel);
	}

	private void set3DModelColor(RGBColor color){
		if (m3DModel != null) {
			m3DModel.setAdditionalColor(color);
		}
	}

	private Bitmap getTextBitmap(Paint paint, String text){
		int width = (int)paint.measureText(text);
		int height = (int) paint.getTextSize();


		Bitmap.Config config = Bitmap.Config.ARGB_8888;
		Bitmap charImage = Bitmap.createBitmap(512, 512, config);		
		Canvas canvas = new Canvas(charImage);

		paint.setColor(m3DNameColor);
		canvas.drawText(text, (charImage.getWidth() - width) / 2, (charImage.getHeight() - height) / 2, paint);
		paint.setColor(m3DNameStrokeColor);
		paint.setStyle(Style.STROKE);
		paint.setStrokeWidth(1);
		canvas.drawText(text, (charImage.getWidth() - width) / 2, (charImage.getHeight() - height) / 2, paint);

		return charImage;
	}

	private void animate(){
		if (mAnimCount <= 4) {
			mAnimCount += 0.1f;
			m3DModel.translate(0, 0, -0.1f);
		} else {
			m3DModel.translate(0, 0, mAnimCount);
			mAnimCount = 0;
		}
		m3DModel.setTransparency((int) (10 + mAnimCount));
	}

	private Point mTouchPoint = new Point();

	protected boolean mIsTouchable = true;

	protected boolean mIsDraggable = false;
	protected boolean mIsDragged = false;

	protected volatile boolean mIsMarkerTouched = false;
	protected boolean mIsMarkerClicked = false;

	@Override
	public ETouchObjectResult onTouchDown(Touch aTouch) {
		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;
		try {
			mLock.acquire();
			Touch rotatedTouch = new Touch(aTouch);
			// If the map is rotating the touch event needs to be rotated also
			if (mLastAngle != 0f) {
				rotatedTouch.rotate(mLastAngle, mLastPos.x, mLastPos.y);
			}

			debug.set(rotatedTouch.getPosition(0).x, rotatedTouch.getPosition(0).y);


			if (mIsTouchable) {
				if (isTouchInsideMarker(rotatedTouch)) {
					mIsMarkerTouched = true;

					mTouchPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);

					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				} else if(isTouchInsideAction(rotatedTouch)) {
					mIsActionTouched = true;

					mTouchPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);
					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				} else if(isTouchInsideAnnotation(rotatedTouch)) {
					mIsAnnotationTouched = true;

					mTouchPoint.set(aTouch.getPosition().x, aTouch.getPosition().y);
					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				}
			} else {

			}

		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}

		return touchResult;
	}

	@Override
	public ETouchObjectResult onTouchMove(Touch aTouch) {
		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;
		try {
			mLock.acquire();
			Touch rotatedTouch = new Touch(aTouch);
			if (mLastAngle != 0f) {
				rotatedTouch.rotate(mLastAngle, mLastPos.x, mLastPos.y);
			}

			if(mIsMarkerTouched) {
				if(mIsDragged) {
					Point currentPoint = aTouch.getPosition();

					int dx = currentPoint.x - mTouchPoint.x;
					int dy = currentPoint.y - mTouchPoint.y;

					//move this RTO 
					PointD p = mPosition.getCoord();
					p.x += dx/mLastRatio;
					p.y += dy/mLastRatio;

					mTouchPoint = currentPoint;

					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				} else if(isTouchInsideMarker(rotatedTouch)) {
					touchResult = ETouchObjectResult.RESULT_CONSUME;

					if(mIsDraggable && mPosition != null) {
						mIsDragged = true;

						Point currentPoint = aTouch.getPosition();

						int dx = currentPoint.x - mTouchPoint.x;
						int dy = currentPoint.y - mTouchPoint.y;

						//move this RTO 
						PointD p = mPosition.getCoord();
						p.x += dx/mLastRatio;
						p.y += dy/mLastRatio;

						mTouchPoint = currentPoint;

						touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
					}
				} else {
					mIsMarkerTouched = false;
				}
			}

			if(mIsActionTouched) {
				if(isTouchInsideAction(rotatedTouch)) {
					touchResult = ETouchObjectResult.RESULT_CONSUME;
				} else {
					mIsActionTouched = false;
				}
			}

			if(mIsAnnotationTouched) {
				if(isTouchInsideAnnotation(rotatedTouch)) {
					touchResult = ETouchObjectResult.RESULT_CONSUME;
				} else {
					mIsAnnotationTouched = false;
				}
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}

		return touchResult;
	}

	@Override
	public ETouchObjectResult onTouchUp(Touch aTouch) {
		ETouchObjectResult touchResult = ETouchObjectResult.RESULT_NOTHING;
		try {
			mLock.acquire();
			Touch rotatedTouch = new Touch(aTouch);
			// If the map is rotating the touch event needs to be rotated also
			if (mLastAngle != 0f) {
				rotatedTouch.rotate(mLastAngle, mLastPos.x, mLastPos.y);
			}

			if (mIsMarkerTouched) {
				if (isTouchInsideMarker(rotatedTouch)) {
					mIsMarkerClicked = true;

					if(shouldToggleWindowOnMarkerClick()) {
						mIsWindowDisplayed = !mIsWindowDisplayed;
					}

					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				}
			} else if(mIsActionTouched) {
				if (isTouchInsideAction(rotatedTouch)) {
					mIsActionClicked = true;
					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				}
			} else if(mIsAnnotationTouched) {
				if (isTouchInsideAnnotation(rotatedTouch)) {
					mIsAnnotationClicked = true;
					touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
				}
			}

			if(mIsDragged) {
				touchResult = ETouchObjectResult.RESULT_NOTIFY_CONSUME;
			}

			mIsMarkerTouched = mIsDragged = mIsAnnotationTouched = mIsActionTouched = false;

		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			mLock.release();
		}
		return touchResult;
	}

	/**
	 * Checks whether the {@link Touch} is inside our {@link GenericRTO}. It calls
	 * {@link GenericRTO#isTouchInsideMarker(Touch)} or {@link GenericRTO#isTouchInside3DIcon(Touch)}
	 * depending on the {@link ERenderMode}.
	 * @param touch to check
	 * @return true if the {@link Touch} is inside the {@link GenericRTO}.
	 */
	public boolean isTouchInsideMarker(Touch touch){
		if(mRenderingMode == ERenderMode.RENDER_MODE_3D) {
			boolean isInside = false;
			if (m3DObject != null) {
				if (!touch.isMultiTouch()) {
					SimpleVector pos = touch.getTouchedRays().get(0).getPos();
					SimpleVector dir = touch.getTouchedRays().get(0).getDir();

					for (Object3D obj : m3DObject.getObjects()) {
						if (obj.rayIntersectsAABB(pos, dir, true) != Object3D.RAY_MISSES_BOX) {
							isInside = false;
							break;
						}
					}
				}
			}
			return isInside;
		} else if(mRenderingMode == ERenderMode.RENDER_MODE_2D){
			return m2DIconRect.contains(touch.getPosition(0).x, touch.getPosition(0).y);
		} else {
			return false;
		}
	}
