package com.insiteo.lbs.map.render;

import android.content.res.Resources;
import android.graphics.Canvas;
import android.graphics.Point;
import android.graphics.PointF;

import com.insiteo.lbs.map.entities.Map;
import com.insiteo.lbs.map.entities.Zone;
import com.insiteo.lbs.map.entities.ZonePoi;
import com.threed.jpct.World;


/**
 * Interface for interactive objects in Insiteo libraries (Renderable and Touchable Object). 
 * These objects can be drawn and touched.
 */
public interface IRTO extends Cloneable {

	/**
	 * Default ID for RTO's map. 
	 * Using this map ID will cause the RTO to never be drawn
	 */
	public static final int NO_MAP_ID = -1;

	/**
	 * ID for RTO's map, meaning it is associated with all the maps. 
	 * Using this map ID will cause the RTO to be drawn, whatever map is currently displayed
	 */
	public static final int ALL_MAP_ID = -999;

	
	/**
	 * Returns the unique ID of this object. 
	 * Beware, since 2+ objects of different classes could have the same ID !
	 * @return	A unique ID for this class.
	 */
	public int getRtoID();

	/**
	 * Returns the {@link ZonePoi} associated with this {@link IRTO}.
	 * @return the {@link ZonePoi} associated.
	 */
	public ZonePoi getZonePoi();
	
	/**
	 * Associate a {@link ZonePoi} with this {@link IRTO}. 
	 * @param zonePoi
	 */
	public void setZonePoi(ZonePoi zonePoi);
	
	/**
	 * Return the ID of this RTO's map
	 * @return The ID of the map where this RTO should be displayed
	 */
	public int getMapID();

	/**
	 * Create a copy of this RTO. The cloned RTO shall have the same ID as the original.
	 * This method should copy the behavior of Object.clone() method. 
	 * It is called when putting a single RTO in multiple zones : thus, the cloned RTO can share resources - such as bitmaps - with the original. 
	 * @return a new RTO that is the exact copy of the original, or null if object could not be cloned.
	 */
	public IRTO cloneRTO();

	/**
	 * Sets the zone offset of this IRTO. This should be used in the draw method when the IRTO is being drawn in a zone.
	 * @param zoneOffset The zone offset of this IRTO.
	 */
	public void setZoneOffset(PointF zoneOffset);

	/**
	 * Returns the zone offset of this IRTO.
	 * @return The zone offset.
	 */
	public PointF getZoneOffset();


	//******************************************************************************************************************
	//  DRAWING RESOURCES
	// *****************************************************************************************************************

	/**
	 * Sets the ERenderMode of this IRTO
	 * @param renderingMode
	 */
	public void setRenderMode(ERenderMode renderingMode);
	
	/**
	 * Return the ERenderMode that is currently used by this IRTO to be drawn.
	 * @return The ERenderMode used.
	 */
	public ERenderMode getRenderMode();
	
	/**
	 * You should call this routine when adding this object to a renderer.
	 * It'll init all this object's graphic resources. 
	 * @param aResources Resources to load drawable, strings, etc. 
	 */
	public void setResources(Resources aResources);

	/**
	 * You should call this routine when removing this object from a renderer.
	 * It'll free all this object's graphic resources. 
	 */
	public void freeResources();
	
	//******************************************************************************************************************
	//  DRAWING 2D
	// *****************************************************************************************************************

	/**
	 * Draws this object on the given Canvas. 
	 * To obtain object position on screen, do (object_position_in_meter * aRatio + aOffset).
	 * If this object has no position (ie : IRTO is put in a zone), simply draw it at aOffset.
	 * @param aCanvas	The destination Canvas where this object should be drawn to.
	 * @param aRatio	The real/screen coord ratio, this is used to transform the object's real coords into screen coords.
	 * @param aOffset	In pixels, the offset from the (0, 0) of the drawing Canvas that should be added to this object coords.
	 * @param aAngle	The offset angle in case a rotation should be done. The rotation is relative to the screen center.
	 */
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle);

	//******************************************************************************************************************
	//  DRAWING 3D
	// *****************************************************************************************************************

	/**
	 * Draws this object with its 3D mode on the given Canvas. 
	 * To obtain object position on screen, do (object_position_in_meter * aRatio + aOffset).
	 * If this object has no position (ie : IRTO is put in a zone), simply draw it at aOffset.
	 * @param world The JPCT current World.
	 * @param map The current map.
	 * @param ratio The ratio to apply to meter coordinates to transform them in screen coordinates.
	 * @param angle The angle between screen's y axis and the canvas y axis.
	 */
	public void draw3D(World world, Map map, double ratio, float angle);

	/**
	 * Returns the 3D model used by this IRTO.
	 * @return The ISObject3D used by this IRTO.
	 */
	public ISObject3D get3DObject();
	
	//******************************************************************************************************************
	//  TOUCH
	// *****************************************************************************************************************

	/**
	 * Callback for touch down event 
	 * @param aTouch the touch corresponding to the event
	 * @return a ETouchObjectResult depending the IRTO internal process 
	 */
	public ETouchObjectResult onTouchDown(Touch aTouch);

	/**
	 * Callback for touch pointer down event
	 * @param aTouch The Touch that should be handle.
	 * @return a ETouchObjectResult depending the IRTO internal process 
	 */
	public ETouchObjectResult onTouchPointerDown(Touch aTouch);

	/**
	 * Callback for touch move event 
	 * @param aTouch the touch corresponding to the event
	 * @return a ETouchObjectResult depending the IRTO internal process 
	 */
	public ETouchObjectResult onTouchMove(Touch aTouch);

	/**
	 * Callback for pointer touch up event
	 * @param aTouch The Touch that should be handle.
	 * @return a ETouchObjectResult depending the IRTO internal process 
	 */
	public ETouchObjectResult onTouchPointerUp(Touch aTouch);

	/**
	 * Callback for touch up event 
	 * @param aTouch the touch corresponding to the event
	 * @return a ETouchObjectResult depending the IRTO internal process 
	 */
	public ETouchObjectResult onTouchUp(Touch aTouch);




}
