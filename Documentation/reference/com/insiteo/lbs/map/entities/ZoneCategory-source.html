package com.insiteo.lbs.map.entities;

import android.graphics.PointF;

import com.insiteo.lbs.common.CommonConstants;
import com.insiteo.lbs.common.utils.geometry.PointD;
import com.insiteo.lbs.common.utils.geometry.Position;

/**
 * Class containing data needed to associate a zone and a Category (IDs and positions)
 */
public class ZoneCategory {

	private int mZoneID;
	private String mAssocName;
	private int mCategoryID;
	private String mCategoryExtID;
	private Position mZonePos;
	private PointF mOffset;
	
	private String mExtra1;
	private String mExtra2;
	private String mExtra3;
	private Float mFExtra1;
	
	
	/**
	 * 
	 * @param aZoneID
	 * @param aAssocName
	 * @param aCategoryID
	 * @param aCategoryExtID
	 * @param aMapID
	 * @param aCoord
	 * @param aOffset
	 */
	public ZoneCategory(int aZoneID, String aAssocName, int aCategoryID, String aCategoryExtID, int aMapID, PointD aCoord, PointF aOffset) {
		mZoneID = aZoneID;
		mAssocName = aAssocName;
		mCategoryID = aCategoryID;
		mCategoryExtID = aCategoryExtID;
		mZonePos = new Position(aMapID, aCoord);
		mOffset = aOffset;
		
		mExtra1 = null;
		mExtra2 = null;
		mExtra3 = null;
		mFExtra1 = null;
	}

	/**
	 * Returns the association {@link Zone} id.
	 * @return the {@link Zone} id.
	 */
	public int getZoneID() {
		return mZoneID;
	}

	/**
	 * Returns the association's name.
	 * @return the association's name.
	 */
	public String getAssocName() {
		return mAssocName;
	}

	/**
	 * Returns the association category id.
	 * @return the association category id.
	 */
	public int getCategoryID() {
		return mCategoryID;
	}
	
	/**
	 * Returns true if the association links to external content.
	 * @return true if it uses an external category.
	 */
	public boolean mIsExtAssoc() {
		return mCategoryID != CommonConstants.NULL_ID;
	}
	
	/**
	 * Returns the category external id.
	 * @return the category external id.
	 */
	public String getCategoryExtID() {
		return mCategoryExtID;
	}

	/**
	 * Returns the association map id.
	 * @return the association map id.
	 */
	public int getMapID() {
		return mZonePos.getMapID();
	}
	
	/**
	 * Returns the {@link Position} of this association.
	 * @return the association {@link Position}.
	 */
	public Position getZonePosition() {
		return mZonePos;
	}
	
	/**
	 * Get the distance to the zone center, for this association
	 * @return the distance to the zone center
	 */
	public PointF getOffset() {
		return mOffset;
	}
	
	/**
	 * Returns the extra 1.
	 * @return the extra 1.
	 */
	public String getExtra1() {
		return mExtra1;
	}
	

	/**
	 * Returns the extra 2.
	 * @return the extra 2.
	 */
	public String getExtra2() {
		return mExtra2;
	}
	

	/**
	 * Returns the extra 3.
	 * @return the extra 3.
	 */
	public String getExtra3() {
		return mExtra3;
	}
	

	/**
	 * Returns the {@link Float} extra 1.
	 * @return the {@link Float} extra 1.
	 */
	public Float getFExtra1() {
		return mFExtra1;
	}

	/**
	 * Sets the different extra values.
	 * @param extra1
	 * @param extra2
	 * @param extra3
	 * @param fExtra1
	 */
	public void setExtras(String extra1, String extra2, String extra3, Float fExtra1) {
		mExtra1 = extra1;
		mExtra2 = extra2;
		mExtra3 = extra3;
		mFExtra1 = fExtra1;
	}
		
}
