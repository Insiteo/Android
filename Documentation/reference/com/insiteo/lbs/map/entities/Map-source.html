	/**
	 * Constructor
	 * @param id the ID of the map
	 * @param name the name of the map
	 * @param isDebug if true, the map exists only for debug purpose
	 * @param sortIndex used to sort maps
	 * @param scale
	 * @param zoomLevels
	 * @param displayBackgroundTile
	 * @param isSlideActive
	 */
	public Map(int id, String name, boolean isDebug, int sortIndex, float scale, SparseArray<ZoomLevel> zoomLevels, boolean displayBackgroundTile, boolean isSlideActive) {
		mId = id;
		mName = name;
		mIsDebugMap = isDebug;
		mSortIndex = sortIndex;
		
		mIsOriented = false;
		mAzimuth = 0f;
		mIsGeoreferenced = false;
		mGeoMatrix = null;
		
		mTypeId = null;
		mFloor = null;
		mFloorName = null;
		mExtra1 = null;
		mExtra2 = null;
		mExtra3 = null;
		mFExtra1 = null;
		
		mScale = scale;
		mZoomLevels = zoomLevels;
		mIsSlideActive = isSlideActive;		
		mDisplayBackgroundTile = displayBackgroundTile;
		
		//set zoomMin and zoomMax : search for levels in zoomLevel list
		mZoomMin = 3000;
		mZoomMax = -1;
		
		if (zoomLevels != null) {
			for (int i=0 ; i<zoomLevels.size() ; i++) {
				int level = zoomLevels.keyAt(i);
				if (mZoomMin > level) {
					mZoomMin = level;
				}
				if (mZoomMax < level) {
					mZoomMax = level;
				}
			}
		}
		
		mDefaultZoomLevel = null;
		mStartCenterPosition = null;
		mStartViewMode = EStartViewMode.FIT_TO_SCREEN; 

		mZoneColor = null;
		mDispExtra1 = null;
		mDispExtra2 = null;
		mDispExtra3 = null;
		mDispFExtra1 = null;
	}
