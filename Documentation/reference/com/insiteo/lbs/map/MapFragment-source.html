package com.insiteo.lbs.map;

import android.app.Activity;
import android.os.Bundle;
import android.support.v4.app.Fragment;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;

import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.common.utils.geometry.Position;
import com.insiteo.lbs.map.render.EZoneAction;

/**
 * A Basic fragment component to take full advantage of Insiteo's Map API. 
 * This Fragment is based on the android.support.v4.app.Fragment version.
 * @author Insiteo
 *
 */
public class MapFragment extends Fragment implements IMapListener {

	private final static String TAG = "INSITEO_MAP_FRAGMENT";

	private IMapListener mListener;
	private MapView mMapView;

	private boolean mIsMapReady;

	//******************************************************************************************************************
	// 	FRAGMENT LIFECYCLE
	// *****************************************************************************************************************

	@Override
	public void onAttach(Activity activity) {
		try {
			mListener = (IMapListener) activity;
		} catch (ClassCastException e) {
			Log.d(TAG, "The current activity does not implement the IMapListener interface");
		}
		super.onAttach(activity);
	}

	@Override
	public void onCreate(Bundle savedInstanceState) {
		Log.d(TAG, "onCreate");
		super.onCreate(savedInstanceState);
	}

	@Override
	public View onCreateView(LayoutInflater inflater, ViewGroup container,
			Bundle savedInstanceState) {
		Log.d(TAG, "onCreateView");
		String type = getTag();
		
		mIsMapReady = false;
		
		if (type != null && type.equals("3d")) {
			mMapView = new Map3DView(getActivity());
		} else {
			mMapView = new Map2DView(getActivity());
		}

		mMapView.setListener(this);

		return mMapView;
	}

	@Override
	public void onActivityCreated(Bundle savedInstanceState) {
		Log.d(TAG, "onActivityCreated");
		super.onActivityCreated(savedInstanceState);
	}

	@Override
	public void onStart() {
		Log.d(TAG, "onStart");
		super.onStart();
	}

	@Override
	public void onResume() {
		Log.d(TAG, "onResume");
		if(mIsMapReady) mMapView.startRendering();
		super.onResume();
	}

	@Override
	public void onPause() {
		Log.d(TAG, "onPause");
		mMapView.stopRendering();
		super.onPause();
	}

	@Override
	public void onStop() {
		Log.d(TAG, "onStop");
		super.onStop();
	}

	@Override
	public void onDestroyView() {
		Log.d(TAG, "onDestroyView");
		super.onDestroyView();
	}

	@Override
	public void onDestroy() {
		Log.d(TAG, "onDestroy");
		super.onDestroy();
	}

	@Override
	public void onDetach() {
		Log.d(TAG, "onDetach");
		super.onDetach();
	}

	//******************************************************************************************************************
	// 	
	// *****************************************************************************************************************

	/**
	 * Return the MapView's instance used by this Fragment.
	 * @return The MapView's instance used.
	 */
	public MapView getMapView(){
		return mMapView;
	}

	//******************************************************************************************************************
	// 	IMapListener
	// *****************************************************************************************************************
	@Override
	public void onMapViewReady(int aMapID, String aMapName) {
		mIsMapReady = true;
		if(mListener != null) mListener.onMapViewReady(aMapID, aMapName);
	}


	@Override
	public void onMapInitFailed(String aReason) {
		if(mListener != null) mListener.onMapInitFailed(aReason);
	}


	@Override
	public void onMapChanged(int mapId, String mapName) {
		if(mListener != null) mListener.onMapChanged(mapId, mapName);

	}


	@Override
	public void onMapMoved() {
		if(mListener != null) mListener.onMapMoved();
	}


	@Override
	public void onMapClicked(Position clickedPosition) {
		if(mListener != null) mListener.onMapClicked(clickedPosition);
	}


	@Override
	public void onZoneClicked(int aZoneID, EZoneAction aActionType,
			String aActionParam) {
		if(mListener != null) mListener.onZoneClicked(aZoneID, aActionType, aActionParam);
	}


	@Override
	public void onZoomEnd(int newZoomLevel) {
		if(mListener != null) mListener.onZoomEnd(newZoomLevel);
	}


	@Override
	public void onMapReleased() {
		if(mListener != null) mListener.onMapReleased();
	}

}
