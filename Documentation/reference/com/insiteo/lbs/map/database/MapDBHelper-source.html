	private static MapDBHelper sInstance = null;
	
	/**
	 * Returns the singleton instance
	 * @return
	 */
	public static MapDBHelper getInstance() {
		if (sInstance == null) {
			sInstance = new MapDBHelper();
			InitProvider.getInstance().registerDBHelper(sInstance, EPackageType.MAPDATA.getDirName());
		}
		return sInstance;
	}
	
	private MapDBHelper() {
		super();
		setDBPath();
	}
	
	@Override
	public void setDBPath() {
		mDBPath = InitProvider.getInstance().getDataDirPath() + "/" + EPackageType.MAPDATA.getDirName() + "/" + CommonConstants.RO_DATABASE_FILE;
	}
	
	//------------------ Root map -------------------
	
	/**
	 * Get the root-map id
	 * @return the root map ID
	 */
	public static synchronized int getMapRootId() {
		int mapRootId = CommonConstants.NULL_ID;
		Cursor cursor = null;		
		RODBHelper dbHelper = MapDBHelper.getInstance();
		
		try {
			dbHelper.lock();
			SQLiteDatabase db = dbHelper.getDB();
			if ((db != null) && db.isOpen()) {
				cursor = db.query("Site", new String[] { "mapRootId" }, null, null, null, null, null);
				if ((cursor != null) && cursor.moveToFirst()) {
					mapRootId = cursor.getInt(cursor.getColumnIndexOrThrow("mapRootId"));
				}
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			dbHelper.unlock();
		}
		
		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
		
		return mapRootId;
	}
	
	//------------------ Maps -------------------
	
	/**
	 * Get a map in database : this will not fill zooms. 
	 * @param aMapId ID of the map to get
	 * @return the desired map, or null if no result for given ID 
	 */
	public static Map getMap(int aMapId) {
		Map map = null;
		Cursor cursor = null;
		RODBHelper dbHelper = MapDBHelper.getInstance();
		
		final String request = 	"SELECT * FROM Map WHERE id = " + aMapId;		
		try {
			dbHelper.lock();
			SQLiteDatabase db = dbHelper.getDB();
			if ((db != null) && db.isOpen()) {
				cursor = db.rawQuery(request, null);
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			dbHelper.unlock();
		}
		
		//parse resulting cursor, and fill the map hashmap
		if ((cursor != null) && cursor.moveToFirst()) {
			map = MapDataCreator.createMap(cursor);
		}
		
		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
		
		return map;
	}
	
	/**
	 * Get all maps in database : this will not fill zooms. 
	 * @param debug if false, maps tagged "debug" are not returned.
	 * @return an ArrayList containing maps ordered by sortIndex 
	 */
	public static ArrayList<Map> getMaps(boolean debug) {
		ArrayList<Map> maps = new ArrayList<Map>(0);
		Cursor cursor = null;
		RODBHelper dbHelper = MapDBHelper.getInstance();
		
		String debugFilter = "";
		if (debug == false) {
			debugFilter = "WHERE isDebugMap = 0 ";	//if debug mode is not activated, do not select debug maps
		}		
		
		final String request = 	"SELECT * FROM Map " + debugFilter + "ORDER BY sortIndex ASC, dispName ASC";		
		
		try {
			dbHelper.lock();
			SQLiteDatabase db = dbHelper.getDB();
			if ((db != null) && db.isOpen()) {
				cursor = db.rawQuery(request, null);
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		} finally {
			dbHelper.unlock();
		}
		
		//parse resulting cursor, and fill the map hashmap
		if ((cursor != null) && cursor.moveToFirst()) {
			maps = MapDataCreator.createMaps(cursor);
		}
		
		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
		
		return maps;
	}
	
	/**
	 * Get all maps in database : this will not fill zooms. 
	 * @param debug if false, maps tagged "debug" are not returned.
	 * @return a SparseArray containing maps indexed by MapId 
	 */
	public static SparseArray<Map> getMapsIndexed(boolean debug) {
		ArrayList<Map> maps = getMaps(debug);
		SparseArray<Map> mapsIndexed = new SparseArray<Map>(maps.size());
		for (Map map : maps) {
			mapsIndexed.put(map.getId(), map);
		}
		return mapsIndexed;
	}
	
	
	//------------------ Zooms -------------------
	
	/**
	 * Get all zoom levels for a given map
	 * @param mapId ID of the map to get zoom levels from
	 * @return a SparseArray containing zooms indexed by their level
	 */
	public static SparseArray<ZoomLevel> getZoomLevels(int mapId) {
		SparseArray<ZoomLevel> zooms = null;
		Cursor cursor = null;
		RODBHelper dbHelper = MapDBHelper.getInstance();
		
		try {
			dbHelper.lock();
			SQLiteDatabase db = dbHelper.getDB();
			if ((db != null) && db.isOpen()) {
				cursor = db.query("Zoom", null, "mapId=" + mapId, null, null, null, null);
			}
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			dbHelper.unlock();
		}
		
		//parse resulting cursor, and fill the zoom array
		if ((cursor != null) && cursor.moveToFirst()) {
			zooms = new SparseArray<ZoomLevel>(cursor.getColumnCount());
			do {
				zooms = MapDataCreator.createZoomLevel(cursor, zooms);
			} while(cursor.moveToNext());
		}

		if (cursor != null && !cursor.isClosed()) {
			cursor.close();
		}
		
		return zooms;
	}
