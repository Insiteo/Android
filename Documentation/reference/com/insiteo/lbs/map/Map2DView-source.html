package com.insiteo.lbs.map;

import android.content.Context;
import android.graphics.Canvas;
import android.graphics.PixelFormat;
import android.graphics.Point;
import android.util.AttributeSet;
import android.view.SurfaceHolder;

import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.common.utils.geometry.PointD;
import com.insiteo.lbs.map.render.IRenderer;
import com.insiteo.lbs.map.utils.CoordConverter;

/**
 * An instance of this class will create a 2D MapView 
 * @author Insiteo
 *
 */
public class Map2DView extends MapView implements SurfaceHolder.Callback{

	private final static String TAG = "Map2DView";

	protected MapDrawThread mDrawThread;

	public Map2DView(Context context) {
		super(context);
		init();
	}

	public Map2DView(Context context, AttributeSet attributes) {
		super(context, attributes);
		init();
	}

	private void init() {
		SurfaceHolder holder = getHolder();
		holder.setFormat(PixelFormat.TRANSLUCENT);
		holder.addCallback(this);
	}

	//******************************************************************************************************************
	//  Map Rendering
	// *****************************************************************************************************************

	@Override
	public void surfaceChanged(SurfaceHolder holder, int format, int width, int height) {
		initDisplay(width, height);
	}

	@Override
	public void surfaceCreated(SurfaceHolder holder) {
		mDrawThread = new MapDrawThread(holder);
		mDrawThread.start();
	}

	@Override
	public void surfaceDestroyed(SurfaceHolder holder) {
		mDrawThread.cancel();
		destroyDisplay();
	}

	private void drawMap(Canvas aCanvas) {
		boolean acq = false;
		try {
			acq = mRenderingLock.tryAcquire();
			if(acq) {

				// clear surface
				if (mBGColor == null) {
					aCanvas.drawColor(0, android.graphics.PorterDuff.Mode.CLEAR);
				} else {
					aCanvas.drawColor(mBGColor);
				}
				// compute drawing data
				PointD displayCenter = mMapDataManager.getDisplayCenter();
				float zoomLevel = mMapDataManager.getZoomController().getZoomLevel();

				// only recompute ratio while zooming
				if(mZoomReset || (zoomLevel != mMapDataManager.mLastZoomLevel)) {
					mMapDataManager.mLastRatio = CoordConverter.getRatio(zoomLevel, mMapDataManager.getMapScale());
					mZoomReset = false;
					mMapDataManager.mLastZoomLevel = zoomLevel;
				}

				Point displayOffset = CoordConverter.getDisplayOffsetAtZ(displayCenter, zoomLevel);
				mMapDataManager.mLastOffset.set(displayOffset.x, displayOffset.y);

				float angle = mMapDataManager.getRotationAngle();
				// Rotate the canvas so that everything drawn on it is rotated
				if (angle != 0f) {
					aCanvas.rotate(angle, CoordConverter.getScreenDimen().x, CoordConverter.getScreenDimen().y);
				}

				//get display orientation, in case lansdcape/portrait switching
				int displayOrientation = getScreenOrientation();

				// loop through all IRTO renderers and draw all their objects
				for(IRenderer r : mRenderers) {
					r.setCurrentMap(mMapDataManager.getMap());
					r.setDisplayOrientation(displayOrientation);
					r.draw(aCanvas, mMapDataManager.mLastRatio, mMapDataManager.mLastOffset, angle);
				}		        	
			}
		} catch(Exception e) {
			e.printStackTrace();
		} finally {
			if(acq) {
				mRenderingLock.release();
			}
		}
	}

	protected class MapDrawThread extends Thread {

		private SurfaceHolder 		mSurfaceHolder;

		private boolean mContinue 			= true;

		MapDrawThread(SurfaceHolder aSurfaceHolder) {
			mSurfaceHolder = aSurfaceHolder;
		}

		public void cancel(){
			mContinue = false;
		}

		@Override
		public void run() {
			Canvas c = null;
			
			long start = System.currentTimeMillis();
			int fps = 0;

			while (mContinue) {
				if (isRendering()) {
					try {
						c = mSurfaceHolder.lockCanvas(null);
						if (c != null) {
							synchronized (mSurfaceHolder) {
								drawMap(c);
								Thread.sleep(10);
								fps ++;
								if (System.currentTimeMillis() - start >= 1000) {
									Log.d(TAG, "FPS : " + fps);
									fps = 0;
									start = System.currentTimeMillis();
								}
							}
						}
					} catch(Exception e) {
						Log.d("draw", "Error while drawing MapView", e);
					} finally {
						if (c != null && mSurfaceHolder != null) {
							mSurfaceHolder.unlockCanvasAndPost(c);
						}
					}
				} else {
					try {
						Thread.sleep(100);
					} catch (InterruptedException e) {
						e.printStackTrace();
					}
				}
			}
		}
	}

}
