package com.insiteo.lbs.map;

import android.graphics.Color;

import com.insiteo.lbs.common.CommonConstants;

/**
 * Class holding constant map configuration parameters.
 * @author Insiteo
 *
 */
public class MapConstants {

	/**
	 * The number of virtual zoom level used in 2D mode.
	 */
	public static int MAP_2D_VIRTUAL_ZOOM = 2;
	
	/**
	 * The number of virtual zoom level used in 3D mode.
	 */
	public static int MAP_3D_VIRTUAL_ZOOM = 0;

	/**
	 * If set to true a dialog box will be displayed during the map loading.
	 */
	public static boolean SHOW_PROGRESS_ON_MAP_LOAD = false;
	
	/**
	 * If set to true the multitouch map rotation will be enabled.
	 */
	public static boolean ENABLE_ROTATION = true;
	
	/**
	 * If set to true the multitouch map zoom will be enabled.
	 */
	public static boolean ENABLE_SCALE    = true;

	/**
	 * The default color used for the map background if none are defined on the server.
	 */
	public static final int DEFAULT_MAP_BACKGROUND_COLOR = Color.WHITE;
	
	/**
	 * The default color used for map zones.
	 */
	public static final int DEFAULT_MAP_ZONE_COLOR = Color.WHITE;
	
	/**
	 * The default color used for map zones.
	 */
	public static Integer DEFAULT_ZONE_SELECTED_COLOR = Color.RED;

	//******************************************************************************************************************
	//  3D Fields
	// *****************************************************************************************************************

	// 3D configuration parameters
	/** @hide */
	public static int JPCT_BOX_SIZE = 1024;
	/** @hide */
	public static int JPCT_FAR_PLANE = 4000;
	/** @hide */
	public static int JPCT_MAX_PARENT = 4;
	/** @hide */
	public static boolean JPCT_UNLOAD_IMMEDIATELY = true;
	/** @hide */
	public static float JPCT_TRANSPARENCY_OFFSET = 0f;
	/** @hide */
	public static float JPCT_TRANSPARENCY_MULTIPLY = 1f;

	// 3D Camera parameters
	/** @hide */
	public static final float CAMERA_ZOOM_STEP = 50;
	/** @hide */
	public static final float CAMERA_MIN_DISTANCE = 2;
	/** @hide */
	public static final float CAMERA_Z_OFFSET = 2;
	/** @hide */
	public static final float CAMERA_DEFAULT_ANGLE = -45;
	/** @hide */
	public static final float CAMERA_MIN_FOV = 1.25f;
	/** @hide */
	public static final float CAMERA_MAX_FOV = 10f;
	/** @hide */
	public static final String IMMERSION_FIELD = "immersion";

	public final static float DEFAULT_ZONE_HEIGHT = 3.0f;

	//******************************************************************************************************************
	//  Hidden Fields
	// *****************************************************************************************************************

	/** @hide tile width inn pixel */
	public static final int TILE_WIDTH = 256;

	/** @hide Number of tiles in RAM cache */
	public static int TILE_CACHE_SIZE = 160;

	/** @hide */
	public static int MAP_MIN_MOVE_OFFSET = 2; // The minimum distance delta between 2 touch event to consider a move

	
	/** @hide */
	private static final String	URLKEY_MAP 			= "${map}";
	/** @hide */
	private static final String URLKEY_ZOOM			= "${zoom}";
	/** @hide */
	private static final String URLKEY_X 			= "${x}";
	/** @hide */
	private static final String URLKEY_Y 			= "${y}";

	/** @hide */
	public static final String INIT_MAPS_SERVICE 			= "/rest/api/" + CommonConstants.URLKEY_SITE_ID + "/" + 
			CommonConstants.URLKEY_APP_VERS + "/mapsinfos/" + CommonConstants.URLKEY_LANG + 
			"?IdsM=" + CommonConstants.URLKEY_MAP_ID;

	/** @hide */
	public static final String ZONES_FILE_NAME 				= "zones.bin";

	/** @hide */
	private static final String TILES_DIR					= "low/" + URLKEY_MAP + "/";
	/** @hide */
	private static final String IMAGE_TILES_NAME 			= "map" + URLKEY_MAP + "_" + URLKEY_ZOOM + "_" + URLKEY_X + "_" + URLKEY_Y + ".3cm";

	/** @hide */
	private static final String IMAGE_TILES_RESOURCES_URI	= IMAGE_TILES_NAME;
	/** @hide */
	private static final String IMAGE_TILES_FILE_PATH 		= TILES_DIR + IMAGE_TILES_NAME;


	/**
	 * @hide Returns the local path of a map tiles directory.
	 * @param aMapID
	 * @return the local path of a map tiles directory
	 */
	public static String getTilesDirPath(int aMapID) {
		return TILES_DIR.replace(URLKEY_MAP, Integer.toString(aMapID));
	}

	/**
	 * @hide Returns the server URL used to get a map image tile, depending on the position, map, and zoom level.
	 * @param aX
	 * @param aY
	 * @param aZoomLevel
	 * @param aMapID
	 * @return the server URL
	 */
	public static String getImageTileName(int aX, int aY, int aZoomLevel, int aMapID) {
		return IMAGE_TILES_NAME
				.replace(URLKEY_X, Integer.toString(aX))
				.replace(URLKEY_Y, Integer.toString(aY))
				.replace(URLKEY_ZOOM, Integer.toString(aZoomLevel))
				.replace(URLKEY_MAP, Integer.toString(aMapID));    
	}

	/**
	 * @hide Returns the path of a map image tile, depending on the position, map, and zoom level. 
	 * @param aX
	 * @param aY
	 * @param aZoomLevel
	 * @param aMapID
	 * @return the path of a map image tile
	 */
	public static String getImageTilePath(int aX, int aY, int aZoomLevel, int aMapID) {
		return IMAGE_TILES_FILE_PATH
				.replace(URLKEY_X, Integer.toString(aX))
				.replace(URLKEY_Y, Integer.toString(aY))
				.replace(URLKEY_ZOOM, Integer.toString(aZoomLevel))
				.replace(URLKEY_MAP, Integer.toString(aMapID));
	}

	/**
	 * @hide Returns the resource path of a map image tile, depending on the position, map, and zoom level. 
	 * @param aX
	 * @param aY
	 * @param aZoomLevel
	 * @param aMapID
	 * @return the resource path of a map image tile
	 */
	public static String getImageTileResource(int aX, int aY, int aZoomLevel, int aMapID) {
		return IMAGE_TILES_RESOURCES_URI
				.replace(URLKEY_X, Integer.toString(aX))
				.replace(URLKEY_Y, Integer.toString(aY))
				.replace(URLKEY_ZOOM, Integer.toString(aZoomLevel))
				.replace(URLKEY_MAP, Integer.toString(aMapID));
	}

}
