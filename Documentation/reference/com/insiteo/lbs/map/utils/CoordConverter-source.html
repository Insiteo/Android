package com.insiteo.lbs.map.utils;

import android.graphics.Point;
import android.graphics.PointF;
import android.graphics.Rect;

import com.insiteo.lbs.common.utils.Log;
import com.insiteo.lbs.common.utils.geometry.PointD;

/**
 * Converts coordinates for 3 systems of references :
 *  - screen coordinates (in pixel, with (0,0) at the top left corner of the displayed surface)
 *  - real coordinates (in meter, with (0,0) at the top left corner of the map)
 *  - map coordinates (in pixel at zoom level 0, with (0,0) at the top left corner of the map) 
 */
public class CoordConverter {

	
	private final static String TAG = "INSITEO_MAP_COORD_CONVERTER";
	
	private static Point sScreenDimen = null;	//point storing half screen width, and half screen height
	
	
	//constructor, unreachable
	private CoordConverter() {
	}
		
	/**
	 * This should be called before any use of the coord converter, in order to initialize it
	 * @param screenWidth map canvas width
	 * @param screenHeight map canvas height
	 */
	public static void setScreenDimensions(int screenWidth, int screenHeight) {
		Log.d(TAG, "ScreenDimension(" + (screenWidth/2) + ", " + (screenHeight/2) + ")");
		sScreenDimen = new Point(screenWidth / 2, screenHeight / 2);
	}
	
	
	/**
	 * Computes the "map ratio", that is a combination of the map's scale and the map's current zoom level.
	 * The result is used everywhere in the coord transformations.
	 * @param aZoomLevel
	 * @param aScale
	 * @return
	 */
	public static double getRatio(float aZoomLevel, float aScale) {
		return (Math.pow(2, aZoomLevel) / aScale);
	}
	
	/**
	 * Computes the position of map (0,0) on the screen, in pixels at z0 
	 * @param aDisplayCenter
	 * @return
	 */
	public static PointD getDisplayOffset(PointD aDisplayCenter, float aZoomLevel) {
		double ratio = Math.pow(2, aZoomLevel);
		return new PointD(sScreenDimen.x/ratio - aDisplayCenter.x, sScreenDimen.y/ratio - aDisplayCenter.y);
	}
	/**
	 * Computes the position of map (0,0) on the screen, in pixels at zoomLevel 
	 * @param aDisplayCenter
	 * @param zoomLevel
	 * @return
	 */
	public static Point getDisplayOffsetAtZ(PointD aDisplayCenter, float zoomLevel) {
		PointD centerAtZ = convertCoordForZoomLevel(aDisplayCenter, 0, zoomLevel);
		return new Point((int)(sScreenDimen.x - centerAtZ.x), (int)(sScreenDimen.y - centerAtZ.y));
	}
	
	
	/**
	 * convert coordinates in pixel from source zoom level, to dest zoom level
	 * @param pixels
	 * @param srcZoomLevel
	 * @param destZoomLevel
	 * @return
	 */
	public static PointD convertCoordForZoomLevel(PointD pixels, float srcZoomLevel, float destZoomLevel) {
    	double ratio = Math.pow(2, destZoomLevel - srcZoomLevel);
		return new PointD(pixels.x * ratio, pixels.y * ratio);
	}
    public static Rect convertCoordForZoomLevel(Rect pixels, float srcZoomLevel, float destZoomLevel) {
    	float ratio = (float)Math.pow(2, destZoomLevel - srcZoomLevel);
		return new Rect(Math.round(pixels.left * ratio), Math.round(pixels.top * ratio), Math.round(pixels.right * ratio), Math.round(pixels.bottom * ratio));
	}
    
	
	// ==================================================================================================
	// Real <=> screen
	// ==================================================================================================	
	
	
	/**
	 * Converts a raw distance into a screen distance, 
	 * in the routine above we already computed the ratio using a scale and a zoom level 
	 * @param aDistance
	 * @param aRatio
	 * @return
	 */
	public static int convertRealDistanceToScreenDistance(float aDistance, float aRatio) {
		return (int)(aDistance * aRatio);
	}
	public static float convertScreenDistanceToRealDistance(int aDistance, float aRatio) {
		return (float)(aDistance / aRatio);
	}
	
	
	public static PointD convertScreenPointToRealPoint(Point aScreenPoint, PointD aDisplayCenter, float aZoomLevel, float aScale) {
		float ratio = (float) getRatio(aZoomLevel, aScale);
		PointD displayOffset = getDisplayOffset(aDisplayCenter, aZoomLevel);
		PointD realPoint = new PointD(aScreenPoint.x / ratio - displayOffset.x, aScreenPoint.y / ratio - displayOffset.y);
		return realPoint;		
	}
	public static Point convertRealPointToScreenPoint(PointD aRealPoint, PointD aDisplayCenter, float aZoomLevel, float aScale) {
		float ratio = (float) getRatio(aZoomLevel, aScale);
		PointD displayOffset = getDisplayOffset(aDisplayCenter, aZoomLevel);
		Point screenPoint = new Point((int)((aRealPoint.x + displayOffset.x) * ratio), (int)((aRealPoint.y + displayOffset.y) * ratio));
		return screenPoint;		
	}
	
	
	// ==================================================================================================
	// Map <=> Real
	// ==================================================================================================	
		
	/**
	 * convert real distance into map coordinates (pixels at z0) 
	 * @param realDistance
	 * @param scale
	 * @return
	 */
	public static float convertRealDistanceInMapDistance(float realDistance, float scale) {
		return realDistance / scale;
    }
	public static float convertMapDistanceInRealDistance(float mapDistance, float scale) {
        return mapDistance * scale;
    }
	
	/**
	 * Converts a real point (in meters) into a map point (in pixel)
	 * @param x
	 * @param y
	 * @param scale
	 * @return
	 */
	public static PointD convertRealPointToMapPoint(double x, double y, float scale) {
		PointD mapPoint = new PointD((x / scale), (y / scale));
		return mapPoint;
    }
	
	/**
	 * Converts a map point (in pixels) into a real point (in meters)
	 * @param x
	 * @param y
	 * @param scale
	 * @return
	 */
	public static PointD convertMapPointToRealPoint(double x, double y, float scale) {
		PointD realPoint = new PointD((x * scale), (y * scale));
		return realPoint;
    }
	
	// ==================================================================================================
	// Map <=> screen
	// ==================================================================================================	
		
	/**
	 * convert screen distance into map coordinates (pixels at z0) 
	 * @param screenDistance
	 * @param zoomLevel
	 * @return
	 */
	public static float convertScreenDistanceInMapDistance(float screenDistance, float zoomLevel) {
        double ratio = Math.pow(2, zoomLevel);
		return (float)(screenDistance / ratio);
    }
	public static float convertMapDistanceInScreenDistance(float mapDistance, float zoomLevel) {
        float ratio = (float)Math.pow(2, zoomLevel);
        return mapDistance * ratio;
    }	
	public static PointF convertScreenDistanceInMapDistance(Point screenDistance, float zoomLevel) {
        float ratio = (float)Math.pow(2, zoomLevel);
        PointF mapDistance = new PointF((screenDistance.x / ratio), (screenDistance.y / ratio));
		return mapDistance;
    }
	public static PointF convertMapDistanceInScreenDistance(Point mapDistance, float zoomLevel) {
        float ratio = (float)Math.pow(2, zoomLevel);
        PointF screenDistance = new PointF((mapDistance.x * ratio), (mapDistance.y * ratio));
		return screenDistance;
    }
	
	/**
	 * convert map coordinates (pixels at z0) into screen coordinates
	 * input : pixels (float)
	 * output : pixels (integer)
	 */
	public static Point convertMapPointToScreenPoint(double x, double y, PointD aDisplayCenter, float aZoomLevel) {
		PointD displayOffset = getDisplayOffset(aDisplayCenter, aZoomLevel);
		PointD tmpPoint = new PointD(x + displayOffset.x, y + displayOffset.y);
		tmpPoint = convertCoordForZoomLevel(tmpPoint, 0, aZoomLevel);
    	return new Point(Math.round((int)tmpPoint.x), Math.round((int)tmpPoint.y));
    }
	public static PointD convertScreenPointToMapPoint(int x, int y, PointD aDisplayCenter, float aZoomLevel) {
		PointD displayOffset = getDisplayOffset(aDisplayCenter, aZoomLevel);
		PointD tmpPoint = convertCoordForZoomLevel(new PointD(x, y), aZoomLevel, 0);
		return new PointD(tmpPoint.x - displayOffset.x, tmpPoint.y - displayOffset.y);
    }
	
	
	// ==================================================================================================
	// View port
	// ==================================================================================================	
	
    /**
     * calculate the current screen rectangle in pixels at this zoomlevel
     * @param aDisplayCenter
     * @param aZoomLevel
     * @return
     */
	public static Rect getViewPort(PointD aDisplayCenter, float aZoomLevel) {
		PointD displayCenterAtZ = convertCoordForZoomLevel(aDisplayCenter, 0, aZoomLevel);
		int centerX = (int)displayCenterAtZ.x;
		int centerY = (int)displayCenterAtZ.y;
		
        return new Rect(	centerX - sScreenDimen.x, centerY - sScreenDimen.y,
        					centerX + sScreenDimen.x, centerY + sScreenDimen.y);
    }
	
	/**
	 * convert the view port at srcZoomLevel, in destZoomLevel coordinates
	 * @param viewPort
	 * @param srcZoomLevel
	 * @param destZoomLevel
	 * @return
	 */
	public static Rect convertViewPort(Rect viewPort, float srcZoomLevel, float destZoomLevel) {
    	if (srcZoomLevel == destZoomLevel) {
    		return new Rect(viewPort);
    	}
    	else {
    		return convertCoordForZoomLevel(viewPort, srcZoomLevel, destZoomLevel);
    	}
	}
    
	/**
	 * Get the zoomlevel for a given width in meter (use only with the width, ie the X axis)
	 * @param scale In meters/pixels
	 * @param realWidth In meters
	 * @return
	 */
	public static double getZoomLevelForRealViewport(float scale, double realWidth) {
	    return Math.log(sScreenDimen.x * 2 * scale / realWidth) / Math.log(2);
	}
	
	/**
	 * @return the MapView half-width and half-height (in pixels)
	 */
	public static Point getScreenDimen() {
		if (sScreenDimen == null) {
			return null;
		}
		else {
			return new Point(sScreenDimen);
		}
	}
}
