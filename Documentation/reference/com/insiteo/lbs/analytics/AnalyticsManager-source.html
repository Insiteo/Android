	private enum AnalyticsParameter{
		ANALYTICS_MESSAGE_SIZE,
		ANALYTICS_MESSAGE_FREQUENCY,
		ANALYTICS_LOCATION_FREQUENCY
	}

	private boolean setAnalyticsConfiguration(){

		boolean success = false;
		// set the services url configuration

		File configFile = new File(InitProvider.getInstance().getDataDirPath() + "/" + EPackageType.ANALYTICS.getDirName() + "/" + "analytics_config.csv");

		if(!configFile.exists()) return false;

		InputStream csvStream;
		BufferedReader bufferedReader;

		try {
			csvStream = new FileInputStream(configFile);
			bufferedReader = new BufferedReader(new InputStreamReader(csvStream));

			bufferedReader.readLine();
			mAnalyticsConfiguration = bufferedReader.readLine().split(";");

			success = true;
			bufferedReader.close();

		} catch (FileNotFoundException e) {
			Log.d(TAG, "An error occured while setting the configuration");
			e.printStackTrace();
		} catch (IOException e) {
			Log.d(TAG, "An error occured while setting the configuration");
			stop();
			e.printStackTrace();
		} finally {
			Log.d(TAG, "configuration set");
		}

		return success;
	}

	private int getParameters(AnalyticsParameter aParam){
		try{
			switch (aParam) {
			case ANALYTICS_MESSAGE_SIZE:
				return Integer.valueOf(mAnalyticsConfiguration[0]);

			case ANALYTICS_MESSAGE_FREQUENCY:
				return Integer.valueOf(mAnalyticsConfiguration[1]);

			case ANALYTICS_LOCATION_FREQUENCY:
				return Integer.valueOf(mAnalyticsConfiguration[2]);

			default:
				break;
			}
		} catch (Exception e) {
			Log.d(TAG, "An error occured while getting the parameters");
		}

		return -1;
	}

	/**
	 * @hide
	 * Method to get the Unique Device Identifier.
	 * The GUID used is a combination of multiple values:
	 *  - Settings.Secure.ANDROID_ID see {@link http://developer.android.com/reference/android/provider/Settings.Secure.html#ANDROID_ID} for details.
	 *  - getDeviceId() IMEI for GSM and the MEID or ESN for CDMA {@link http://developer.android.com/reference/android/telephony/TelephonyManager.html#getDeviceId()}
	 * @return
	 */
	public String getUUID(){
		if (mDeviceId == null && mApplicationContext != null) {
			String tmDevice = "", androidId = "";

			try {
				final TelephonyManager tm = (TelephonyManager) mApplicationContext.getSystemService(Context.TELEPHONY_SERVICE);
				tmDevice = "" + tm.getDeviceId();

				androidId = "" + android.provider.Settings.Secure.getString(mApplicationContext.getContentResolver(), android.provider.Settings.Secure.ANDROID_ID);

				UUID deviceUuid = new UUID(androidId.hashCode(), ((long)tmDevice.hashCode() << 32));
				mDeviceId = deviceUuid.toString();

			} catch (Exception e) {
				mDeviceId = "";
				e.printStackTrace();
			}
		}

		return (mDeviceId != null) ? mDeviceId : "";
	}

	/**
	 * @hide
	 * Starts the analytics process and automatically adds a new UserEvent.
	 * An Analytics package is required in order to start the analytics process
	 * @param aContext if null the process won't be started
	 */
	public void start(Context aContext){
		if (aContext != null && !mIsAnalyticsServiceStarted) {

			Log.d(TAG, "Starting Analytics API");

			boolean hasConfiguration = setAnalyticsConfiguration();

			mApplicationContext = aContext.getApplicationContext();


			mIsAnalyticsServiceStarted = hasConfiguration && AnalyticsDBHelper.getInstance().isDatabaseInPackage(); // If no configuration file Analytics won't be started

			try {
				// The request thread will only be started if the url is found. Otherwise the events will stored for further handling
				if (InitProvider.getInstance().getAnalyticsUrl() != null && mIsAnalyticsServiceStarted) {
					startAnalyticsThread();
				}
			} catch (Exception e) {
				Log.d(TAG, AnalyticsConstants.ERROR_STARTING_ANALYTICS_THREAD);
			}

			tryAddUserEvent();
		}
	}

	/**
	 * @hide
	 * Stops the analytics process. Events will not be added and sent anymore.
	 */
	public void stop(){
		Log.d(TAG, "Stopping Analytics API");
		if (isStarted()) {
			stopAnalyticsThread();
		}
		mIsAnalyticsServiceStarted = false;
	}

	/**
	 * Getter to check if the Analytics service has been started

	public boolean isStarted(){
		return mIsAnalyticsServiceStarted;
	}

	/**
	 * This method is used to add a new UserEvent in the analytics event stack. This event will be stored and then send when network connection is available.
	 * The {@link AnalyticsUserEvent} will be added only if it differs from the last one used. See {@link AnalyticsUserEvent#isEqual(AnalyticsUserEvent)}.
	 */
	public void tryAddUserEvent(){
		if (isStarted()) {
			AnalyticsUserEvent newUser = new AnalyticsUserEvent();
			if (mCurrentUserEvent == null || !mCurrentUserEvent.isEqual(newUser)) {
				Log.d(TAG, "Adding User Event : " + newUser.toString());

				try {
					mAnalyticsLock.acquire();
					AnalyticsDBHelper.getInstance().addUserEvent(newUser);
					mCurrentUserEvent = newUser;
					startAnalyticsThreadIfRequired();
				} catch (InterruptedException e) {
					e.printStackTrace();
				} finally {
					mAnalyticsLock.release();
				}
			} 
		}
	}

	/**
	 * This method is used to add a new AnalyticsGenericEvent in the analytics event stack. This event will be stored and then send when network connection is available.
	 * @param aGenericEvent the AnalyticsGenericEvent that will be sent
	 */
	public void addGenericEvent(AnalyticsGenericEvent aGenericEvent){
		if (isStarted()) {
			Log.d(TAG, "Adding Generic Event : " + aGenericEvent.toString());
			try {
				mAnalyticsLock.acquire();
				if(mListener == null || mListener.analyticsManagerShouldSendGenericEvent(aGenericEvent)) {
					AnalyticsDBHelper.getInstance().addGenericEvent(aGenericEvent);
					startAnalyticsThreadIfRequired();
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			} finally {
				mAnalyticsLock.release();
			}
		}
	}

	/**
	 * This method is used to add a new {@link AnalyticsLocationEvent} in the analytics event stack. This event will be stored and then send when network connection is available.
	 * This event gets stored only if the delay with the last {@link AnalyticsLocationEvent} stored is bigger then the frequency define if the analytics configuration file.
	 * @param aLocationEvent the AnalyticsLocationEvent that will be sent
	 */
	public void addLocationEvent(AnalyticsLocationEvent aLocationEvent){
		if (isStarted()) {
			Log.d(TAG, "Adding Location Event : " + aLocationEvent.toString());
			if (mLastAddedLocationEvent == null) {
				tryAddUserEvent();

				try {
					mAnalyticsLock.acquire();
					AnalyticsDBHelper.getInstance().addLocationEvent(aLocationEvent);
					mLastAddedLocationEvent = aLocationEvent;
					startAnalyticsThreadIfRequired();
				} catch (InterruptedException e) {
					e.printStackTrace();
				} finally {
					mAnalyticsLock.release();
				}
			} else {
				if (mLastAddedLocationEvent.getTimestamp() + getParameters(AnalyticsParameter.ANALYTICS_LOCATION_FREQUENCY) * 1000 < aLocationEvent.getTimestamp()) {
					Log.d(TAG, "new LocationEvent");
					tryAddUserEvent();

					try {
						mAnalyticsLock.acquire();
						AnalyticsDBHelper.getInstance().addLocationEvent(aLocationEvent);
						mLastAddedLocationEvent = aLocationEvent;
						startAnalyticsThreadIfRequired();
					} catch (InterruptedException e) {
						e.printStackTrace();
					} finally {
						mAnalyticsLock.release();
					}
				} else {
					Log.d(TAG, "LocationEvent postponed");
				}
			}
		}		
	}

	//******************************************************************************************************************

	private void startAnalyticsThread() {
		Log.d(TAG, "Starting analytics thread");
		mRequestsThread = new AnalyticsRequestThread();
		mRequestsThread.start();
	}

	private void startAnalyticsThreadIfRequired() {
		if(mRequestsThread == null) {
			startAnalyticsThread();
		} else {
			Log.d(TAG, "Analytics thread already started");
		}
	}

	private void stopAnalyticsThread() {
		Log.d(TAG, "Stopping analytics thread");
		if(mRequestsThread != null) {
			try {
				mRequestsThread.interrupt();
				mRequestsThread = null;
			} catch (Exception e) {
				Log.d(TAG, "Unable to stop analytics request thread");
			}
		}
	}

	private class AnalyticsRequestThread extends Thread {

		private Timer mRequestTimer = null;
		private AnalyticsMessage mAnalyticsMessage = null;
		//		private boolean mIsInterruped = false;

		@Override
		public void run() {
			super.run();
			if (mRequestTimer == null) {
				mRequestTimer = new Timer();
			}
			mRequestTimer.scheduleAtFixedRate(new TimerTask() {

				@Override
				public void run() {
					if(mAnalyticsMessage == null && !isInterrupted()){

						// If no request is pending
						// Create a new Analytics Message and fill with at most the message size of event
						// send the message 

						try {
							mAnalyticsLock.acquire();
							mAnalyticsMessage = new AnalyticsMessage();

							List<AnalyticsAbstractEvent> events  = new ArrayList<AnalyticsAbstractEvent>();
							try {
								events  = AnalyticsDBHelper.getInstance().getEvents(AnalyticsManager.this.getParameters(AnalyticsParameter.ANALYTICS_MESSAGE_SIZE));

							} catch (Exception e) {
								Log.d(TAG, "An error occured while getting the events from the database");
							}

							if (events.size() > 0) {

								Log.d(TAG, "Sending " +  events.size() + " analytics events to the server");

								mAnalyticsMessage.setEvents(events);

								AnalyticsResponseParser responseParser = new AnalyticsResponseParser();
								AnalyticsRequestEncoder requestEncoder = new AnalyticsRequestEncoder(mAnalyticsMessage);

								InsiteoError error = null;
								try{
									error = ProtoLoader.load(InitProvider.getInstance().getAnalyticsUrl(), requestEncoder, responseParser);
								} catch (Exception e) {

								}


								if (error != null && error.getCode().equals(InsiteoError.NO_ERROR_CODE)) {

									Log.d(TAG, events.size() + " analytics message were sent.");

									for (AnalyticsAbstractEvent e : events) {
										Log.d(TAG, e.toString());
									}

									Log.d(TAG, "\n\n");

									AnalyticsDBHelper.removeMessageEvents(mAnalyticsMessage);
								} else {
									stopAnalyticsThread();
								}

								mAnalyticsMessage = null;
							} else {
								Log.d(TAG, "No more events stopping thread");
								stopAnalyticsThread();
							}

							mAnalyticsMessage = null;
						} catch (InterruptedException e1) {
							e1.printStackTrace();
						} finally {
							mAnalyticsLock.release();
						}


					} else {
						// If a message is already being processed by the server NO OP
					}
				}

			}, 0, getParameters(AnalyticsParameter.ANALYTICS_MESSAGE_FREQUENCY));
		}

		@Override
		public void interrupt() {
			mRequestTimer.cancel();
			super.interrupt();
		}

	}
