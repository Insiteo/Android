package com.insiteo.common;

import android.graphics.Canvas;
import android.graphics.Point;

import com.insiteo.common.rendertouch.IRTO;
import com.threed.jpct.World;

/**
 * Interface for an Insiteo zone
 * @author Insiteo
 */
public interface IZone {

	/**
	 * Draws an {@link IRTO} inside this zone in 2D mode.
	 * @param aCanvas The destination Canvas where this object should be drawn to.
	 * @param aRatio The real/screen coord ratio, this is used to transform the object's real coords into screen coords.
	 * @param aOffset In pixels, the offset from the (0, 0) of the drawing Canvas that should be added to this object coords.
	 * @param aAngle The offset angle in case a rotation should be done. The rotation is relative to the screen center.
	 * @param aRto The RTO that should be drawn inside this zone.
	 */
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle, IRTO aRto);
	
	/**
	 * Draws an {@link IRTO} inside this zone in 3D mode.
	 * @param world The current JPCT World
	 * @param currentMap The current map
	 * @param ratio aRatio The real/screen coord ratio, this is used to transform the object's real coords into screen coords.
	 * @param angle aAngle The offset angle in case a rotation should be done. The rotation is relative to the screen center.
	 * @param rto aRto The RTO that should be drawn inside this zone.
	 */
	public void draw3D(World world, IMapData currentMap, double ratio, float angle, IRTO rto);
	
	/**
	 * Return the ID of the {@link IZone}
	 * @return The id of the {@link IZone}
	 */
	public int getID();
	
	/**
	 * Return the map ID of the {@link IZone}
	 * @return The map ID of the {@link IZone}
	 */
	public int getMapID();
}
