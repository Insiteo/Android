package com.insiteo.common.rendertouch;

import java.util.ArrayList;
import java.util.List;

import com.threed.jpct.SimpleVector;

import android.graphics.Matrix;
import android.graphics.Point;
import android.view.MotionEvent;


/**
 * Simple class that store one or more touch position 
 */
public class Touch {

	private MotionEvent mEvent;

	private List<Point> mPositions = new ArrayList<Point>();
	private List<Point> mPositionsOrig = new ArrayList<Point>();
	private List<Ray> mTouchedRays = new ArrayList<Ray>();

	/**
	 * Creates a Touch from a motion event.
	 * @param event the motion event to get touch from
	 */
	public Touch(MotionEvent event) {
		mEvent = event;
		for (int i = 0; i < event.getPointerCount(); i++) {
			mPositions.add(new Point((int) event.getX(i), (int)event.getY(i)));        	
			mPositionsOrig.add(new Point((int) event.getX(i), (int)event.getY(i)));
		}
	}


	/**
	 * Copy constructor
	 * @param aTouch the value to copy
	 */
	public Touch(Touch aTouch) {

		mEvent = aTouch.getEvent();

		mPositions.clear();
		for (int i = 0; i < aTouch.getPositions().size(); i++) {
			mPositions.add(new Point(aTouch.getPositions().get(i)));
		}

		mPositionsOrig.clear();
		for (int i = 0; i < aTouch.getPositionsOrig().size(); i++) {
			mPositionsOrig.add(new Point(aTouch.getPositionsOrig().get(i)));
		}

		mTouchedRays.clear();
		for (int i = 0; i < aTouch.getTouchedRays().size(); i++) {
			mTouchedRays.add(new Ray(aTouch.getTouchedRays().get(i)));
		}

	}

	/**
	 * Return the associated MotionEvent.
	 * @return The MotionEvent.
	 */
	public MotionEvent getEvent(){
		return mEvent;
	}

	/**
	 * Rotate the current touch.
	 * @param angle the angle to rotate, in degrees
	 * @param centerX x coordinate of the rotation center (ie : point that will not change through the rotation)
	 * @param centerY y coordinate of the rotation center (ie : point that will not change through the rotation)
	 * @return A new Touch object, with values rotated using angle and center point.
	 */
	public void rotate(float angle, float centerX, float centerY) {
		if (angle != 0) {
			//create rotation matrix
			Matrix rm = new Matrix();
			rm.setRotate(angle, centerX, centerY);

			//get touch coordinates
			float[] coords = new float[mPositions.size() * 2];

			for (int i = 0; i < mPositions.size(); i++) {
				coords[i*2] = mPositions.get(i).x;     
				coords[i*2 + 1] = mPositions.get(i).y;     
			}

			//apply rotation matrix
			rm.mapPoints(coords);

			for (int i = 0; i < mPositions.size(); i++) {
				int x = Math.round(coords[i*2]);
				int y = Math.round(coords[i*2+1]);
				setPosition(i, x, y);
			}
		}
	}

	/**
	 * Set the position for the given index. An index will represent one finger touch on the screen.
	 * @param index The index of the finger
	 * @param x The x position of this finger.
	 * @param y The y position of this finger.
	 */
	private void setPosition(int index, int x, int y){
		mPositions.get(index).set(x, y);
	}

	/**
	 * Returns the list of positions for this touch.
	 * @return The List<Point> of Point defining this TouchEvent.
	 */
	public List<Point> getPositions(){
		return mPositions;
	}
	
	/**
	 * Returns the list of positions for this touch before they were rotated with the map's rotation angle.
	 * @return The List<Point> of Point defining this TouchEvent.
	 */
	public List<Point> getPositionsOrig(){
		return mPositionsOrig;
	}

	/**
	 * Returns whether or not this Touch is considered multitouch (ie there are multiple position).
	 * @return True if it is a multi-touch event, false otherwise 
	 */
	public boolean isMultiTouch() {
		return mPositions.size() > 1;
	}
	

	/**
	 * Returns the position of the first touch.
	 * @return The position of the first touch.
	 */
	public Point getPosition() {
		Point pos = null;
		if (mPositions.size() > 0) {
			pos = mPositions.get(0);
		}
		return pos;
	}

	/**
	 * Returns the position of the touch for the finger identified by this index.
	 * @param index of the position.
	 * @return The position at this index rotated by the map's angle.
	 */
	public Point getPosition(int index) {
		Point pos = null;
		if (mPositions.size() > index) {
			pos = mPositions.get(index);
		}
		return pos;
	}

	/**
	 * Returns the position of the first touch without the map's rotation applied. Those values corresponds
	 * to the one returned by the Android OS.
	 * @return The position of the first touch.
	 */
	public Point getPositionOrig() {
		Point pos = null;
		if (mPositionsOrig.size() > 0) {
			pos = mPositionsOrig.get(0);
		}
		return pos;
	}

	/**
	 * Returns the position of the touch defined by the index without the map's rotation applied. Those values corresponds
	 * to the one returned by the Android OS.
	 * @param index of the touch's position
	 * @return The position of the original touch at this index.
	 */
	public Point getPositionOrig(int index) {
		Point pos = null;
		if (mPositionsOrig.size() > index) {
			pos = mPositionsOrig.get(index);
		}
		return pos;
	}

	/**
	 * Return the number of simultaneous touch.
	 * @return The number of simultaneous touch 
	 */
	public int countPosition() {
		return mPositions.size();
	}

	//******************************************************************************************************************
	// 3D Specific
	// All those methods are hidden at the moment.
	// *****************************************************************************************************************

	/**
	 * @hide
	 * @return
	 */
	public List<Ray> getTouchedRays(){
		return mTouchedRays;
	}

	/**
	 * @hide
	 * @param ray
	 */
	public void addRay(Ray ray){
		if (mTouchedRays != null) {
			mTouchedRays.add(ray);
		}
	}
}
