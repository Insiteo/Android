/**
 * Concrete implementation of IRenderer, that can handle any type of IRTO.

public class GenericRenderer implements IRenderer {

	private static final String TAG = GenericRenderer.class.getSimpleName();

	private int									mPriority		= 0;
	protected IMapData							mCurrentMap		= null;
	protected int								mCurrentMapID	= IRTO.NO_MAP_ID;
	protected boolean 							mIsDisplayed 	= true;
	protected boolean 							mIsTouchable	= true;
	protected boolean 							mHasMoved		= false;
	private Class<? extends IRTO> 				mInnerClass		= null;
	protected IRTOListener						mRTOListener	= null;
	protected Semaphore 						mRTOSema		= new Semaphore(1);

	private WeakReference<World> mWorld;

	protected HashMap<Integer, List<Pair<IRTO, Integer>>> mRTOs = new HashMap<Integer, List<Pair<IRTO,Integer>>>();	//associate an RTO ID, with the list of its pair RTO-zoneID

	/**
	 * Creates a renderer that can only hold objects of the given inner class.
	 * @param aClass
	 */
	public GenericRenderer(Class<? extends IRTO> aClass) {
		Log.d(TAG, "Creating GenericRenderer with class " + aClass);
		mInnerClass = aClass;
	}

	/**
	 * Set the current map. This map is used to filter RTO that should be drawn.
	 * @param aMap the currently displayed map. 
	 */
	@Override
	public void setCurrentMap(final IMapData aMap) {		
		if (aMap == null) {
			mCurrentMap = aMap;
			mCurrentMapID = IRTO.NO_MAP_ID;
		} else {
			boolean mapChanged = (mCurrentMapID != aMap.getId());
			mCurrentMapID = aMap.getId();
			mCurrentMap = aMap;
			
			if (mapChanged) onMapChanged();
		}
	}
	
	private void onMapChanged(){
		if (mRenderMode != null) {
			if (mRenderMode == ERenderMode.RENDER_MODE_3D && mWorld != null) clear3DObjectsFromWorld(mWorld.get());
		}
		
	}

	@Override
	public Class<? extends IRTO> getRTOClass() {

	@Override
	public void clear() {
		try {
			mRTOSema.acquire();
			for (List<Pair<IRTO, Integer>> rtos : mRTOs.values()) {
				for (Pair<IRTO, Integer> pair : rtos) {
					pair.first.freeResources();
				}
			}
			mRTOs.clear();
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	/**
	 * if aIgnoreSema is true, no mutex lock will be done on rto list
	 */
	protected boolean addRTOInZone(int aZoneID, IRTO aObject, boolean aIgnoreSema) {
		boolean isAdded = false;
		if(aObject.getClass() == mInnerClass) {
			int rtoID = aObject.getRtoID(); 


			try {
				if (!aIgnoreSema) mRTOSema.acquire();
				
				List<Pair<IRTO, Integer>> rtos = mRTOs.get(rtoID);
				if (rtos == null) {
					rtos = new ArrayList<Pair<IRTO,Integer>>();
					isAdded = rtos.add(new Pair<IRTO, Integer>(aObject, aZoneID));
					mRTOs.put(rtoID, rtos);
				}
				else {
					//search for this association RTO/Zone and remove it if found
					//only one zone/RTO association is possible
					for (Pair<IRTO, Integer> pair : rtos) {
						if (pair.second == aZoneID) {
							if (mRenderMode == ERenderMode.RENDER_MODE_3D) clear3DObjectFromWorld(mWorld.get(), pair.first);
							rtos.remove(pair);
							break;
						}
					}
					IRTO clonedObject = aObject.cloneRTO();
					if (clonedObject != null) {
						rtos.add(new Pair<IRTO, Integer>(clonedObject, aZoneID));
						isAdded = true;
					} else {
						Log.d("RTO", "RTO " + rtoID + " could not be cloned");
					}
				}
			} catch (InterruptedException e) {
				e.printStackTrace();
			} finally {
				if (!aIgnoreSema) mRTOSema.release();
			}
		}
		return isAdded;
	}

	@Override
	public void removeRTOFromZone(int aZoneID, IRTO aObject) {
		int rtoID = aObject.getRtoID(); 
		try {
			mRTOSema.acquire();
			List<Pair<IRTO, Integer>> rtos = mRTOs.get(rtoID);
			if (rtos != null) {
				for (Pair<IRTO, Integer> pair : rtos) {
					if (pair.second == aZoneID) {
						if (mRenderMode == ERenderMode.RENDER_MODE_3D) clear3DObjectFromWorld(mWorld.get(), pair.first);
						rtos.remove(pair);
						break;
					}
				}
				if (rtos.size() == 0) {
					mRTOs.remove(rtoID);
				}
			}
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	@Override
	public void clearZone(int aZoneId) {
		try {
			mRTOSema.acquire();
			Collection<List<Pair<IRTO, Integer>>> rtoMapValues = mRTOs.values();
			for (List<Pair<IRTO, Integer>> rtos : rtoMapValues) {
				for (Pair<IRTO, Integer> pair : rtos) {
					if (pair.second == aZoneId) {
						if (mRenderMode == ERenderMode.RENDER_MODE_3D) clear3DObjectFromWorld(mWorld.get(), pair.first);
						rtos.remove(pair);
						break;
					}
				}
				if (rtos.size() == 0) {
					rtoMapValues.remove(rtos);
				}
			}
		}
		catch (InterruptedException e) {
			e.printStackTrace();
		}
		finally {
			mRTOSema.release();
		}
	}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if(mIsDisplayed && mRTOs != null && !mRTOs.isEmpty()) {
			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					for (List<Pair<IRTO, Integer>> rtos : mRTOs.values()) {
						for (Pair<IRTO, Integer> pair : rtos) {
							IRTO r = pair.first;
							if (pair.second == CommonConstants.NULL_ID) {
								//draw the RTO using it's position. Display only RTO that are on the current map, or are associated with no map
								if ((r.getMapID() == mCurrentMapID) || (r.getMapID() == IRTO.ALL_MAP_ID)) { 
									r.draw(aCanvas, aRatio, aOffset, aAngle);
								}
							}
							else {
								//draw the RTO associated to zones
								IZone zone = InitProvider.getInstance().getZone(pair.second);
								if (zone != null && (zone.getMapID() == mCurrentMapID)) {
									zone.draw(aCanvas, aRatio, aOffset, aAngle, r);
								}
							}
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

	@Override
	public void draw3D(World world, double ratio, float angle) {
		if(mIsDisplayed && mRTOs != null && !mRTOs.isEmpty()) {

			if (mWorld == null) mWorld = new WeakReference<World>(world);

			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					for (List<Pair<IRTO, Integer>> rtos : mRTOs.values()) {
						for (Pair<IRTO, Integer> pair : rtos) {
							IRTO r = pair.first;
							if (pair.second == CommonConstants.NULL_ID) {
								//draw the RTO using it's position. Display only RTO that are on the current map, or are associated with no map
								if ((r.getMapID() == mCurrentMapID) || (r.getMapID() == IRTO.ALL_MAP_ID)) { 
									add3DObjectToWorld(mWorld.get(), r);
									r.draw3D(world, mCurrentMap, ratio, angle);
								}
							}
							else {
								//draw the RTO associated to zones
								IZone zone = InitProvider.getInstance().getZone(pair.second);
								if (zone != null && (zone.getMapID() == mCurrentMapID)) {
									add3DObjectToWorld(mWorld.get(), r);
									zone.draw3D(world, mCurrentMap, ratio, angle, r);
								}
							}
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

	private void clear3DObjectsFromWorld(World world){
		if(mIsDisplayed && mRTOs != null && !mRTOs.isEmpty() && world != null) {
			boolean acq = false;
			try {
				acq = mRTOSema.tryAcquire();
				if(acq) {
					for (List<Pair<IRTO, Integer>> rtos : mRTOs.values()) {
						for (Pair<IRTO, Integer> pair : rtos) {
							IRTO r = pair.first;
							if ((r.getMapID() != mCurrentMapID) || (r.getMapID() == IRTO.NO_MAP_ID)) { 
								clear3DObjectFromWorld(world, pair.first);
							}
						}
					}
				}
			}
			finally {
				if(acq) {
					mRTOSema.release();
				}
			}
		}
	}

		}

	@Override
	public boolean onTouchDown(Touch aTouch) {
		boolean ret = false;
		if(mIsTouchable) {
			mHasMoved = false;
			try {
				mRTOSema.acquire();
				//touch in reverse order from draw
				Object[] rtoCollec = mRTOs.values().toArray();
				for (int i=rtoCollec.length-1 ; i>=0 ; i--) {
					@SuppressWarnings("unchecked")
					List<Pair<IRTO, Integer>> rtos = (List<Pair<IRTO, Integer>>)rtoCollec[i];
					for (int j=rtos.size()-1 ; j>=0 ; j--) {
						Pair<IRTO, Integer> pair = rtos.get(j);
						IRTO r = pair.first;
						int zoneId = pair.second;

						//check if the rto is on a zone
						if (zoneId == CommonConstants.NULL_ID) {
							//check its map
							if ((r.getMapID() == mCurrentMapID) || (r.getMapID() == IRTO.ALL_MAP_ID)) { 
								ret = dispatchTouchDown(r, aTouch, null);
							}
						}
						else {
							//check its zone
							IZone zone = InitProvider.getInstance().getZone(zoneId);
							if (zone != null && (zone.getMapID() == mCurrentMapID)) {
								ret = dispatchTouchDown(r, aTouch, zone);
							}
						}
						if (ret) {
							break;
						}
					}
					if (ret) {
						break;
					}
				}
			}
			catch (InterruptedException e) {
				e.printStackTrace();
			}
			finally {
				mRTOSema.release();
			}
		}
		return ret;
	}

	protected boolean dispatchTouchDown(IRTO rto, Touch aTouch, IZone aZone) {
		boolean ret = false;
		IRTOListener listener = mRTOListener;

		switch(rto.onTouchDown(aTouch)) {
		case RESULT_CONSUME:
			// do not dispatch event, but block all other listeners
			ret = true;
			break;
		case RESULT_NOTIFY:
			// dispatch event, but do not block other listeners
			if (listener != null) {
				listener.onRTOSelected(rto, aZone);
			}
			break;
		case RESULT_NOTIFY_CONSUME:
			//  dispatch event, and block other listeners
			if (listener != null) {
				listener.onRTOSelected(rto, aZone);
			}
			ret = true;
			break;
		}
		return ret;
	}

	protected boolean dispatchTouchUp(IRTO rto, Touch aTouch, IZone aZone) {
		boolean ret = false;
		IRTOListener listener = mRTOListener;
		switch(rto.onTouchUp(aTouch)) {
		case RESULT_CONSUME:
			// do not dispatch event, but block all other listeners
			ret = true;

			// dispatch event, but do not block other listeners
			if(mHasMoved) {
				if (listener != null) {
					listener.onRTOReleased(rto, aZone);
				}
			}
			else {
				if (listener != null) {
					listener.onRTOClicked(rto, aZone);
				}
			}
			break;
		case RESULT_NOTIFY_CONSUME:
			//  dispatch event, and block other listeners
			if(mHasMoved) {
				if (listener != null) {
					listener.onRTOReleased(rto, aZone);
				}
			}
			else {
				if (listener != null) {
					listener.onRTOClicked(rto, aZone);
				}
			}
			ret = true;
			break;
		} // end switch
		return ret;
