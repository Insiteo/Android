package com.insiteo.common.utils.geom;

/**
 * Class used to convert local coordinates in meter, to lat-lon coordinates
 */
public class GeoMatrix {
	
	protected double mH11;
	protected double mH12;
	protected double mH21;
	protected double mH22;
	protected double mOffset1;
	protected double mOffset2;
	protected double mDet;
	
	public GeoMatrix(double h11, double h12, double h21, double h22, double off1, double off2) {
		mH11 = h11;
		mH12 = h12;
		mH21 = h21;
		mH22 = h22;
		mOffset1 = off1;
		mOffset2 = off2;
		mDet = mH11 * mH22 - mH12 * mH21;	
	}
	
	/**
	 * Convert given position (in meter) to lat-long (x=>long, y=>lat)
	 * @param x the x value to convert (in meter)
	 * @param y the y value to convert (in meter)
	 * @return a point containing lat-long values
	 */
	public PointD toLatLong(double x, double y) {
		double newLon = mH21 * x + mH22 * y + mOffset2;
		double newLat = mH11 * x + mH12 * y + mOffset1;
		return new PointD(newLon, newLat);
	}
	
	/**
	 * Convert given lat-long to map's coordinates (in meter)
	 * @param lat the latitude of the point to convert
	 * @param lon the longitude of the point to convert
	 * @return a point containing x-y values, in map's coordinates (in meter)
	 */
	public PointD toMetersFromLatLong(double lat, double lon) {
		double newX = (mH22 * (lat-mOffset1) - mH12 * (lon-mOffset2)) / mDet;
		double newY = (mH11 * (lon-mOffset2) - mH21 * (lat-mOffset1)) / mDet;
		return new PointD(newX, newY);
	}
	
	/**
	 * Convert given lat-long span to map's coordinates (in meter)
	 * @param latSpan the latitude span to convert
	 * @param lonSpan the longitude span to convert
	 * @return a point containing x-y span values, in map's coordinates (in meter)
	 */
	public PointD spanToMeterFromLatLong(double latSpan, double lonSpan) {
		double newX = (mH22 * (latSpan) - mH12 * (lonSpan)) / mDet;
		double newY = (mH11 * (lonSpan) - mH21 * (latSpan)) / mDet;
		return new PointD(newX, newY);
	}
	
	/**
	 * Convert given x-y (meter) span to lat-long (x=>long, y=>lat)
	 * @param xSpan the x span to convert
	 * @param ySpan the y span to convert
	 * @return a point containing lat-long span values
	 */
	public PointD spanToLatLongFromMeter(double xSpan, double ySpan) {
		double newLon = mH21 * xSpan + mH22 * ySpan;
		double newLat = mH11 * xSpan + mH12 * ySpan;
		return new PointD(newLon, newLat);
	}
