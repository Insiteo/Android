	public void setLocation(InsLocation aLoc) {
		if (mLocation != null && aLoc != null && mLocation.getMapID() == aLoc.getMapID()) {

			// If previously the location was lost we reset the proper colors
			if (isLocationLost) {
				isLocationLost = false;
				setDrawingColors();
			}

			//animate location from old point to new one
			//compute distance between the two positions
			final double xDistance = (aLoc.getCoord().x - mLocation.getCoord().x) / CommonConstants.COS_SQUARE_THETA_SUM;
			final double yDistance = (aLoc.getCoord().y - mLocation.getCoord().y) / CommonConstants.COS_SQUARE_THETA_SUM;
			//set new location with old position
			aLoc.setCoord(mLocation.getCoord());
			mLocation = aLoc;
			//get a reference on the position object, and run a thread to update it
			final PointD coord = mLocation.getCoord();
			new Thread(new Runnable() {
				@Override public void run() {
					for (double d : CommonConstants.COS_SQUARE_THETA) {
						coord.x += xDistance * d;
						coord.y += yDistance * d;
						try {
							Thread.sleep(10);
						} 
						catch (InterruptedException e) {
							e.printStackTrace();
						}
					}
				}
			}).start();
		}
		else {
			mLocation = aLoc;
		}
		mIsLocUpdated = true;
	}

	public void setLocationLost(InsLocation aLastKnownLoc){
		if (mLocation != null && aLastKnownLoc != null && mLocation.getMapID() == aLastKnownLoc.getMapID()) {

			if (!isLocationLost) {

			//animate location from old point to new one
			//compute distance between the two positions
			final double xDistance = (aLastKnownLoc.getCoord().x - mLocation.getCoord().x) / CommonConstants.COS_SQUARE_THETA_SUM;
			final double yDistance = (aLastKnownLoc.getCoord().y - mLocation.getCoord().y) / CommonConstants.COS_SQUARE_THETA_SUM;
			//set new location with old position
			aLastKnownLoc.setCoord(mLocation.getCoord());
			mLocation = aLastKnownLoc;
			//get a reference on the position object, and run a thread to update it

			new Thread(new Runnable() {

			}).start();
		}
		else {
			mLocation = aLastKnownLoc;
		}

	@Override
	public void draw(Canvas aCanvas, double aRatio, Point aOffset, float aAngle) {
		if (mLocation != null && mImage != null && mFovImage != null) {
			//calculate position on screen
			int posX =  (int)(mLocation.getCoord().x * aRatio) + aOffset.x;
			int posY =  (int)(mLocation.getCoord().y * aRatio) + aOffset.y;
			float accRadius = (float)(mLocation.getAccuracy() * aRatio);

			aCanvas.save();

			//draw accuracy circle
			//			float radiusRatio = (SystemClock.elapsedRealtime() % ANIM_DURATION) / (float)ANIM_DURATION;
			//    		float alphaRatio = 1f - radiusRatio;
			//	    	mAccAnimPaint.setAlpha((int)(ANIM_MAX_ALPHA * alphaRatio));
			//    		aCanvas.drawCircle(posX, posY, accRadius * radiusRatio, mAccAnimPaint);

			aCanvas.drawCircle(posX, posY, accRadius, mAccFillPaint);
			aCanvas.drawCircle(posX, posY, accRadius, mAccBorderPaint);

			//if azimuth is set, draw fov image
			if (mAzimuth != null && mMapAzimuth != null) {
				//compute angle for drawing (used by derived classes
				float angle = computeDrawingAzimuth();

				Rect rect = new Rect(posX-mFovHalfWidth, posY-mFovHalfHeight, posX+mFovHalfWidth, posY+mFovHalfHeight);
				aCanvas.rotate(angle - mMapAzimuth + mDisplayAngle, posX, posY);
				aCanvas.drawBitmap(mFovImage, null, rect, null);
			}

			aCanvas.restore();

			//draw location point
			Rect rect = new Rect(posX-mImageHalfWidth, posY-mImageHalfHeight, posX+mImageHalfWidth, posY+mImageHalfHeight);
			aCanvas.drawBitmap(mImage, null, rect, mPaint);
		}
	}

	private boolean mIsInWorld;
	private Object3D m3DObject;
	private static final String TEXTURE = "GFX_LOCATION_TEXTURE";
	private boolean mIsLocUpdated = false;

	private void create3DObject(){
		Texture color = new Texture(1, 1, ColorUtils.getRGBColor(mLocationColor));
		if (!TextureManager.getInstance().containsTexture(TEXTURE)) {
			TextureManager.getInstance().addTexture(TEXTURE, color);
		}

		m3DObject = Primitives.getSphere(10);
		m3DObject.setTexture(TEXTURE);
	}

	@Override
	public void draw3D(World world, IMapData map, double ratio, float angle) {
		if (!mIsInWorld) {
			mIsInWorld = true;
			world.addObject(m3DObject);
		}
		
		if (mIsLocUpdated) {
			mIsLocUpdated = false;
			int posX =  (int)(mLocation.getCoord().x / map.getMapScale());
			int posY =  (int)(mLocation.getCoord().y / map.getMapScale());
			
			float r = (ratio < 1) ? 1 : (int) ratio;
			Log.d(TAG, "value = " + r + " | " + ratio + " ---> " + (1/r));
			m3DObject.setScale((float) (1/r));
			m3DObject.setOrigin(new SimpleVector(posX, posY, 0));
			
			Enumeration<Object3D> objs = world.getObjects();
			
			while (objs.hasMoreElements()) {
				Object3D obj = objs.nextElement();
				if (obj.getName().equals(CommonConstants.ZONE_OBJECT_3D_NAME)) {
					if (obj.sphereIntersectsAABB(m3DObject.getOrigin(), 10*m3DObject.getScale())) {
						obj.setUserObject(true);//setTransparency(10);
					} else {
//						obj.setUserObject(false);
					}
				}
				
			}
			
			
		}
		
		
	}
